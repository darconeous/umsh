1//! Power-loss-safe BLE security snapshot encoding.
2//!
3//! Fixed-size snapshots (pairing PIN, local IRK, bonds) written through
4//! the shared [`record`](crate::record) engine. Which two pages the
5//! journal owns is the firmware's memory-map decision.
6
7use crate::record::{crc32, generation_is_newer};
8
9pub const MAX_BONDS: usize = 4;
10pub const SLOT_SIZE: usize = 256;
11pub const COMMIT_OFFSET: usize = SLOT_SIZE - 4;
12const CRC_OFFSET: usize = COMMIT_OFFSET - 4;
13const MAGIC: [u8; 4] = *b"UBLS";
14// Version 1 may contain a bond captured at the first protected GATT edge,
15// before SMP identity-key distribution completed. Do not restore those
16// incomplete records.
17const VERSION: u8 = 3;
18const BOND_SIZE: usize = 44;
19const LOCAL_IRK_OFFSET: usize = 16;
20const BONDS_OFFSET: usize = 32;
21
22#[derive(Clone, Copy, Debug, PartialEq, Eq)]
23pub struct StoredBond {
24    pub address_kind: u8,
25    pub address: [u8; 6],
26    pub irk: Option<[u8; 16]>,
27    pub ltk: [u8; 16],
28    pub security_level: u8,
29    pub is_bonded: bool,
30}
31
32/// Outcome of [`upsert_bond`].
33#[derive(Clone, Copy, Debug, PartialEq, Eq)]
34pub enum BondUpsert {
35    /// The bond's content and position were already current.
36    Unchanged,
37    /// An existing bond's content and/or MRU position changed.
38    Updated,
39    /// A new bond was appended, evicting the LRU entry first if the list
40    /// was already at `MAX_BONDS`.
41    Inserted { evicted: Option<StoredBond> },
42}
43
44/// Insert or refresh `bond` in `bonds`, which is kept ordered
45/// least-recently-used-first / most-recently-used-last (index 0 is the
46/// eviction candidate). A bond matching by `address_kind` + `address` is
47/// refreshed in place and moved to the MRU end; a new bond is appended at
48/// the MRU end, evicting the current LRU entry first if the list is full.
49pub fn upsert_bond(bonds: &mut heapless::Vec<StoredBond, MAX_BONDS>, bond: StoredBond) -> BondUpsert {
50    if let Some(index) = bonds
51        .iter()
52        .position(|existing| existing.address_kind == bond.address_kind && existing.address == bond.address)
53    {
54        if bonds[index] == bond && index == bonds.len() - 1 {
55            return BondUpsert::Unchanged;
56        }
57        bonds.remove(index);
58        let _ = bonds.push(bond);
59        return BondUpsert::Updated;
60    }
61    let evicted = if bonds.len() == MAX_BONDS {
62        Some(bonds.remove(0))
63    } else {
64        None
65    };
66    let _ = bonds.push(bond);
67    BondUpsert::Inserted { evicted }
68}
69
70/// Move the bond matching `address_kind` + `address` to the MRU end, if
71/// present and not already there. Returns `true` when the order changed
72/// (and the caller should persist), `false` when the bond was already MRU
73/// or is not stored at all.
74pub fn touch_bond(bonds: &mut heapless::Vec<StoredBond, MAX_BONDS>, address_kind: u8, address: [u8; 6]) -> bool {
75    let Some(index) = bonds
76        .iter()
77        .position(|existing| existing.address_kind == address_kind && existing.address == address)
78    else {
79        return false;
80    };
81    if index == bonds.len() - 1 {
82        return false;
83    }
84    let bond = bonds.remove(index);
85    let _ = bonds.push(bond);
86    true
87}

89#[derive(Clone, Debug, PartialEq, Eq)]
90pub struct Snapshot {
91    pub generation: u32,
92    pub pin: Option<u32>,
93    pub local_irk: Option<[u8; 16]>,
94    pub bonds: heapless::Vec<StoredBond, MAX_BONDS>,
95}
96
97impl Snapshot {
98    pub const fn empty() -> Self {
99        Self {
100            generation: 0,
101            pin: None,
102            local_irk: None,
103            bonds: heapless::Vec::new(),
104        }
105    }
106
107    pub fn encode(&self) -> [u8; SLOT_SIZE] {
108        let mut out = [0xff; SLOT_SIZE];
109        out[..4].copy_from_slice(&MAGIC);
110        out[4] = VERSION;
111        out[5] = self.bonds.len() as u8;
112        out[6] = u8::from(self.pin.is_some());
113        out[7] = u8::from(self.local_irk.is_some());
114        out[8..12].copy_from_slice(&self.generation.to_le_bytes());
115        out[12..16].copy_from_slice(&self.pin.unwrap_or(u32::MAX).to_le_bytes());
116        out[LOCAL_IRK_OFFSET..BONDS_OFFSET].copy_from_slice(&self.local_irk.unwrap_or([0; 16]));
117        for (index, bond) in self.bonds.iter().enumerate() {
118            let start = BONDS_OFFSET + index * BOND_SIZE;
119            out[start] = bond.address_kind;
120            out[start + 1..start + 7].copy_from_slice(&bond.address);
121            out[start + 7] = u8::from(bond.irk.is_some());
122            out[start + 8..start + 24].copy_from_slice(&bond.irk.unwrap_or([0; 16]));
123            out[start + 24..start + 40].copy_from_slice(&bond.ltk);
124            out[start + 40] = bond.security_level;
125            out[start + 41] = u8::from(bond.is_bonded);
126        }
127        let crc = crc32(&out[..CRC_OFFSET]);
128        out[CRC_OFFSET..COMMIT_OFFSET].copy_from_slice(&crc.to_le_bytes());
129        out
130    }
131
132    pub fn decode(bytes: &[u8; SLOT_SIZE]) -> Option<Self> {
133        if bytes[COMMIT_OFFSET..] != [0, 0, 0, 0]
134            || bytes[..4] != MAGIC
135            || bytes[4] != VERSION
136            || usize::from(bytes[5]) > MAX_BONDS
137            || crc32(&bytes[..CRC_OFFSET])
138                != u32::from_le_bytes(bytes[CRC_OFFSET..COMMIT_OFFSET].try_into().ok()?)
139        {
140            return None;
141        }
142        let pin = match (bytes[6], u32::from_le_bytes(bytes[12..16].try_into().ok()?)) {
143            (0, _) => None,
144            (1, value @ 0..=999_999) => Some(value),
145            _ => return None,
146        };
147        let local_irk = match bytes[7] {
148            0 => None,
149            1 => {
150                let value: [u8; 16] = bytes[LOCAL_IRK_OFFSET..BONDS_OFFSET].try_into().ok()?;
151                if value == [0; 16] {
152                    return None;
153                }
154                Some(value)
155            }
156            _ => return None,
157        };
158        let mut bonds = heapless::Vec::new();
159        for index in 0..usize::from(bytes[5]) {
160            let start = BONDS_OFFSET + index * BOND_SIZE;
161            let irk = match bytes[start + 7] {
162                0 => None,
163                1 => Some(bytes[start + 8..start + 24].try_into().ok()?),
164                _ => return None,
165            };
166            let security_level = bytes[start + 40];
167            if security_level > 2 || bytes[start + 41] > 1 {
168                return None;
169            }
170            bonds
171                .push(StoredBond {
172                    address_kind: bytes[start],
173                    address: bytes[start + 1..start + 7].try_into().ok()?,
174                    irk,
175                    ltk: bytes[start + 24..start + 40].try_into().ok()?,
176                    security_level,
177                    is_bonded: bytes[start + 41] == 1,
178                })
179                .ok()?;
180        }
181        Some(Self {
182            generation: u32::from_le_bytes(bytes[8..12].try_into().ok()?),
183            pin,
184            local_irk,
185            bonds,
186        })
187    }
188}

190/// Select the newest valid committed snapshot from `(flash_address, record)`
191/// pairs. Invalid, corrupt, and incompletely committed records are ignored.
192#[cfg(test)]
193pub fn latest_snapshot<'a>(
194    records: impl IntoIterator<Item = (u32, &'a [u8; SLOT_SIZE])>,
195) -> Option<(u32, Snapshot)> {
196    let mut latest: Option<(u32, Snapshot)> = None;
197    for (address, bytes) in records {
198        latest = consider_snapshot(latest, address, bytes);
199    }
200    latest
201}
202
203/// Consider one journal slot while mounting without retaining its flash buffer.
204pub fn consider_snapshot(
205    current: Option<(u32, Snapshot)>,
206    address: u32,
207    bytes: &[u8; SLOT_SIZE],
208) -> Option<(u32, Snapshot)> {
209    let Some(candidate) = Snapshot::decode(bytes) else {
210        return current;
211    };
212    if current
213        .as_ref()
214        .is_none_or(|(_, snapshot)| generation_is_newer(candidate.generation, snapshot.generation))
215    {
216        Some((address, candidate))
217    } else {
218        current
219    }
220}

232    /// Test-only journal addresses (production addresses are the
233    /// firmware's memory-map decision).
234    const PAGE0: u32 = 0x000E_4000;
235    const PAGE1: u32 = PAGE0 + PAGE_SIZE;

237    fn block_on<F: Future>(future: F) -> F::Output {
238        let mut future = core::pin::pin!(future);
239        let mut context = Context::from_waker(Waker::noop());
240        loop {
241            match future.as_mut().poll(&mut context) {
242                Poll::Ready(output) => return output,
243                Poll::Pending => std::thread::yield_now(),
244            }
245        }
246    }

249    struct MockWriter {
250        fail_call: Option<usize>,
251        calls: std::vec::Vec<(u32, std::vec::Vec<u8>)>,
252        erase_failure: bool,
253        erases: std::vec::Vec<(u32, u32)>,

256    impl RecordWriter for MockWriter {
257        type Error = usize;
258
259        async fn write_record(&mut self, address: u32, bytes: &[u8]) -> Result<(), Self::Error> {
260            let call = self.calls.len();
261            self.calls.push((address, bytes.to_vec()));
262            if self.fail_call == Some(call) {
263                Err(call)
264            } else {
265                Ok(())
266            }
267        }
268    }
269
270    impl PageEraser for MockWriter {
271        type Error = ();
272
273        async fn erase_page(&mut self, start: u32, end: u32) -> Result<(), Self::Error> {
274            self.erases.push((start, end));
275            if self.erase_failure { Err(()) } else { Ok(()) }
276        }
277    }

338    #[test]
339    fn interrupted_new_record_never_replaces_committed_old_record() {
340        let mut old = sample();
341        old.generation = 7;
342        let mut old_bytes = old.encode();
343        old_bytes[COMMIT_OFFSET..].fill(0);
344
345        let mut new = sample();
346        new.generation = 8;
347        new.pin = Some(654_321);
348        let encoded_new = new.encode();
349
350        // Simulate power loss after every possible byte of the body write and
351        // after each byte of the final commit-word write. Until all four
352        // commit bytes are present, mount must recover the old snapshot.
353        for body_bytes in 0..=COMMIT_OFFSET {
354            let mut interrupted = [0xff; SLOT_SIZE];
355            interrupted[..body_bytes].copy_from_slice(&encoded_new[..body_bytes]);
356            assert_eq!(
357                latest_snapshot([(PAGE0, &old_bytes), (PAGE1, &interrupted)]),
358                Some((PAGE0, old.clone())),
359                "body power cut after {body_bytes} bytes"
360            );
361        }
362        for commit_bytes in 0..4 {
363            let mut interrupted = encoded_new;
364            interrupted[COMMIT_OFFSET..COMMIT_OFFSET + commit_bytes].fill(0);
365            assert_eq!(
366                latest_snapshot([(PAGE0, &old_bytes), (PAGE1, &interrupted)]),
367                Some((PAGE0, old.clone())),
368                "commit power cut after {commit_bytes} bytes"
369            );
370        }
371
372        let mut committed = encoded_new;
373        committed[COMMIT_OFFSET..].fill(0);
374        assert_eq!(
375            latest_snapshot([(PAGE0, &old_bytes), (PAGE1, &committed)]),
376            Some((PAGE1, new))
377        );
378    }
