1//! Full-protocol snapshot journal records (spec §Saved State).
2//!
3//! Persists the opaque snapshot payload produced by
4//! `Session::encode_snapshot` — the journal knows nothing about its
5//! contents. Deliberately separate from the BLE bond/PIN journal
6//! ([`ble`](crate::ble)): the two have different lifecycles (`CMD_CLEAR`
7//! erases this journal but never touches bonds or the pairing PIN) and
8//! different record sizes. The record machinery — two-page rotation,
9//! CRC over the body, a trailing commit word written last, newest
10//! generation wins — is the shared [`record`](crate::record) engine.
11//!
12//! The same record format also carries the device identity and the
13//! device node's frame-counter map ([`counter`](crate::counter)), each
14//! in a journal of its own; which pages each journal owns is the
15//! firmware's memory-map decision.
16
17use crate::record::{self, CommitError, RecordWriter};

19/// A device-identity record payload: the Ed25519 private key followed
20/// by its public key (stored so boot does not repeat the derivation).
21pub const IDENTITY_PAYLOAD_LEN: usize = 64;
22
23pub fn encode_identity(secret: &[u8; 32], public: &[u8; 32]) -> [u8; IDENTITY_PAYLOAD_LEN] {
24    let mut payload = [0u8; IDENTITY_PAYLOAD_LEN];
25    payload[..32].copy_from_slice(secret);
26    payload[32..].copy_from_slice(public);
27    payload
28}
29
30/// Split a persisted identity payload into (secret, public); anything
31/// but the exact expected length is treated as no identity.
32pub fn decode_identity(payload: &[u8]) -> Option<([u8; 32], [u8; 32])> {
33    if payload.len() != IDENTITY_PAYLOAD_LEN {
34        return None;
35    }
36    Some((
37        payload[..32].try_into().expect("length checked"),
38        payload[32..].try_into().expect("length checked"),
39    ))
40}

42/// Two records per page; the snapshot payload is bounded by
43/// `umsh_companion_ncp::SNAPSHOT_MAX` (1024) with headroom.
44pub const SLOT_SIZE: usize = 2048;
45pub const COMMIT_OFFSET: usize = SLOT_SIZE - 4;
46const CRC_OFFSET: usize = COMMIT_OFFSET - 4;
47const MAGIC: [u8; 4] = *b"UPRS";
48const KIND_SNAPSHOT: u8 = 0;
49const KIND_CLEARED: u8 = 1;
50const HEADER_LEN: usize = 4 + 4 + 1 + 2;
51/// Largest payload a record can carry.
52pub const MAX_PAYLOAD: usize = CRC_OFFSET - HEADER_LEN;
53
54/// What a journal record asserts about the saved protocol state.
55#[derive(Clone, Debug, PartialEq, Eq)]
56pub enum Record {
57    /// A saved snapshot with its opaque payload.
58    Snapshot(heapless::Vec<u8, MAX_PAYLOAD>),
59    /// A committed `CMD_CLEAR`: nothing is saved, and any older
60    /// snapshot records still physically present are void. Erasing
61    /// pages is never the clear transaction — a single committed
62    /// tombstone is, so an interrupted clear can never resurrect an
63    /// older snapshot from a surviving page.
64    Cleared,
65}
66
67/// Borrowed form of [`Record`] for the write path: persist callers pass
68/// their payload by reference so the record machinery never buffers a
69/// second copy (the write path's task futures hold these across awaits,
70/// so every avoided `MAX_PAYLOAD` copy is RAM off a task pool).
71#[derive(Clone, Copy, Debug)]
72pub enum RecordRef<'a> {
73    Snapshot(&'a [u8]),
74    Cleared,
75}
76
77/// One journal record with its monotonically increasing generation;
78/// the newest valid record is authoritative.
79#[derive(Clone, Debug, PartialEq, Eq)]
80pub struct Stored {
81    pub generation: u32,
82    pub record: Record,
83}
84
85/// Encode one record body into a slot image. The payload must fit
86/// `MAX_PAYLOAD`; the commit word stays erased (0xFF) —
87/// `write_committed_record` writes zeros there only after the body
88/// lands.
89pub fn encode_record(generation: u32, record: RecordRef<'_>) -> [u8; SLOT_SIZE] {
90    let mut bytes = [0xFFu8; SLOT_SIZE];
91    bytes[..4].copy_from_slice(&MAGIC);
92    bytes[4..8].copy_from_slice(&generation.to_le_bytes());
93    let (kind, payload): (u8, &[u8]) = match record {
94        RecordRef::Snapshot(payload) => (KIND_SNAPSHOT, payload),
95        RecordRef::Cleared => (KIND_CLEARED, &[]),
96    };
97    bytes[8] = kind;
98    bytes[9..11].copy_from_slice(&(payload.len() as u16).to_le_bytes());
99    bytes[HEADER_LEN..HEADER_LEN + payload.len()].copy_from_slice(payload);
100    let crc = record::crc32(&bytes[..CRC_OFFSET]);
101    bytes[CRC_OFFSET..COMMIT_OFFSET].copy_from_slice(&crc.to_le_bytes());
102    bytes
103}

105impl Stored {
106    /// Owned-record slot image, kept for the host tests; the production
107    /// write path encodes through [`RecordRef`] without owning the
108    /// payload.
109    #[cfg(test)]
110    pub fn encode(&self) -> [u8; SLOT_SIZE] {
111        let record = match &self.record {
112            Record::Snapshot(payload) => RecordRef::Snapshot(payload),
113            Record::Cleared => RecordRef::Cleared,
114        };
115        encode_record(self.generation, record)
116    }
117
118    pub fn decode(bytes: &[u8; SLOT_SIZE]) -> Option<Self> {
119        if bytes[..4] != MAGIC {
120            return None;
121        }
122        if bytes[COMMIT_OFFSET..] != [0; 4] {
123            return None;
124        }
125        let crc = u32::from_le_bytes(bytes[CRC_OFFSET..COMMIT_OFFSET].try_into().ok()?);
126        if crc != record::crc32(&bytes[..CRC_OFFSET]) {
127            return None;
128        }
129        let generation = u32::from_le_bytes(bytes[4..8].try_into().ok()?);
130        let len = usize::from(u16::from_le_bytes(bytes[9..11].try_into().ok()?));
131        if len > MAX_PAYLOAD {
132            return None;
133        }
134        let record = match bytes[8] {
135            KIND_SNAPSHOT => {
136                let mut payload = heapless::Vec::new();
137                payload
138                    .extend_from_slice(&bytes[HEADER_LEN..HEADER_LEN + len])
139                    .ok()?;
140                Record::Snapshot(payload)
141            }
142            KIND_CLEARED if len == 0 => Record::Cleared,
143            _ => return None,
144        };
145        Some(Self { generation, record })
146    }
147}

149/// Consider one journal slot while mounting.
150pub fn consider_record(
151    current: Option<(u32, Stored)>,
152    address: u32,
153    bytes: &[u8; SLOT_SIZE],
154) -> Option<(u32, Stored)> {
155    let Some(candidate) = Stored::decode(bytes) else {
156        return current;
157    };
158    if current.as_ref().is_none_or(|(_, stored)| {
159        record::generation_is_newer(candidate.generation, stored.generation)
160    }) {
161        Some((address, candidate))
162    } else {
163        current
164    }
165}

167/// Write one committed record. Failure leaves any previously committed
168/// record untouched: the body lands first and the commit word last, so
169/// a mount never selects a partial write.
170pub async fn write_record<W: RecordWriter>(
171    writer: &mut W,
172    target: u32,
173    generation: u32,
174    record: RecordRef<'_>,
175) -> Result<(), CommitError<W::Error>> {
176    let bytes = encode_record(generation, record);
177    crate::record::write_committed_record(writer, target, &bytes).await
178}

187    /// Test-only journal addresses (production addresses are the
188    /// firmware's memory-map decision).
189    const PAGE0: u32 = 0x000E_8000;
190    const PAGE1: u32 = PAGE0 + PAGE_SIZE;
191
192    fn block_on<F: Future>(future: F) -> F::Output {
193        let mut future = core::pin::pin!(future);
194        let waker = Waker::noop();
195        let mut context = Context::from_waker(&waker);
196        loop {
197            if let Poll::Ready(output) = future.as_mut().poll(&mut context) {
198                return output;
199            }
200        }
201    }

203    /// Mock flash that persists into a byte map and can cut power after
204    /// a byte budget.
205    struct MockFlash {
206        bytes: std::collections::BTreeMap<u32, u8>,
207        budget: Option<usize>,
208    }
209
210    impl MockFlash {
211        fn new() -> Self {
212            Self {
213                bytes: std::collections::BTreeMap::new(),
214                budget: None,
215            }
216        }
217
218        fn slot(&self, address: u32) -> [u8; SLOT_SIZE] {
219            let mut out = [0xFFu8; SLOT_SIZE];
220            for (offset, byte) in out.iter_mut().enumerate() {
221                if let Some(value) = self.bytes.get(&(address + offset as u32)) {
222                    *byte = *value;
223                }
224            }
225            out
226        }
227
228        fn mount(&self) -> Option<(u32, Stored)> {
229            let mut latest = None;
230            for page in [PAGE0, PAGE1] {
231                let mut address = page;
232                while address < page + PAGE_SIZE {
233                    latest = consider_record(latest, address, &self.slot(address));
234                    address += SLOT_SIZE as u32;
235                }
236            }
237            latest
238        }
239
240        /// The snapshot payload a boot would restore: the newest valid
241        /// record when it is a snapshot, nothing when it is a
242        /// tombstone.
243        fn mounted_snapshot(&self) -> Option<Stored> {
244            match self.mount() {
245                Some((_, stored)) if matches!(stored.record, Record::Snapshot(_)) => Some(stored),
246                _ => None,
247            }
248        }
249    }

251    impl RecordWriter for MockFlash {
252        type Error = ();
253
254        async fn write_record(&mut self, address: u32, bytes: &[u8]) -> Result<(), Self::Error> {
255            for (offset, byte) in bytes.iter().enumerate() {
256                if let Some(budget) = &mut self.budget {
257                    if *budget == 0 {
258                        return Err(());
259                    }
260                    *budget -= 1;
261                }
262                self.bytes.insert(address + offset as u32, *byte);
263            }
264            Ok(())
265        }
266    }

278    fn record(generation: u32, fill: u8, len: usize) -> Stored {
279        let mut payload = heapless::Vec::new();
280        payload.resize(len, fill).unwrap();
281        Stored {

283            record: Record::Snapshot(payload),

366    /// A committed tombstone is the clear transaction: it voids every
367    /// older snapshot on either page without any erase, and an
368    /// interrupted tombstone write leaves the previous snapshot
369    /// authoritative.
370    #[test]
371    fn tombstone_clears_and_survives_interruption() {
372        // Old snapshots on both pages, newest on PAGE1.
373        let mut flash = MockFlash::new();
374        block_on(write_record(&mut flash, PAGE0, &record(1, 0xAA, 40))).unwrap();
375        block_on(write_record(&mut flash, PAGE1, &record(2, 0xBB, 40))).unwrap();
376        assert_eq!(flash.mounted_snapshot().unwrap(), record(2, 0xBB, 40));
377
378        // Cut the tombstone write at every distinct byte boundary: the
379        // newest snapshot must remain authoritative — never the older
380        // one. Every header byte, the CRC and commit regions, and
381        // samples of the 0xFF-filled body (where all cuts are
382        // physically identical: writing 0xFF to erased flash changes
383        // nothing).
384        let total = COMMIT_OFFSET + 4;
385        for cut in (0..=HEADER_LEN + 1)
386            .chain((HEADER_LEN..CRC_OFFSET).step_by(89))
387            .chain(CRC_OFFSET - 1..total)
388        {
389            let mut flash = flash_with_two_snapshots();
390            flash.budget = Some(cut);
391            let target = PAGE0 + SLOT_SIZE as u32;
392            assert!(block_on(write_record(&mut flash, target, &tombstone(3))).is_err());
393            flash.budget = None;
394            assert_eq!(
395                flash.mounted_snapshot().expect("snapshot must survive"),
396                record(2, 0xBB, 40),
397                "cut at {cut} lost or replaced the committed snapshot"
398            );
399        }
400
401        // The committed tombstone mounts as no snapshot, with both
402        // older snapshot records still physically present.
403        let mut flash = flash_with_two_snapshots();
404        block_on(write_record(
405            &mut flash,
406            PAGE0 + SLOT_SIZE as u32,
407            &tombstone(3),
408        ))
409        .unwrap();
410        assert!(flash.mounted_snapshot().is_none());
411        assert_eq!(flash.mount().unwrap().1, tombstone(3));
412
413        // Clearing again is idempotent, and a later save supersedes.
414        block_on(write_record(
415            &mut flash,
416            PAGE1 + SLOT_SIZE as u32,
417            &tombstone(4),
418        ))
419        .unwrap();
420        assert!(flash.mounted_snapshot().is_none());
421        block_on(write_record(&mut flash, PAGE0, &record(5, 0xCC, 8))).unwrap();
422        assert_eq!(flash.mounted_snapshot().unwrap(), record(5, 0xCC, 8));
423    }

456    /// Cut the write at every byte boundary: a mount afterwards always
457    /// yields the previously committed record, never a mixture.
458    #[test]
459    fn power_cut_at_every_byte_never_replaces_the_committed_record() {
460        let old = record(7, 0x11, 700);
461        let new = record(8, 0x22, 700);
462        // Total bytes a full record write issues (body + commit word).
463        let total = COMMIT_OFFSET + 4;
464        for cut in 0..total {
465            let mut flash = MockFlash::new();
466            block_on(write_record(&mut flash, PAGE0, &old)).unwrap();
467            flash.budget = Some(cut);
468            let result = block_on(write_record(&mut flash, PAGE0 + SLOT_SIZE as u32, &new));
469            assert!(result.is_err(), "cut at {cut} must fail the write");
470            flash.budget = None;
471            let (_, mounted) = flash.mount().expect("old record must survive");
472            assert_eq!(mounted, old, "cut at {cut} corrupted the mount");
473        }
474        // And the complete write wins.
475        let mut flash = MockFlash::new();
476        block_on(write_record(&mut flash, PAGE0, &old)).unwrap();
477        block_on(write_record(&mut flash, PAGE0 + SLOT_SIZE as u32, &new)).unwrap();
478        assert_eq!(flash.mount().unwrap().1, new);
479    }
