430struct Phase1Vectors {
431    node_a_private: [u8; 32],
432    node_b_private: [u8; 32],
433    node_a_public: PublicKey,
434    node_b_public: PublicKey,
435    shared_secret: SharedSecret,
436    channel_key: ChannelKey,
437    channel: DerivedChannelKeys,
438    broadcast_hint: Vec<u8>,
439    broadcast_full: Vec<u8>,
440    unicast_encrypted: Vec<u8>,
441    unicast_ackreq_fullsrc: Vec<u8>,
442    multicast_encrypted: Vec<u8>,
443    multicast_authenticated: Vec<u8>,
444    unicast_with_options: Vec<u8>,
445    blind_unicast: Vec<u8>,
446}

448fn generate_phase1_vectors() -> Phase1Vectors {
449    let engine = SoftwareCryptoEngine::new(SoftwareAes, SoftwareSha256);
450
451    let node_a_private = [
452        0x11, 0x12, 0x13, 0x14, 0x15, 0x16, 0x17, 0x18, 0x19, 0x1A, 0x1B, 0x1C, 0x1D, 0x1E, 0x1F,
453        0x20, 0x21, 0x22, 0x23, 0x24, 0x25, 0x26, 0x27, 0x28, 0x29, 0x2A, 0x2B, 0x2C, 0x2D, 0x2E,
454        0x2F, 0x30,
455    ];
456    let node_b_private = [
457        0x31, 0x32, 0x33, 0x34, 0x35, 0x36, 0x37, 0x38, 0x39, 0x3A, 0x3B, 0x3C, 0x3D, 0x3E, 0x3F,
458        0x40, 0x41, 0x42, 0x43, 0x44, 0x45, 0x46, 0x47, 0x48, 0x49, 0x4A, 0x4B, 0x4C, 0x4D, 0x4E,
459        0x4F, 0x50,
460    ];
461    let node_a_identity = SoftwareIdentity::from_secret_bytes(&node_a_private);
462    let node_b_identity = SoftwareIdentity::from_secret_bytes(&node_b_private);
463    let node_a_public = *node_a_identity.public_key();
464    let node_b_public = *node_b_identity.public_key();
465    let shared_secret = node_a_identity.shared_secret_with(&node_b_public).unwrap();
466    let pairwise = engine.derive_pairwise_keys(&shared_secret);
467    let channel_key = ChannelKey([0x5A; 32]);
468    let channel = engine.derive_channel_keys(&channel_key);
469    let blind_keys = engine.derive_blind_keys(&pairwise, &channel);
470
471    Phase1Vectors {
472        node_a_private,
473        node_b_private,
474        node_a_public,
475        node_b_public,
476        shared_secret,
477        channel_key,
478        broadcast_hint: build_broadcast_hint(&node_a_public),
479        broadcast_full: build_broadcast_full(&node_a_public),
480        unicast_encrypted: build_unicast_encrypted(
481            &engine,
482            &pairwise,
483            &node_a_public,
484            node_b_public.hint(),
485        ),
486        unicast_ackreq_fullsrc: build_unicast_ackreq_fullsrc(
487            &engine,
488            &pairwise,
489            &node_a_public,
490            node_b_public.hint(),
491        ),
492        multicast_encrypted: build_multicast_encrypted(&engine, &channel, &node_a_public),
493        multicast_authenticated: build_multicast_authenticated(&engine, &channel, &node_a_public),
494        unicast_with_options: build_unicast_with_options(
495            &engine,
496            &pairwise,
497            &node_a_public,
498            node_b_public.hint(),
499        ),
500        blind_unicast: build_blind_unicast(
501            &engine,
502            &blind_keys,
503            &channel,
504            &node_a_public,
505            node_b_public.hint(),
506        ),
507        channel,
508    }
509}

511fn build_broadcast_hint(node_a: &PublicKey) -> Vec<u8> {
512    let mut buf = [0u8; 64];
513    PacketBuilder::new(&mut buf)
514        .broadcast()
515        .source_hint(node_a.hint())
516        .build()
517        .unwrap()
518        .to_vec()
519}

521fn build_broadcast_full(node_a: &PublicKey) -> Vec<u8> {
522    let mut buf = [0u8; 64];
523    PacketBuilder::new(&mut buf)
524        .broadcast()
525        .source_full(node_a)
526        .build()
527        .unwrap()
528        .to_vec()
529}

531fn build_unicast_encrypted(
532    engine: &SoftwareCryptoEngine,
533    pairwise: &PairwiseKeys,
534    node_a: &PublicKey,
535    dst: NodeHint,
536) -> Vec<u8> {
537    let mut buf = [0u8; 128];
538    let mut packet = PacketBuilder::new(&mut buf)
539        .unicast(dst)
540        .source_hint(node_a.hint())
541        .frame_counter(42)
542        .encrypted()
543        .mic_size(MicSize::Mic16)
544        .payload(b"Hello")
545        .build()
546        .unwrap();
547    engine.seal_packet(&mut packet, pairwise).unwrap();
548    packet.as_bytes().to_vec()
549}

572fn build_multicast_encrypted(
573    engine: &SoftwareCryptoEngine,
574    channel: &DerivedChannelKeys,
575    node_a: &PublicKey,
576) -> Vec<u8> {
577    let mut buf = [0u8; 128];
578    let mut packet = PacketBuilder::new(&mut buf)
579        .multicast(channel.channel_id)
580        .source_hint(node_a.hint())
581        .frame_counter(5)
582        .encrypted()
583        .mic_size(MicSize::Mic16)
584        .payload(b"Hello")
585        .build()
586        .unwrap();
587    let keys = PairwiseKeys {
588        k_enc: channel.k_enc,
589        k_mic: channel.k_mic,
590    };
591    engine.seal_packet(&mut packet, &keys).unwrap();
592    packet.as_bytes().to_vec()
593}

640    engine: &SoftwareCryptoEngine,
641    blind_keys: &PairwiseKeys,
642    channel: &DerivedChannelKeys,
643    node_a: &PublicKey,
644    dst: NodeHint,
645) -> Vec<u8> {
646    let mut buf = [0u8; 160];
647    let mut packet = PacketBuilder::new(&mut buf)
648        .blind_unicast(channel.channel_id, dst)
649        .source_hint(node_a.hint())
650        .frame_counter(7)
651        .mic_size(MicSize::Mic16)
652        .payload(b"Hello")
653        .build()
654        .unwrap();
655    engine
656        .seal_blind_packet(&mut packet, blind_keys, channel)
657        .unwrap();
658    packet.as_bytes().to_vec()
659}

661fn render_example(
662    out: &mut String,
663    title: &str,
664    description: &str,
665    fields: &[(&str, &str, String)],
666    bytes: &[u8],
667) {
668    writeln!(out).unwrap();
669    writeln!(out, "## {}", title).unwrap();
670    writeln!(out).unwrap();
671    writeln!(out, "{}", description).unwrap();
672    writeln!(out).unwrap();
673    writeln!(out, "| Field | Value | Hex |").unwrap();
674    writeln!(out, "|---|---|---|").unwrap();
675    for (field, value, hex) in fields {
676        writeln!(out, "| {} | {} | `{}` |", field, value, hex).unwrap();
677    }
678    writeln!(out).unwrap();
679    writeln!(out, "```text").unwrap();
680    writeln!(out, "{}", wrap_hex_lines(bytes, 16)).unwrap();
681    writeln!(out, "```").unwrap();
682    writeln!(out).unwrap();
683    writeln!(out, "Total: {} bytes.", bytes.len()).unwrap();
684}

686fn hex_spaced(bytes: &[u8]) -> String {
687    bytes
688        .iter()
689        .map(|byte| format!("{:02X}", byte))
690        .collect::<Vec<_>>()
691        .join(" ")
692}
693
694fn hex_words(bytes: &[u8]) -> String {
695    bytes
696        .chunks(2)
697        .map(|chunk| {
698            chunk
699                .iter()
700                .map(|byte| format!("{:02X}", byte))
701                .collect::<String>()
702        })
703        .collect::<Vec<_>>()
704        .join(" ")
705}
706
707fn wrap_hex_lines(bytes: &[u8], line_width: usize) -> String {
708    bytes
709        .chunks(line_width)
710        .map(hex_spaced)
711        .collect::<Vec<_>>()
712        .join("\n")
713}
