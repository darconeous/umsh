1//! Host-side client for the minimal companion-radio (NCP) protocol.
2//!
3//! [`CompanionRadio`] drives an NCP over a frame-oriented [`FrameLink`], and
4//! exposes the link as a [`umsh_hal::Radio`] so the host can run the
5//! full MAC/node stack with the NCP acting purely as the PHY.
6//!
7//! The wire format lives in [`umsh_companion`] (re-exported as
8//! [`crate::companion`]); this module owns the host-side session
9//! behavior: the reset/configure handshake, request/response
10//! transactions, and queueing of frames that arrive while a command is
11//! in flight.
12//!
13//! See `docs/protocol/src/companion-radio-minimal.md` for the protocol.
14

35/// Capacity of the HDLC reassembly buffer (unescaped frame + FCS).
36const WIRE_BUF: usize = 1024;
37/// Size of one read from the underlying stream.
38const READ_CHUNK: usize = 256;
39/// Received frames buffered while a command transaction is in flight.
40/// The oldest frame is dropped on overflow, matching radio-FIFO
41/// overrun semantics.

43/// Stale command responses retained before the oldest is dropped.
44const RESPONSE_QUEUE_DEPTH: usize = 8;
45/// Unsolicited property notifications retained before the oldest is
46/// dropped.

51#[derive(Debug)]
52pub enum CompanionRadioError {
53    Io(io::Error),
54    /// The stream reached end-of-file; the NCP link is gone.
55    Disconnected,
56    /// The NCP violated the companion-radio protocol.
57    Protocol(&'static str),
58    /// The NCP reported a failure status for a command.
59    Status(Status),
60    /// The NCP reset outside of an initialization handshake, losing
61    /// its configuration. The radio must be re-initialized.
62    UnexpectedReset(Status),
63    /// The frame exceeds the NCP's advertised MTU.
64    FrameTooLarge(usize),
65    /// The NCP did not answer a command in time.
66    Timeout,
67    /// A non-stream transport failed.
68    Transport(String),
69}
70
71impl core::fmt::Display for CompanionRadioError {
72    fn fmt(&self, formatter: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
73        match self {
74            Self::Io(error) => write!(formatter, "io error: {error}"),
75            Self::Disconnected => write!(formatter, "companion link disconnected"),
76            Self::Protocol(message) => write!(formatter, "protocol error: {message}"),
77            Self::Status(status) => write!(formatter, "NCP reported {status:?}"),
78            Self::UnexpectedReset(status) => {
79                write!(formatter, "NCP reset unexpectedly ({status:?})")
80            }
81            Self::FrameTooLarge(len) => write!(formatter, "frame too large: {len} bytes"),
82            Self::Timeout => write!(formatter, "timed out waiting for NCP response"),
83            Self::Transport(message) => write!(formatter, "transport error: {message}"),
84        }
85    }
86}

350/// A cancel-safe, frame-oriented companion transport.
351#[allow(async_fn_in_trait)]
352pub trait FrameLink {
353    /// Send one complete companion frame.
354    async fn send_frame(&mut self, frame: &[u8]) -> Result<(), CompanionRadioError>;
355
356    /// Poll for the next complete companion frame.
357    ///
358    /// Implementations keep all partial state in `self`, so cancellation cannot
359    /// discard a partial frame or unread bytes following a completed frame.
360    fn poll_recv_frame(
361        &mut self,
362        cx: &mut core::task::Context<'_>,
363    ) -> core::task::Poll<Result<Vec<u8>, CompanionRadioError>>;
364
365    /// Receive the next complete companion frame.
366    async fn recv_frame(&mut self) -> Result<Vec<u8>, CompanionRadioError> {
367        core::future::poll_fn(|cx| self.poll_recv_frame(cx)).await
368    }
369}

380impl<IO> SerialFrameLink<IO> {
381    /// Wrap a byte stream in companion HDLC framing.
382    pub fn new(io: IO) -> Self {
383        Self {
384            io,
385            decoder: hdlc::Decoder::new(),
386            read_buf: [0; READ_CHUNK],
387            read_pos: 0,
388            read_len: 0,
389        }
390    }
391
392    /// Recover the underlying byte stream.
393    pub fn into_inner(self) -> IO {
394        self.io
395    }
396}
397
398impl<IO> FrameLink for SerialFrameLink<IO>
399where
400    IO: AsyncRead + AsyncWrite + Unpin,
401{
402    async fn send_frame(&mut self, frame: &[u8]) -> Result<(), CompanionRadioError> {
403        let mut wire = vec![0u8; hdlc::max_encoded_len(frame.len())];
404        let len = hdlc::encode_frame(frame, &mut wire).expect("buffer sized with max_encoded_len");
405        self.io.write_all(&wire[..len]).await?;
406        self.io.flush().await?;
407        Ok(())
408    }
409
410    fn poll_recv_frame(
411        &mut self,
412        cx: &mut core::task::Context<'_>,
413    ) -> core::task::Poll<Result<Vec<u8>, CompanionRadioError>> {
414        loop {
415            while self.read_pos < self.read_len {
416                let byte = self.read_buf[self.read_pos];
417                self.read_pos += 1;
418                if let Some(Ok(frame)) = self.decoder.push(byte) {
419                    return core::task::Poll::Ready(Ok(frame.to_vec()));
420                }
421            }
422
423            self.read_pos = 0;
424            self.read_len = 0;
425            let mut read_buf = ReadBuf::new(&mut self.read_buf);
426            match core::pin::Pin::new(&mut self.io).poll_read(cx, &mut read_buf) {
427                core::task::Poll::Ready(Ok(())) => {
428                    self.read_len = read_buf.filled().len();
429                    if self.read_len == 0 {
430                        return core::task::Poll::Ready(Err(CompanionRadioError::Disconnected));
431                    }
432                }
433                core::task::Poll::Ready(Err(error)) => {
434                    return core::task::Poll::Ready(Err(CompanionRadioError::Io(error)));
435                }
436                core::task::Poll::Pending => return core::task::Poll::Pending,
437            }
438        }
439    }
440}

491#[cfg(feature = "ble-radio")]
492struct BleNotificationReceiver {
493    notifications: tokio::sync::mpsc::Receiver<Vec<u8>>,
494    reassembler: umsh_companion::gatt::Reassembler<{ umsh_companion::gatt::MAX_FRAME }>,
495}
496
497#[cfg(feature = "ble-radio")]
498impl BleNotificationReceiver {
499    fn new(notifications: tokio::sync::mpsc::Receiver<Vec<u8>>) -> Self {
500        Self {
501            notifications,
502            reassembler: umsh_companion::gatt::Reassembler::new(),
503        }
504    }
505
506    fn poll_recv_frame(
507        &mut self,
508        cx: &mut core::task::Context<'_>,
509    ) -> core::task::Poll<Result<Vec<u8>, CompanionRadioError>> {
510        loop {
511            match self.notifications.poll_recv(cx) {
512                core::task::Poll::Ready(Some(segment)) => {
513                    if let Some(Ok(frame)) = self.reassembler.push(&segment) {
514                        return core::task::Poll::Ready(Ok(frame.to_vec()));
515                    }
516                    // Transport-level malformed/oversize segments are dropped.
517                }
518                core::task::Poll::Ready(None) => {
519                    self.reassembler.reset();
520                    return core::task::Poll::Ready(Err(CompanionRadioError::Disconnected));
521                }
522                core::task::Poll::Pending => return core::task::Poll::Pending,
523            }
524        }
525    }
526}

538#[cfg(feature = "ble-radio")]
539impl BleFrameLink {
540    /// Discover and attach to a Companion Link Service peripheral.
541    ///
542    /// `selector` matches a local-name substring or the platform peripheral ID.
543    /// With no selector, discovery must yield exactly one companion radio.
544    pub async fn connect(
545        selector: Option<&str>,
546        config: BleFrameLinkConfig,
547    ) -> Result<Self, CompanionRadioError> {
548        use btleplug::api::{Central, Manager as _, Peripheral as _, ScanFilter};
549        use futures_util::StreamExt;
550
551        config.validate()?;
552
553        let manager = btleplug::platform::Manager::new()
554            .await
555            .map_err(ble_error)?;
556        let adapters = manager.adapters().await.map_err(ble_error)?;
557        let service = uuid::Uuid::from_u128(umsh_companion::gatt::SERVICE_UUID);
558        let deadline = Instant::now() + config.discovery_timeout;
559        let mut matches = Vec::new();
560
561        for adapter in adapters {
562            adapter
563                .start_scan(ScanFilter {
564                    services: vec![service],
565                })
566                .await
567                .map_err(ble_error)?;
568            loop {
569                if Instant::now() >= deadline {
570                    break;
571                }
572                tokio::time::sleep(Duration::from_millis(250)).await;
573                matches.clear();
574                let peripherals =
575                    match tokio::time::timeout_at(deadline, adapter.peripherals()).await {
576                        Ok(result) => result.map_err(ble_error)?,
577                        Err(_) => break,
578                    };
579                for peripheral in peripherals {
580                    let properties =
581                        match tokio::time::timeout_at(deadline, peripheral.properties()).await {
582                            Ok(result) => result.map_err(ble_error)?,
583                            Err(_) => break,
584                        };
585                    let id = peripheral.id().to_string();
586                    let name = properties
587                        .as_ref()
588                        .and_then(|properties| properties.local_name.as_deref());
589                    let selected = selector.is_none_or(|selector| {
590                        id == selector || name.is_some_and(|name| name.contains(selector))
591                    });
592                    let advertises_service = properties
593                        .as_ref()
594                        .is_some_and(|properties| properties.services.contains(&service));
595                    if selected && advertises_service {
596                        matches.push(peripheral);
597                    }
598                }
599                if !matches.is_empty() || Instant::now() >= deadline {
600                    break;
601                }
602            }
603            // CoreBluetooth operations can block indefinitely. Discovery's
604            // configured deadline applies to every await, including cleanup.
605            let _ = tokio::time::timeout(Duration::from_secs(1), adapter.stop_scan()).await;
606            if !matches.is_empty() {
607                break;
608            }
609        }
610
611        let peripheral = match matches.len() {
612            0 => {
613                return Err(CompanionRadioError::Transport(
614                    "no Companion Link Service peripheral found".into(),
615                ));
616            }
617            1 => matches.pop().unwrap(),
618            _ => {
619                return Err(CompanionRadioError::Transport(
620                    "multiple companion radios found; provide a selector".into(),
621                ));
622            }
623        };
624
625        let setup = async {
626            let is_connected =
627                tokio::time::timeout(config.operation_timeout, peripheral.is_connected())
628                    .await
629                    .map_err(|_| ble_timeout("querying connection state"))?
630                    .map_err(ble_error)?;
631            if !is_connected {
632                tokio::time::timeout(config.operation_timeout, peripheral.connect())
633                    .await
634                    .map_err(|_| ble_timeout("connecting"))?
635                    .map_err(ble_error)?;
636            }
637            tokio::time::timeout(config.operation_timeout, peripheral.discover_services())
638                .await
639                .map_err(|_| ble_timeout("discovering services"))?
640                .map_err(ble_error)?;
641
642            let frame_in_uuid = uuid::Uuid::from_u128(umsh_companion::gatt::FRAME_IN_UUID);
643            let frame_out_uuid = uuid::Uuid::from_u128(umsh_companion::gatt::FRAME_OUT_UUID);
644            let characteristics = peripheral.characteristics();
645            let frame_in = characteristics
646                .iter()
647                .find(|characteristic| characteristic.uuid == frame_in_uuid)
648                .cloned()
649                .ok_or(CompanionRadioError::Protocol("missing BLE Frame In"))?;
650            let frame_out = characteristics
651                .iter()
652                .find(|characteristic| characteristic.uuid == frame_out_uuid)
653                .cloned()
654                .ok_or(CompanionRadioError::Protocol("missing BLE Frame Out"))?;
655
656            let mut stream =
657                tokio::time::timeout(config.operation_timeout, peripheral.notifications())
658                    .await
659                    .map_err(|_| ble_timeout("opening notifications"))?
660                    .map_err(ble_error)?;
661            let (tx, notifications) = tokio::sync::mpsc::channel(32);
662            tokio::spawn(async move {
663                while let Some(notification) = stream.next().await {
664                    if notification.uuid == frame_out_uuid
665                        && tx.send(notification.value).await.is_err()
666                    {
667                        break;
668                    }
669                }
670            });
671            // This security-gated CCCD write is the protocol attach edge.
672            // Pairing prompts are mediated by the host OS.
673            tokio::time::timeout(config.pairing_timeout, peripheral.subscribe(&frame_out))
674                .await
675                .map_err(|_| ble_timeout("subscribing to Frame Out"))?
676                .map_err(ble_error)?;
677            Ok::<_, CompanionRadioError>((frame_in, notifications))
678        }
679        .await;
680
681        let (frame_in, notifications) = match setup {
682            Ok(setup) => setup,
683            Err(error) => {
684                // Failed setup must not leave the single-connection NCP
685                // occupied and invisible to the next retry.
686                let _ = tokio::time::timeout(Duration::from_secs(1), peripheral.disconnect()).await;
687                return Err(error);
688            }
689        };
690
691        Ok(Self {
692            peripheral,
693            frame_in,
694            receiver: BleNotificationReceiver::new(notifications),
695            segment_payload: config.segment_payload,
696            operation_timeout: config.operation_timeout,
697        })
698    }

700    /// Capture the backend's view of a failed link, then make a bounded
701    /// best-effort disconnect so a subsequent discovery does not inherit a
702    /// stale CoreBluetooth/BlueZ connection object.
703    async fn diagnose_and_disconnect(&self, failure: String) -> CompanionRadioError {
704        use btleplug::api::Peripheral as _;
705
706        let connected = match tokio::time::timeout(
707            Duration::from_secs(2),
708            self.peripheral.is_connected(),
709        )
710        .await
711        {
712            Ok(Ok(value)) => value.to_string(),
713            Ok(Err(error)) => format!("error({error})"),
714            Err(_) => "query-timeout".into(),
715        };
716        let cleanup = match tokio::time::timeout(
717            Duration::from_secs(2),
718            self.peripheral.disconnect(),
719        )
720        .await
721        {
722            Ok(Ok(())) => "ok".into(),
723            Ok(Err(error)) => format!("error({error})"),
724            Err(_) => "timeout".into(),
725        };
726        CompanionRadioError::Transport(format!(
727            "{failure}; backend is_connected={connected}; disconnect cleanup={cleanup}"
728        ))
729    }

839    /// Attach to an already-operating NCP without resetting or
840    /// reconfiguring it.
841    ///
842    /// This is the full-protocol attach (spec §Attach, Detach, and
843    /// Synchronization): attach implies no known state, so the host
844    /// synchronizes by fetching. Only the identity handshake runs here
845    /// — retained `PROP_LAST_STATUS` (the reset cause, preserved for
846    /// [`Self::boot_status`] and [`Self::sync`]), the protocol version
847    /// check, `PROP_NCP_VERSION`, and `PROP_PHY_MTU`. The PHY keeps
848    /// whatever configuration and enable state it had; queued frames
849    /// and provisioning are untouched. Follow with [`Self::sync`] and
850    /// drain the queue when actually ready to process it.
851    pub async fn attach_existing(
852        link: L,
853        config: CompanionRadioConfig,
854    ) -> Result<Self, CompanionRadioError> {
855        let mut radio = Self::bare(link, config);
856        // Reading LAST_STATUS does not overwrite it, so sync() still
857        // sees a retained reset code after this handshake.
858        let boot_status = radio.get_prop(prop::LAST_STATUS).await?;
859        radio.boot_status = decode_status(&boot_status);
860
861        let version = radio.get_prop(prop::PROTOCOL_VERSION).await?;
862        if version.first().copied() != Some(ids::PROTOCOL_MAJOR_VERSION) {
863            return Err(CompanionRadioError::Protocol(
864                "protocol major version mismatch",
865            ));
866        }
867        let ncp_version = radio.get_prop(prop::NCP_VERSION).await?;
868        radio.ncp_version = String::from_utf8_lossy(&ncp_version)
869            .trim_end_matches('\0')
870            .to_owned();
871
872        let mtu = radio.get_prop(prop::PHY_MTU).await?;
873        let [mtu_lo, mtu_hi, ..] = mtu[..] else {
874            return Err(CompanionRadioError::Protocol("malformed PROP_PHY_MTU"));
875        };
876        radio.max_frame_size = usize::from(u16::from_le_bytes([mtu_lo, mtu_hi]));
877        if radio.max_frame_size == 0 {
878            return Err(CompanionRadioError::Protocol("NCP advertised zero MTU"));
879        }
880        radio.t_frame_ms = lora_airtime_ms(
881            radio.config.spreading_factor,
882            radio.config.bandwidth_hz,
883            radio.config.coding_rate_denom,
884            radio.max_frame_size,
885        )
886        .max(1);
887        Ok(radio)
888    }

941    async fn initialize(&mut self) -> Result<(), CompanionRadioError> {
942        // The reset-status property is deliberately read before CMD_RST. The
943        // protocol requires the NCP to retain its hardware boot cause for this
944        // first query; CMD_RST would replace it with RESET_SOFTWARE.
945        let boot_status = self.get_prop(prop::LAST_STATUS).await?;
946        self.boot_status = decode_status(&boot_status);
947
948        // Reset and wait for the reset notification. The TID is
949        // ignored for CMD_RST; the notification is unsolicited.
950        let mut buf = [0u8; 2];
951        let len = frame::reset(&mut buf, TID_UNSOLICITED)
952            .map_err(|_| CompanionRadioError::Protocol("frame encode"))?;
953        self.send(&buf[..len]).await?;
954        let deadline = Instant::now() + self.config.response_timeout;
955        self.wait_reset(deadline).await?;
956
957        // Reject NCPs speaking an incompatible protocol revision.
958        let version = self.get_prop(prop::PROTOCOL_VERSION).await?;
959        if version.first().copied() != Some(ids::PROTOCOL_MAJOR_VERSION) {
960            return Err(CompanionRadioError::Protocol(
961                "protocol major version mismatch",
962            ));
963        }
964
965        let ncp_version = self.get_prop(prop::NCP_VERSION).await?;
966        self.ncp_version = String::from_utf8_lossy(&ncp_version)
967            .trim_end_matches('\0')
968            .to_owned();
969
970        let mtu = self.get_prop(prop::PHY_MTU).await?;
971        let [mtu_lo, mtu_hi, ..] = mtu[..] else {
972            return Err(CompanionRadioError::Protocol("malformed PROP_PHY_MTU"));
973        };
974        self.max_frame_size = usize::from(u16::from_le_bytes([mtu_lo, mtu_hi]));
975        if self.max_frame_size == 0 {
976            return Err(CompanionRadioError::Protocol("NCP advertised zero MTU"));
977        }
978
979        let config = self.config.clone();
980        self.set_prop(prop::PHY_FREQ, &config.freq_khz.to_le_bytes())
981            .await?;
982        self.set_prop(prop::PHY_LORA_BW, &config.bandwidth_hz.to_le_bytes())
983            .await?;
984        self.set_prop(prop::PHY_LORA_SF, &[config.spreading_factor])
985            .await?;
986        self.set_prop(prop::PHY_LORA_CR, &[config.coding_rate_denom])
987            .await?;
988        self.set_prop(prop::PHY_TX_POWER, &[config.tx_power_dbm as u8])
989            .await?;
990        self.set_prop(prop::PHY_LORA_SW, &config.sync_word.to_le_bytes())
991            .await?;
992        self.set_prop(prop::PHY_ENABLED, &[1]).await?;
993
994        self.t_frame_ms = lora_airtime_ms(
995            config.spreading_factor,
996            config.bandwidth_hz,
997            config.coding_rate_denom,
998            self.max_frame_size,
999        )
1000        .max(1);
1001        Ok(())
1002    }

1087    /// Drain the NCP's inbound queue (`CMD_QUEUE_DRAIN`).
1088    ///
1089    /// Buffered frames are delivered as ordinary `CMD_STR_RECV` and land
1090    /// in the receive queue for [`Radio::poll_receive`]; this future
1091    /// resolves on the correlated completion status.

1094    }

1096    /// As [`Self::queue_drain`], invoking `on_frame` with each frame
1097    /// (data, trailing metadata bytes) delivered before completion —
1098    /// buffered and interleaved live frames alike. The callback sees
1099    /// **every** such frame: an NCP queue larger than this driver's
1100    /// bounded receive buffer drains losslessly through it. Frames are
1101    /// additionally queued for [`Radio::poll_receive`], where the
1102    /// bounded buffer's oldest-dropped policy still applies.
1103    pub async fn queue_drain_with(
1104        &mut self,
1105        mut on_frame: impl FnMut(&[u8], &[u8]),
1106    ) -> Result<(), CompanionRadioError> {
1107        let tid = self.alloc_tid();
1108        let mut buf = [0u8; 4];
1109        let len = frame::queue_drain(&mut buf, tid)
1110            .map_err(|_| CompanionRadioError::Protocol("frame encode"))?;
1111        self.send(&buf[..len]).await?;
1112
1113        let deadline = Instant::now() + self.config.response_timeout;
1114        loop {
1115            while let Some(response) = self.responses.pop_front() {
1116                if response.tid != tid {
1117                    continue;
1118                }
1119                if response.kind == ResponseKind::Is && response.key == prop::LAST_STATUS {
1120                    let status = decode_status(&response.value);
1121                    return if status == Status::OK {
1122                        Ok(())
1123                    } else {
1124                        Err(CompanionRadioError::Status(status))
1125                    };
1126                }
1127                return Err(CompanionRadioError::Protocol("unexpected drain response"));
1128            }
1129            if let Some(status) = self.seen_reset.take() {
1130                return Err(CompanionRadioError::UnexpectedReset(status));
1131            }
1132            // Deliver at ingest time: each read that queued a stream
1133            // frame reports it immediately, so the callback cannot
1134            // miss frames the bounded receive buffer evicts mid-drain.
1135            if self.read_more(deadline).await? {
1136                let packet = self.rx_queue.back().expect("read_more reported a queued frame");
1137                on_frame(&packet.data, &packet.raw_meta);
1138            }
1139        }
1140    }

1172                        Ok(RestoreCompletion::Updated)

1176                }

1179            match self.seen_reset.take() {

1181                    return Ok(RestoreCompletion::Reset);
1182                }
1183                Some(status) => return Err(CompanionRadioError::UnexpectedReset(status)),
1184                None => {}
1185            }
1186            self.read_more(deadline).await?;
1187        }

1214        self.finish_prop_transaction(tid, prop::BLE_PAIRING_PIN, PropResponsePolicy::StatusOnly)
1215            .await
1216            .map(|_| ())
1217    }

1219    /// Fetch and decode `PROP_CAPS`.
1220    pub async fn capabilities(&mut self) -> Result<Vec<u32>, CompanionRadioError> {
1221        let raw = self.get_prop(prop::CAPS).await?;
1222        let mut caps = Vec::new();
1223        let mut offset = 0;
1224        while offset < raw.len() {
1225            let (value, used) = pui::decode(&raw[offset..])
1226                .map_err(|_| CompanionRadioError::Protocol("malformed PROP_CAPS"))?;
1227            caps.push(value);
1228            offset += used;
1229        }
1230        Ok(caps)
1231    }

1233    /// Run the spec's post-attach synchronization procedure: fetch the
1234    /// retained `PROP_LAST_STATUS` (detecting a reset since the last
1235    /// contact), the capability list, the configured host identity —
1236    /// yielding an ownership verdict against `expected_host_key` — and
1237    /// the state each advertised capability grants, all in digest form.
1238    ///
1239    /// The host must decide ownership before treating queued data as
1240    /// its own: [`HostOwnership::OtherHost`] means the queue and
1241    /// provisioning belong to another identity.
1242    pub async fn sync(
1243        &mut self,
1244        expected_host_key: Option<&[u8; 32]>,
1245    ) -> Result<NcpSync, CompanionRadioError> {
1246        // Step 1: the retained status, before any other command can
1247        // overwrite a reset code.
1248        let last_status = decode_status(&self.get_prop(prop::LAST_STATUS).await?);
1249        let capabilities = self.capabilities().await?;
1250        let has = |capability: u32| capabilities.contains(&capability);
1251
1252        // Step 2: ownership.
1253        let (host_key, ownership) = if has(cap::HOST_FILTER) {
1254            let value = self.get_prop(prop::HOST_KEY).await?;
1255            match <[u8; 32]>::try_from(value.as_slice()) {
1256                Ok(key) => {
1257                    let ownership = match expected_host_key {
1258                        Some(expected) if *expected == key => HostOwnership::Ours,
1259                        _ => HostOwnership::OtherHost(key),
1260                    };
1261                    (Some(key), ownership)
1262                }
1263                Err(_) if value.is_empty() => (None, HostOwnership::Unclaimed),
1264                Err(_) => return Err(CompanionRadioError::Protocol("malformed PROP_HOST_KEY")),
1265            }
1266        } else {
1267            (None, HostOwnership::Unsupported)
1268        };
1269
1270        // Step 3: the device-domain and host-domain state we depend
1271        // on, gated by the advertised capabilities.
1272        let phy_enabled = self.get_prop(prop::PHY_ENABLED).await? == [1];
1273        let freq = self.get_prop(prop::PHY_FREQ).await?;
1274        let freq_khz = u32::from_le_bytes(
1275            freq.as_slice()
1276                .try_into()
1277                .map_err(|_| CompanionRadioError::Protocol("malformed PROP_PHY_FREQ"))?,
1278        );
1279        let device_name = self.device_name().await?;
1280        let saved = match has(cap::SAVE) {
1281            true => Some(self.get_prop(prop::SAVED).await? == [1]),
1282            false => None,
1283        };
1284        let (queue_count, queue_dropped) = if has(cap::HOST_RX_QUEUE) {
1285            let count = self.get_prop(prop::HOST_RX_QUEUE_COUNT).await?;
1286            let dropped = self.get_prop(prop::HOST_RX_QUEUE_DROPPED).await?;
1287            (
1288                Some(u16::from_le_bytes(count.as_slice().try_into().map_err(
1289                    |_| CompanionRadioError::Protocol("malformed PROP_HOST_RX_QUEUE_COUNT"),
1290                )?)),
1291                Some(u32::from_le_bytes(dropped.as_slice().try_into().map_err(
1292                    |_| CompanionRadioError::Protocol("malformed PROP_HOST_RX_QUEUE_DROPPED"),
1293                )?)),
1294            )
1295        } else {
1296            (None, None)
1297        };
1298        let filters = match has(cap::HOST_FILTER) {
1299            true => Some(decode_filter_table(
1300                &self.get_prop(prop::HOST_RX_FILTERS).await?,
1301            )?),
1302            false => None,
1303        };
1304        let (host_channel_ids, host_peer_keys) = if has(cap::HOST_KEYS) {
1305            (
1306                Some(decode_fixed_list::<{ items::CHANNEL_ID_LEN }>(
1307                    &self.get_prop(prop::HOST_CHANNEL_KEYS).await?,
1308                    "malformed PROP_HOST_CHANNEL_KEYS digest",
1309                )?),
1310                Some(decode_fixed_list::<{ items::PUBLIC_KEY_LEN }>(
1311                    &self.get_prop(prop::HOST_PEER_KEYS).await?,
1312                    "malformed PROP_HOST_PEER_KEYS digest",
1313                )?),
1314            )
1315        } else {
1316            (None, None)
1317        };
1318        let auto_ack = match has(cap::HOST_AUTO_ACK) {
1319            true => Some(self.get_prop(prop::HOST_AUTO_ACK).await? == [1]),
1320            false => None,
1321        };
1322        let dev_key = if has(cap::DEV_IDENTITY) {
1323            let value = self.get_prop(prop::DEV_KEY).await?;
1324            match <[u8; 32]>::try_from(value.as_slice()) {
1325                Ok(key) => Some(key),
1326                Err(_) if value.is_empty() => None,
1327                Err(_) => return Err(CompanionRadioError::Protocol("malformed PROP_DEV_KEY")),
1328            }
1329        } else {
1330            None
1331        };
1332
1333        Ok(NcpSync {
1334            reset_since_last_contact: last_status.is_reset(),
1335            last_status,
1336            capabilities,
1337            ownership,
1338            host_key,
1339            phy_enabled,
1340            freq_khz,
1341            device_name,
1342            saved,
1343            queue_count,
1344            queue_dropped,
1345            filters,
1346            host_channel_ids,
1347            host_peer_keys,
1348            auto_ack,
1349            dev_key,
1350        })
1351    }

1353    /// Establish `desired` as the NCP's host domain, reconciling
1354    /// against the digest forms so secrets the NCP already holds are
1355    /// never re-sent. A `host_key` differing from the configured one
1356    /// replaces the whole host domain first (spec §Host Replacement).
1357    ///
1358    /// Each individual write is transactional on the NCP (spec
1359    /// §Mutation Atomicity). Changes are live only: follow with
1360    /// [`Self::save`] to persist the provisioned state for autonomous
1361    /// operation.
1362    pub async fn provision(
1363        &mut self,
1364        desired: &HostProvisioning,
1365    ) -> Result<ProvisionReport, CompanionRadioError> {
1366        let mut report = ProvisionReport::default();
1367        let current_key = self.get_prop(prop::HOST_KEY).await?;
1368        if current_key.as_slice() != desired.host_key.as_slice() {
1369            self.set_prop(prop::HOST_KEY, &desired.host_key).await?;
1370            report.host_replaced = true;
1371        }
1372
1373        // Filters: item and digest forms are identical, so replace the
1374        // whole table (atomically, no secrets involved) when the sets
1375        // differ.
1376        let current_filters = if report.host_replaced {
1377            Vec::new()
1378        } else {
1379            decode_filter_table(&self.get_prop(prop::HOST_RX_FILTERS).await?)?
1380        };
1381        if !same_set(&current_filters, &desired.filters) {
1382            let mut table = Vec::new();
1383            for filter in &desired.filters {
1384                let mut item = [0u8; items::Filter::MAX_WIRE_LEN];
1385                let item_len = filter
1386                    .encode(&mut item)
1387                    .map_err(|_| CompanionRadioError::Protocol("filter encode"))?;
1388                let mut prefixed = [0u8; items::Filter::MAX_WIRE_LEN + 2];
1389                let prefixed_len = items::encode_prefixed_item(&item[..item_len], &mut prefixed)
1390                    .map_err(|_| CompanionRadioError::Protocol("filter encode"))?;
1391                table.extend_from_slice(&prefixed[..prefixed_len]);
1392            }
1393            self.set_prop(prop::HOST_RX_FILTERS, &table).await?;
1394            report.filters_replaced = true;
1395        }
1396
1397        // Channel keys are compared through their derived identifiers.
1398        let engine = CryptoEngine::new(SoftwareAes, SoftwareSha256);
1399        let desired_ids: Vec<[u8; items::CHANNEL_ID_LEN]> = desired
1400            .channel_keys
1401            .iter()
1402            .map(|key| engine.derive_channel_id(&ChannelKey(*key)).0)
1403            .collect();
1404        let current_ids = if report.host_replaced {
1405            Vec::new()
1406        } else {
1407            decode_fixed_list::<{ items::CHANNEL_ID_LEN }>(
1408                &self.get_prop(prop::HOST_CHANNEL_KEYS).await?,
1409                "malformed PROP_HOST_CHANNEL_KEYS digest",
1410            )?
1411        };
1412        if current_ids.iter().any(|id| !desired_ids.contains(id)) {
1413            // The NCP holds a channel we have no key for; its remove
1414            // selector is the key itself, so the only way to shed it
1415            // is an atomic whole-table replacement.
1416            let table: Vec<u8> = desired.channel_keys.concat();
1417            self.set_prop(prop::HOST_CHANNEL_KEYS, &table).await?;
1418            report.channels_replaced = true;
1419        } else {
1420            for (key, id) in desired.channel_keys.iter().zip(&desired_ids) {
1421                if !current_ids.contains(id) {
1422                    self.insert_prop_item(prop::HOST_CHANNEL_KEYS, key).await?;
1423                    report.channels_inserted += 1;
1424                }
1425            }
1426        }
1427
1428        // Peers reconcile by public-key membership; the pairwise key
1429        // material of peers the NCP already reports never crosses the
1430        // link again.
1431        let current_peers = if report.host_replaced {
1432            Vec::new()
1433        } else {
1434            decode_fixed_list::<{ items::PUBLIC_KEY_LEN }>(
1435                &self.get_prop(prop::HOST_PEER_KEYS).await?,
1436                "malformed PROP_HOST_PEER_KEYS digest",
1437            )?
1438        };
1439        for entry in &desired.peer_keys {
1440            if !current_peers.contains(&entry.public_key) {
1441                let mut item = [0u8; items::PeerKeyEntry::WIRE_LEN];
1442                entry
1443                    .encode(&mut item)
1444                    .map_err(|_| CompanionRadioError::Protocol("peer entry encode"))?;
1445                self.insert_prop_item(prop::HOST_PEER_KEYS, &item).await?;
1446                report.peers_inserted += 1;
1447            }
1448        }
1449        for existing in &current_peers {
1450            if !desired
1451                .peer_keys
1452                .iter()
1453                .any(|entry| entry.public_key == *existing)
1454            {
1455                self.remove_prop_item(prop::HOST_PEER_KEYS, existing)
1456                    .await?;
1457                report.peers_removed += 1;
1458            }
1459        }
1460
1461        // Delegation policy last, once the keys it depends on exist.
1462        let current_auto_ack = if report.host_replaced {
1463            false
1464        } else {
1465            self.get_prop(prop::HOST_AUTO_ACK).await? == [1]
1466        };
1467        if current_auto_ack != desired.auto_ack {
1468            self.set_prop(prop::HOST_AUTO_ACK, &[desired.auto_ack as u8])
1469                .await?;
1470            report.auto_ack_changed = true;
1471        }
1472        Ok(report)
1473    }

1475    /// The NCP's device identity public key, generating one on-device
1476    /// if none is configured (`CAP_DEV_IDENTITY`; generation requires
1477    /// the transport's provisioning-security binding).
1478    ///
1479    /// On-device generation is the spec-recommended form: the private
1480    /// key never exists anywhere but the radio, and only the resulting
1481    /// public key crosses the link.
1482    pub async fn ensure_device_identity(&mut self) -> Result<[u8; 32], CompanionRadioError> {
1483        let current = self.get_prop(prop::DEV_KEY).await?;
1484        if let Ok(key) = <[u8; 32]>::try_from(current.as_slice()) {
1485            return Ok(key);
1486        }
1487        if !current.is_empty() {
1488            return Err(CompanionRadioError::Protocol("malformed PROP_DEV_KEY"));
1489        }
1490        // An empty PROP_DEV_PRIVATE_KEY write commands generation;
1491        // success is announced as PROP_IS for PROP_DEV_KEY carrying
1492        // the new public key.
1493        let tid = self.alloc_tid();
1494        let mut buf = [0u8; 8];
1495        let len = frame::prop_set(&mut buf, tid, prop::DEV_PRIVATE_KEY, &[])
1496            .map_err(|_| CompanionRadioError::Protocol("frame encode"))?;
1497        self.send(&buf[..len]).await?;
1498        let value = self
1499            .finish_prop_transaction(tid, prop::DEV_KEY, PropResponsePolicy::Value)
1500            .await?;
1501        <[u8; 32]>::try_from(value.as_slice())
1502            .map_err(|_| CompanionRadioError::Protocol("malformed PROP_DEV_KEY"))
1503    }

1505    async fn finish_prop_transaction(
1506        &mut self,
1507        tid: u8,
1508        key: u32,
1509        policy: PropResponsePolicy,
1510    ) -> Result<Vec<u8>, CompanionRadioError> {
1511        let deadline = Instant::now() + self.config.response_timeout;
1512        let response = self.wait_response(tid, deadline).await?;
1513        if response.kind != ResponseKind::Is {
1514            return Err(CompanionRadioError::Protocol(
1515                "table notification answering a property command",
1516            ));
1517        }
1518        match (policy, response.key) {
1519            (PropResponsePolicy::Value, response_key) if response_key == key => Ok(response.value),
1520            (PropResponsePolicy::StatusOnly, prop::LAST_STATUS) => {
1521                let status = decode_status(&response.value);
1522                if status == Status::OK {
1523                    Ok(Vec::new())
1524                } else {
1525                    Err(CompanionRadioError::Status(status))
1526                }
1527            }
1528            (PropResponsePolicy::Value, prop::LAST_STATUS) => {
1529                let status = decode_status(&response.value);
1530                if status == Status::OK {
1531                    Err(CompanionRadioError::Protocol(
1532                        "unexpected status-only property response",
1533                    ))
1534                } else {
1535                    Err(CompanionRadioError::Status(status))
1536                }
1537            }
1538            _ => Err(CompanionRadioError::Protocol(
1539                "response for unexpected property",
1540            )),
1541        }
1542    }

1544    /// Complete a `CMD_PROP_INSERT`/`CMD_PROP_REMOVE` transaction:
1545    /// success is the matching item notification carrying the digest,
1546    /// failure a correlated `PROP_LAST_STATUS`.
1547    async fn finish_table_transaction(
1548        &mut self,
1549        tid: u8,
1550        key: u32,
1551        expected: ResponseKind,
1552    ) -> Result<Vec<u8>, CompanionRadioError> {
1553        let deadline = Instant::now() + self.config.response_timeout;
1554        let response = self.wait_response(tid, deadline).await?;
1555        match (response.kind, response.key) {
1556            (kind, response_key) if kind == expected && response_key == key => Ok(response.value),
1557            (ResponseKind::Is, prop::LAST_STATUS) => {
1558                let status = decode_status(&response.value);
1559                if status == Status::OK {
1560                    Err(CompanionRadioError::Protocol(
1561                        "status-only success for a table mutation",
1562                    ))
1563                } else {
1564                    Err(CompanionRadioError::Status(status))
1565                }
1566            }
1567            _ => Err(CompanionRadioError::Protocol(
1568                "response for unexpected property",
1569            )),
1570        }
1571    }

1579    /// Sort a complete companion frame into the receive queue, response queue,
1580    /// or the reset flag; returns whether a stream frame was queued
1581    /// (the back of `rx_queue` is then the new packet). Malformed
1582    /// frames are dropped.
1583    fn ingest_frame(&mut self, frame_bytes: &[u8]) -> bool {
1584        if let Some(trace) = &mut self.trace {
1585            trace(TraceDirection::NcpToHost, &describe_frame(frame_bytes));
1586        }
1587        let Ok(frame) = Frame::parse(frame_bytes) else {
1588            return false;
1589        };
1590        match frame.command() {
1591            Some(Cmd::StrRecv) => {
1592                let Ok(payload) = StreamPayload::parse(frame.payload) else {
1593                    return false;
1594                };
1595                if payload.stream != stream::PHY_RAW {
1596                    return false;
1597                }
1598                let meta = RxMeta::decode(payload.metadata).unwrap_or_default();
1599                if self.rx_queue.len() >= RX_QUEUE_DEPTH {
1600                    self.rx_queue.pop_front();
1601                }
1602                self.rx_queue.push_back(RxPacket {
1603                    data: payload.data.to_vec(),
1604                    meta,
1605                    raw_meta: payload.metadata.to_vec(),
1606                });
1607                return true;
1608            }
1609            Some(Cmd::PropIs) => self.ingest_prop_notification(ResponseKind::Is, &frame),
1610            Some(Cmd::PropInserted) => {
1611                self.ingest_prop_notification(ResponseKind::Inserted, &frame)
1612            }
1613            Some(Cmd::PropRemoved) => self.ingest_prop_notification(ResponseKind::Removed, &frame),
1614            _ => {}
1615        }
1616        false
1617    }
1618
1619    fn ingest_prop_notification(&mut self, kind: ResponseKind, frame: &Frame<'_>) {
1620        let Ok(payload) = PropPayload::parse(frame.payload) else {
1621            return;
1622        };
1623        let tid = frame.header.tid();
1624        if tid != TID_UNSOLICITED {
1625            if self.responses.len() >= RESPONSE_QUEUE_DEPTH {
1626                self.responses.pop_front();
1627            }
1628            self.responses.push_back(Response {
1629                tid,
1630                kind,
1631                key: payload.key,
1632                value: payload.value.to_vec(),
1633            });
1634            return;
1635        }
1636        // Unsolicited `PROP_LAST_STATUS` is a reset notice or an
1637        // operation status, not a property update to retain.
1638        if kind == ResponseKind::Is && payload.key == prop::LAST_STATUS {
1639            let status = decode_status(payload.value);
1640            if status.is_reset() {
1641                self.seen_reset = Some(status);
1642            }
1643            return;
1644        }
1645        let event = match kind {
1646            ResponseKind::Is => PropEvent::Is {
1647                key: payload.key,
1648                value: payload.value.to_vec(),
1649            },
1650            ResponseKind::Inserted => PropEvent::Inserted {
1651                key: payload.key,
1652                digest: payload.value.to_vec(),
1653            },
1654            ResponseKind::Removed => PropEvent::Removed {
1655                key: payload.key,
1656                digest: payload.value.to_vec(),
1657            },
1658        };
1659        if self.prop_events.len() >= PROP_EVENT_DEPTH {
1660            self.prop_events.pop_front();
1661        }
1662        self.prop_events.push_back(event);
1663    }

1668    /// at [`PROP_EVENT_DEPTH`], oldest dropped first).

1673    /// Read from the stream until the response for `tid` arrives.
1674    ///
1675    /// Frames received meanwhile are queued for [`Radio::poll_receive`].
1676    async fn wait_response(
1677        &mut self,
1678        tid: u8,
1679        deadline: Instant,
1680    ) -> Result<Response, CompanionRadioError> {
1681        loop {
1682            // Drain responses before honoring a reset notice: if both
1683            // arrived in one read, the response was sent first and the
1684            // command did complete. The reset stays latched for the
1685            // next receive poll.
1686            while let Some(response) = self.responses.pop_front() {
1687                if response.tid == tid {
1688                    return Ok(response);
1689                }
1690                // A stale response from an earlier timed-out
1691                // transaction; drop it.
1692            }
1693            if let Some(status) = self.seen_reset.take() {
1694                return Err(CompanionRadioError::UnexpectedReset(status));
1695            }
1696            self.read_more(deadline).await?;
1697        }
1698    }

1700    /// Read until the NCP announces a reset via `PROP_LAST_STATUS`.
1701    async fn wait_reset(&mut self, deadline: Instant) -> Result<Status, CompanionRadioError> {
1702        loop {
1703            if let Some(status) = self.seen_reset.take() {
1704                return Ok(status);
1705            }
1706            // Accept a reset notice even if the NCP attached a TID.
1707            while let Some(response) = self.responses.pop_front() {
1708                if response.kind == ResponseKind::Is && response.key == prop::LAST_STATUS {
1709                    let status = decode_status(&response.value);
1710                    if status.is_reset() {
1711                        return Ok(status);
1712                    }
1713                }
1714            }
1715            self.read_more(deadline).await?;
1716        }
1717    }
1718
1719    /// Read and ingest one frame before `deadline`; returns whether it
1720    /// was a stream frame that is now the back of `rx_queue`.
1721    async fn read_more(&mut self, deadline: Instant) -> Result<bool, CompanionRadioError> {
1722        let now = Instant::now();
1723        if now >= deadline {
1724            return Err(CompanionRadioError::Timeout);
1725        }
1726        let frame = match tokio::time::timeout(deadline - now, self.link.recv_frame()).await {
1727            Err(_elapsed) => return Err(CompanionRadioError::Timeout),
1728            Ok(Err(error)) => return Err(error),
1729            Ok(Ok(frame)) => frame,
1730        };
1731        Ok(self.ingest_frame(&frame))
1732    }
1733
1734    fn pop_rx(&mut self, buf: &mut [u8]) -> Option<RxInfo> {
1735        let packet = self.rx_queue.pop_front()?;
1736        let len = packet.data.len().min(buf.len());
1737        buf[..len].copy_from_slice(&packet.data[..len]);
1738        Some(RxInfo {
1739            len,
1740            rssi: packet.meta.rssi_dbm.unwrap_or(0),
1741            snr: Snr::from_centibels(packet.meta.snr_cb.unwrap_or(0)),
1742            lqi: packet.meta.lqi,
1743        })
1744    }

1791    /// Transmit one frame and await the NCP's confirmation.
1792    ///
1793    /// A confirmed transmit blocks the caller for up to
1794    /// `response_timeout + 2 × t_frame_ms` while the frame goes out on air. This
1795    /// is inherent to the half-duplex [`Radio::transmit`] contract and a real
1796    /// radio behaves the same way. Frames the NCP receives during this window
1797    /// are not lost — they are queued (see [`wait_response`](Self::wait_response)
1798    /// → [`ingest`](Self::ingest)) and surface on the next
1799    /// [`poll_receive`](Radio::poll_receive). MAC-layer timers (ACK timeouts,
1800    /// retransmit deadlines) cannot advance while this future is pending, but
1801    /// they are only *delayed*, not missed: the coordinator re-evaluates every
1802    /// deadline against the current clock as soon as `transmit` returns, so a
1803    /// deadline that came due mid-transmit fires immediately afterward.
1804    async fn transmit(
1805        &mut self,
1806        data: &[u8],
1807        options: TxOptions,
1808    ) -> Result<(), TxError<Self::Error>> {
1809        if data.len() > self.max_frame_size {
1810            return Err(TxError::Io(CompanionRadioError::FrameTooLarge(data.len())));
1811        }
1812
1813        // The NCP performs CCA itself; the CAD policy becomes a host-side
1814        // retry budget around `STATUS_CCA_FAILURE`.
1815        let mut meta = TxMeta::default();
1816        let cca_deadline = match options.cad {
1817            CadPolicy::Skip => {
1818                meta.flags |= TX_FLAG_NOCCA;
1819                None
1820            }
1821            // Gate is a single attempt: a zero-length budget, so a busy channel
1822            // fails immediately with CadTimeout.
1823            CadPolicy::Gate => Some(Instant::now()),
1824            CadPolicy::RetryFor { timeout_ms } => {
1825                Some(Instant::now() + Duration::from_millis(timeout_ms.into()))
1826            }
1827        };
1828        let mut meta_buf = [0u8; TxMeta::WIRE_LEN];
1829        let meta_len = meta
1830            .encode(&mut meta_buf)
1831            .expect("buffer sized with WIRE_LEN");
1832
1833        loop {
1834            let tid = self.alloc_tid();
1835            let mut frame_buf = vec![0u8; data.len() + 16];
1836            let frame_len = frame::str_send(
1837                &mut frame_buf,
1838                tid,
1839                stream::PHY_RAW,
1840                data,
1841                &meta_buf[..meta_len],
1842            )
1843            .map_err(|_| TxError::Io(CompanionRadioError::Protocol("frame encode")))?;
1844            self.send(&frame_buf[..frame_len])
1845                .await
1846                .map_err(TxError::Io)?;
1847
1848            // The confirmation arrives only after the frame is on the
1849            // air (or definitively failed), so allow for airtime.
1850            let deadline = Instant::now()
1851                + self.config.response_timeout
1852                + Duration::from_millis(u64::from(self.t_frame_ms) * 2);
1853            let response = self
1854                .wait_response(tid, deadline)
1855                .await
1856                .map_err(TxError::Io)?;
1857            if response.kind != ResponseKind::Is || response.key != prop::LAST_STATUS {
1858                return Err(TxError::Io(CompanionRadioError::Protocol(
1859                    "unexpected transmit response",
1860                )));
1861            }
1862            match decode_status(&response.value) {
1863                Status::OK => return Ok(()),
1864                Status::CCA_FAILURE => match cca_deadline {
1865                    Some(deadline) if Instant::now() < deadline => {
1866                        tokio::time::sleep(CCA_RETRY_DELAY).await;
1867                    }
1868                    _ => return Err(TxError::CadTimeout),
1869                },
1870                status => return Err(TxError::Io(CompanionRadioError::Status(status))),
1871            }
1872        }
1873    }
1874
1875    fn poll_receive(
1876        &mut self,
1877        cx: &mut core::task::Context<'_>,
1878        buf: &mut [u8],
1879    ) -> core::task::Poll<Result<RxInfo, Self::Error>> {
1880        loop {
1881            if let Some(status) = self.seen_reset.take() {
1882                return core::task::Poll::Ready(Err(CompanionRadioError::UnexpectedReset(status)));
1883            }
1884            if let Some(info) = self.pop_rx(buf) {
1885                return core::task::Poll::Ready(Ok(info));
1886            }
1887
1888            match self.link.poll_recv_frame(cx) {
1889                core::task::Poll::Ready(Ok(frame)) => {
1890                    self.ingest_frame(&frame);
1891                }
1892                core::task::Poll::Ready(Err(error)) => return core::task::Poll::Ready(Err(error)),
1893                core::task::Poll::Pending => return core::task::Poll::Pending,
1894            }
1895        }
1896    }

1984/// Render one companion frame as a one-line human-readable summary:
1985/// command, TID, property mnemonic, and the decoded status where the
1986/// payload is a `PROP_LAST_STATUS` value. Values are summarized by
1987/// length — never dumped — so traces cannot leak key material.
1988pub fn describe_frame(bytes: &[u8]) -> String {
1989    let Ok(frame) = Frame::parse(bytes) else {
1990        return format!("malformed frame ({} bytes)", bytes.len());
1991    };
1992    let tid = frame.header.tid();
1993    let Some(command) = frame.command() else {
1994        return format!("unknown command tid={tid} ({} bytes)", bytes.len());
1995    };
1996    match command {
1997        Cmd::Nop
1998        | Cmd::Reset
1999        | Cmd::QueueDrain
2000        | Cmd::Save
2001        | Cmd::Clear
2002        | Cmd::Restore => format!("{command:?} tid={tid}"),
2003        Cmd::PropGet
2004        | Cmd::PropSet
2005        | Cmd::PropIs
2006        | Cmd::PropInsert
2007        | Cmd::PropRemove
2008        | Cmd::PropInserted
2009        | Cmd::PropRemoved => {
2010            let Ok(payload) = PropPayload::parse(frame.payload) else {
2011                return format!("{command:?} tid={tid} (malformed payload)");
2012            };
2013            let key = prop_name(payload.key)
2014                .map_or_else(|| format!("prop {}", payload.key), str::to_owned);
2015            if payload.key == prop::LAST_STATUS && command == Cmd::PropIs {
2016                format!(
2017                    "{command:?} tid={tid} {key} = {:?}",
2018                    decode_status(payload.value)
2019                )
2020            } else {
2021                format!(
2022                    "{command:?} tid={tid} {key} ({} value bytes)",
2023                    payload.value.len()
2024                )
2025            }
2026        }
2027        Cmd::StrSend | Cmd::StrRecv => match StreamPayload::parse(frame.payload) {
2028            Ok(payload) => format!(
2029                "{command:?} tid={tid} stream={} ({} data bytes, {} meta bytes)",
2030                payload.stream,
2031                payload.data.len(),
2032                payload.metadata.len()
2033            ),
2034            Err(_) => format!("{command:?} tid={tid} (malformed payload)"),
2035        },
2036    }
