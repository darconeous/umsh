11/// Text-message rendering type.
12#[derive(Clone, Copy, Debug, PartialEq, Eq)]
13pub enum MessageType {
14    Basic = 0,
15    Status = 1,
16    ResendRequest = 2,
17}
18
19impl MessageType {
20    fn from_byte(value: u8) -> Result<Self, ParseError> {
21        match value {
22            0 => Ok(Self::Basic),
23            1 => Ok(Self::Status),
24            2 => Ok(Self::ResendRequest),
25            _ => Err(ParseError::InvalidMessageType(value)),
26        }
27    }
28}

109pub fn parse(payload: &[u8]) -> Result<TextMessage<'_>, ParseError> {
110    let mut decoder = OptionDecoder::new(payload);
111    let mut message_type = MessageType::Basic;
112    let mut sender_handle = None;
113    let mut sequence = None;
114    let mut sequence_reset = false;
115    let mut regarding = None;
116    let mut editing = None;
117    let mut bg_color = None;
118    let mut text_color = None;
119
120    while let Some(item) = decoder.next() {
121        let (number, value) = item?;
122        match number {
123            0 => {
124                message_type = if value.is_empty() {
125                    MessageType::Basic
126                } else if value.len() == 1 {
127                    MessageType::from_byte(value[0])?
128                } else {
129                    return Err(ParseError::InvalidOptionValue);
130                };
131            }
132            1 => sender_handle = Some(parse_utf8(value)?),
133            2 => {
134                sequence = Some(match value {
135                    [message_id] => MessageSequence {
136                        message_id: *message_id,
137                        fragment: None,
138                    },
139                    [message_id, index, count] if *count >= 2 => MessageSequence {
140                        message_id: *message_id,
141                        fragment: Some(Fragment {
142                            index: *index,
143                            count: *count,
144                        }),
145                    },
146                    _ => return Err(ParseError::InvalidOptionValue),
147                });
148            }
149            3 => {
150                if !value.is_empty() {
151                    return Err(ParseError::InvalidOptionValue);
152                }
153                sequence_reset = true;
154            }
155            4 => {
156                regarding = Some(match value {
157                    [message_id] => Regarding::Unicast {
158                        message_id: *message_id,
159                    },
160                    [message_id, a, b, c] => Regarding::Multicast {
161                        message_id: *message_id,
162                        source_prefix: NodeHint([*a, *b, *c]),
163                    },
164                    _ => return Err(ParseError::InvalidOptionValue),
165                });
166            }
167            5 => {
168                editing = match value {
169                    [message_id] => Some(*message_id),
170                    _ => return Err(ParseError::InvalidOptionValue),
171                };
172            }
173            6 => {
174                bg_color = match value {
175                    [r, g, b] => Some([*r, *g, *b]),
176                    _ => return Err(ParseError::InvalidOptionValue),
177                };
178            }
179            7 => {
180                text_color = match value {
181                    [r, g, b] => Some([*r, *g, *b]),
182                    _ => return Err(ParseError::InvalidOptionValue),
183                };
184            }
185            _ => {}
186        }
187    }
188
189    let body = parse_utf8(decoder.remainder())?;
190    Ok(TextMessage {
191        message_type,
192        sender_handle,
193        sequence,
194        sequence_reset,
195        regarding,
196        editing,
197        bg_color,
198        text_color,
199        body,
200    })
201}

203pub fn encode(msg: &TextMessage<'_>, buf: &mut [u8]) -> Result<usize, EncodeError> {
204    let mut encoder = OptionEncoder::new(buf);
205
206    if msg.message_type != MessageType::Basic {
207        encoder.put(0, &[msg.message_type as u8])?;
208    }
209    if let Some(handle) = msg.sender_handle {
210        encoder.put(1, handle.as_bytes())?;
211    }
212    if let Some(sequence) = msg.sequence {
213        let mut seq_buf = [0u8; 3];
214        let seq_len = if let Some(fragment) = sequence.fragment {
215            if fragment.count < 2 {
216                return Err(EncodeError::InvalidField);
217            }
218            seq_buf = [sequence.message_id, fragment.index, fragment.count];
219            3
220        } else {
221            seq_buf[0] = sequence.message_id;
222            1
223        };
224        encoder.put(2, &seq_buf[..seq_len])?;
225    }
226    if msg.sequence_reset {
227        encoder.put(3, &[])?;
228    }
229    if let Some(regarding) = msg.regarding {
230        let mut regarding_buf = [0u8; 4];
231        let regarding_len = match regarding {
232            Regarding::Unicast { message_id } => {
233                regarding_buf[0] = message_id;
234                1
235            }
236            Regarding::Multicast {
237                message_id,
238                source_prefix,
239            } => {
240                regarding_buf = [
241                    message_id,
242                    source_prefix.0[0],
243                    source_prefix.0[1],
244                    source_prefix.0[2],
245                ];
246                4
247            }
248        };
249        encoder.put(4, &regarding_buf[..regarding_len])?;
250    }
251    if let Some(editing) = msg.editing {
252        encoder.put(5, &[editing])?;
253    }
254    if let Some(color) = msg.bg_color {
255        encoder.put(6, &color)?;
256    }
257    if let Some(color) = msg.text_color {
258        encoder.put(7, &color)?;
259    }
260
261    encoder.end_marker()?;
262    let prefix_len = encoder.finish();
263    if buf.len().saturating_sub(prefix_len) < msg.body.len() {
264        return Err(EncodeError::BufferTooSmall);
265    }
266    buf[prefix_len..prefix_len + msg.body.len()].copy_from_slice(msg.body.as_bytes());
267    Ok(prefix_len + msg.body.len())
268}
