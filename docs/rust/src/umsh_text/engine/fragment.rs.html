1//! Outbound fragmentation planning and fixed-capacity reassembly.
2//!
3//! Reassembly storage is one global page pool shared by all conversations: a
4//! fragment's bytes are stored in linked 80-byte pages, and a slot holds only
5//! its key, presence bitmaps, per-fragment lengths/page heads, first-fragment
6//! metadata, and deadlines. A conversation with no incomplete fragmented
7//! message consumes no page storage.
8
9use crate::engine::sequence::{MessageHandle, StreamKey};
10use crate::model::{FRAGMENT_BODY_MAX, FRAGMENT_COUNT_MAX, MessageType, Regarding};
11
12pub const PAGE_SIZE: usize = 80;
13const PAGE_NONE: u8 = 0xFF;
14
15/// Message-level metadata retained from fragment zero (spec: options from the
16/// first fragment apply to the entire reassembled message).
17///
18/// Only fields consulted on *later* receive calls are retained. Presentation
19/// metadata (sender handle, colors) is not: the announcing Insert mutation is
20/// always emitted during the call that delivered fragment zero, so it borrows
21/// those directly from the validated content, at full fidelity.
22#[derive(Clone, Copy, Debug, Default)]
23pub struct FirstMeta {
24    pub message_type_byte: u8,
25    pub regarding: Option<Regarding>,
26    pub editing: Option<u8>,
27}
28
29impl FirstMeta {
30    pub fn message_type(&self) -> MessageType {
31        MessageType::from_byte(self.message_type_byte)
32    }
33}

35/// One in-progress reassembly.
36#[derive(Clone, Debug)]
37pub struct Slot {
38    pub stream: StreamKey,
39    pub epoch: u16,
40    pub message_id: u8,
41    pub count: u8,
42    /// Bitmap of fragments whose bytes are stored.
43    pub present: u16,
44    /// Bitmap of fragments the sender reported unavailable.
45    pub unavailable: u16,
46    /// Missing fragments whose automatic repair budget was exhausted. This
47    /// is separate from `unavailable`: the body remains pending until the
48    /// reassembly TTL, but the scheduler must not reset and retry forever.
49    pub repair_exhausted: u16,
50    pub frag_len: [u8; FRAGMENT_COUNT_MAX as usize],
51    frag_head: [u8; FRAGMENT_COUNT_MAX as usize],
52    pub meta: FirstMeta,
53    pub have_meta: bool,
54    /// An Insert mutation has been emitted for this slot's handle.
55    pub announced: bool,
56    pub handle: MessageHandle,
57    pub created_ms: u64,
58    pub deadline_ms: u64,
59    /// Next time repair scheduling may consider this slot.
60    pub repair_at_ms: u64,
61    /// When the newest fragment was stored; arrivals defer repair from here.
62    pub last_fragment_ms: u64,
63}
64
65impl Slot {
66    pub fn is_complete(&self) -> bool {
67        let all = (1u16 << self.count) - 1;
68        self.present == all
69    }
70
71    /// Every fragment is either present or reported unavailable, so no
72    /// further repair can improve this slot.
73    pub fn is_settled(&self) -> bool {
74        let all = (1u16 << self.count) - 1;
75        (self.present | self.unavailable) == all
76    }
77
78    pub fn missing(&self) -> impl Iterator<Item = u8> + '_ {
79        (0..self.count).filter(|index| {
80            let bit = 1u16 << index;
81            self.present & bit == 0 && self.unavailable & bit == 0
82        })
83    }
84
85    pub fn repairable_missing(&self) -> impl Iterator<Item = u8> + '_ {
86        self.missing().filter(|index| {
87            let bit = 1u16 << index;
88            self.repair_exhausted & bit == 0
89        })
90    }
91
92    fn fragment_len(&self, index: u8) -> usize {
93        self.frag_len[index as usize] as usize
94    }
95}

97/// Outcome of storing one received fragment.
98#[derive(Clone, Copy, Debug, PartialEq, Eq)]
99pub enum InsertOutcome {
100    /// Stored; the slot may now be complete.
101    Stored,
102    /// The fragment was already present with identical bytes.
103    Duplicate,
104    /// The fragment was already present with different bytes; the original
105    /// authenticated bytes were kept.
106    Conflict,
107    /// No page capacity; the caller should evict and retry or drop.
108    NoSpace,
109    /// The fragment body exceeds the wire maximum and was not stored.
110    TooLarge,
111}
112
113/// Fixed-capacity reassembly pool.
114pub struct ReassemblyPool<const SLOTS: usize, const PAGES: usize> {
115    pages: [[u8; PAGE_SIZE]; PAGES],
116    next_page: [u8; PAGES],
117    free_head: u8,
118    pub slots: [Option<Slot>; SLOTS],
119}
120
121impl<const SLOTS: usize, const PAGES: usize> ReassemblyPool<SLOTS, PAGES> {
122    pub fn new() -> Self {
123        assert!(PAGES < PAGE_NONE as usize, "page index must fit in u8");
124        let mut next_page = [PAGE_NONE; PAGES];
125        for (index, next) in next_page
126            .iter_mut()
127            .enumerate()
128            .take(PAGES.saturating_sub(1))
129        {
130            *next = (index + 1) as u8;
131        }
132        Self {
133            pages: [[0; PAGE_SIZE]; PAGES],
134            next_page,
135            free_head: if PAGES == 0 { PAGE_NONE } else { 0 },
136            slots: [const { None }; SLOTS],
137        }
138    }
139
140    fn free_pages(&self) -> usize {
141        let mut count = 0;
142        let mut cursor = self.free_head;
143        while cursor != PAGE_NONE {
144            count += 1;
145            cursor = self.next_page[cursor as usize];
146        }
147        count
148    }
149
150    fn alloc_chain(&mut self, len: usize) -> Option<u8> {
151        let needed = len.div_ceil(PAGE_SIZE).max(1);
152        if self.free_pages() < needed {
153            return None;
154        }
155        let head = self.free_head;
156        let mut cursor = head;
157        for _ in 1..needed {
158            cursor = self.next_page[cursor as usize];
159        }
160        self.free_head = self.next_page[cursor as usize];
161        self.next_page[cursor as usize] = PAGE_NONE;
162        Some(head)
163    }
164
165    fn free_chain(&mut self, head: u8) {
166        if head == PAGE_NONE {
167            return;
168        }
169        let mut cursor = head;
170        while self.next_page[cursor as usize] != PAGE_NONE {
171            cursor = self.next_page[cursor as usize];
172        }
173        self.next_page[cursor as usize] = self.free_head;
174        self.free_head = head;
175    }
176
177    fn write_chain(&mut self, head: u8, bytes: &[u8]) {
178        let mut cursor = head;
179        for chunk in bytes.chunks(PAGE_SIZE) {
180            self.pages[cursor as usize][..chunk.len()].copy_from_slice(chunk);
181            cursor = self.next_page[cursor as usize];
182        }
183    }
184
185    fn read_chain(&self, head: u8, len: usize, out: &mut [u8]) {
186        let mut cursor = head;
187        let mut offset = 0;
188        while offset < len {
189            let take = (len - offset).min(PAGE_SIZE);
190            out[offset..offset + take].copy_from_slice(&self.pages[cursor as usize][..take]);
191            offset += take;
192            cursor = self.next_page[cursor as usize];
193        }
194    }
195
196    /// Copy a stored fragment into `out` (which must hold
197    /// [`FRAGMENT_BODY_MAX`] bytes), returning its length.
198    pub fn read_fragment(&self, slot_index: usize, fragment: u8, out: &mut [u8]) -> usize {
199        let slot = self.slots[slot_index].as_ref().expect("occupied slot");
200        let len = slot.fragment_len(fragment);
201        self.read_chain(slot.frag_head[fragment as usize], len, out);
202        len
203    }
204
205    pub fn find_slot(&self, stream: &StreamKey, epoch: u16, message_id: u8) -> Option<usize> {
206        self.slots.iter().position(|slot| {
207            slot.as_ref().is_some_and(|slot| {
208                slot.stream == *stream && slot.epoch == epoch && slot.message_id == message_id
209            })
210        })
211    }
212
213    pub fn open_slot(&mut self, slot: Slot) -> Option<usize> {
214        let index = self.slots.iter().position(Option::is_none)?;
215        self.slots[index] = Some(slot);
216        Some(index)
217    }
218
219    /// Index of the oldest incomplete slot, for eviction under pressure.
220    pub fn oldest_slot(&self) -> Option<usize> {
221        self.slots
222            .iter()
223            .enumerate()
224            .filter_map(|(index, slot)| slot.as_ref().map(|slot| (index, slot.created_ms)))
225            .min_by_key(|(_, created)| *created)
226            .map(|(index, _)| index)
227    }
228
229    /// Release a slot and all its pages, returning it.
230    pub fn close_slot(&mut self, index: usize) -> Option<Slot> {
231        let slot = self.slots[index].take()?;
232        for head in slot.frag_head {
233            self.free_chain(head);
234        }
235        Some(slot)
236    }
237
238    /// Drop every slot belonging to `stream` (after a sequence reset),
239    /// returning how many were dropped.
240    pub fn drop_stream(&mut self, stream: &StreamKey) -> usize {
241        let mut dropped = 0;
242        for index in 0..SLOTS {
243            if self.slots[index]
244                .as_ref()
245                .is_some_and(|slot| slot.stream == *stream)
246            {
247                self.close_slot(index);
248                dropped += 1;
249            }
250        }
251        dropped
252    }
253
254    /// Store one fragment's bytes into a slot.
255    pub fn insert_fragment(&mut self, index: usize, fragment: u8, bytes: &[u8]) -> InsertOutcome {
256        // The wire maximum is enforced at validation; this guard keeps the
257        // pool's u8 lengths and fixed read buffers sound regardless.
258        if bytes.len() > FRAGMENT_BODY_MAX {
259            return InsertOutcome::TooLarge;
260        }
261        let slot = self.slots[index].as_ref().expect("occupied slot");
262        let bit = 1u16 << fragment;
263        if slot.present & bit != 0 {
264            let mut existing = [0u8; FRAGMENT_BODY_MAX];
265            let len = self.read_fragment(index, fragment, &mut existing);
266            return if &existing[..len] == bytes {
267                InsertOutcome::Duplicate
268            } else {
269                InsertOutcome::Conflict
270            };
271        }
272        let Some(head) = self.alloc_chain(bytes.len()) else {
273            return InsertOutcome::NoSpace;
274        };
275        self.write_chain(head, bytes);
276        let slot = self.slots[index].as_mut().expect("occupied slot");
277        slot.frag_head[fragment as usize] = head;
278        slot.frag_len[fragment as usize] = bytes.len() as u8;
279        slot.present |= bit;
280        slot.unavailable &= !bit;
281        InsertOutcome::Stored
282    }
283}

320/// Incremental UTF-8 writer that carries partial code points across fragment
321/// boundaries and replaces invalid bytes with U+FFFD.
322struct LossyWriter<'a> {
323    out: &'a mut [u8],
324    pos: usize,
325    carry: [u8; 4],
326    carry_len: usize,
327    truncated: bool,
328    had_invalid: bool,
329}
330
331impl<'a> LossyWriter<'a> {
332    fn new(out: &'a mut [u8]) -> Self {
333        Self {
334            out,
335            pos: 0,
336            carry: [0; 4],
337            carry_len: 0,
338            truncated: false,
339            had_invalid: false,
340        }
341    }
342
343    fn emit(&mut self, bytes: &[u8]) {
344        let space = self.out.len() - self.pos;
345        if bytes.len() > space {
346            // Truncate at a code-point boundary.
347            let mut take = space;
348            while take > 0 && bytes[take] & 0xC0 == 0x80 {
349                take -= 1;
350            }
351            self.out[self.pos..self.pos + take].copy_from_slice(&bytes[..take]);
352            self.pos += take;
353            self.truncated = true;
354            return;
355        }
356        self.out[self.pos..self.pos + bytes.len()].copy_from_slice(bytes);
357        self.pos += bytes.len();
358    }
359
360    fn emit_replacement(&mut self) {
361        self.had_invalid = true;
362        self.emit("\u{FFFD}".as_bytes());
363    }
364
365    /// Push raw bytes, validating incrementally.
366    fn push(&mut self, mut bytes: &[u8]) {
367        // Complete a carried partial code point first.
368        while self.carry_len > 0 && !bytes.is_empty() {
369            let needed = utf8_len(self.carry[0]).unwrap_or(1);
370            let take = (needed - self.carry_len).min(bytes.len());
371            self.carry[self.carry_len..self.carry_len + take].copy_from_slice(&bytes[..take]);
372            self.carry_len += take;
373            bytes = &bytes[take..];
374            if self.carry_len == needed {
375                let carried = self.carry;
376                let carry_len = self.carry_len;
377                self.carry_len = 0;
378                match core::str::from_utf8(&carried[..carry_len]) {
379                    Ok(_) => self.emit(&carried[..carry_len]),
380                    Err(_) => {
381                        // The lead byte's sequence is invalid; resynchronize
382                        // after the lead byte.
383                        self.emit_replacement();
384                        let rest = carry_len - 1;
385                        let resume: [u8; 4] = carried;
386                        // Reprocess the bytes after the bad lead byte.
387                        self.push_inner(&resume[1..1 + rest]);
388                    }
389                }
390            }
391        }
392        self.push_inner(bytes);
393    }
394
395    fn push_inner(&mut self, mut bytes: &[u8]) {
396        loop {
397            match core::str::from_utf8(bytes) {
398                Ok(text) => {
399                    self.emit(text.as_bytes());
400                    return;
401                }
402                Err(error) => {
403                    let valid = error.valid_up_to();
404                    self.emit(&bytes[..valid]);
405                    match error.error_len() {
406                        Some(bad) => {
407                            self.emit_replacement();
408                            bytes = &bytes[valid + bad..];
409                        }
410                        None => {
411                            // Incomplete trailing sequence: carry it.
412                            let tail = &bytes[valid..];
413                            self.carry[..tail.len()].copy_from_slice(tail);
414                            self.carry_len = tail.len();
415                            return;
416                        }
417                    }
418                }
419            }
420        }
421    }
422
423    /// End a run of contiguous fragments. When the run borders a gap, an
424    /// incomplete trailing code point is discarded (the sentinel covers it);
425    /// at true end of message it is invalid input and becomes U+FFFD.
426    fn end_run(&mut self, at_gap: bool) {
427        if self.carry_len > 0 {
428            if !at_gap {
429                self.emit_replacement();
430            }
431            self.carry_len = 0;
432        }
433    }
434}

436fn utf8_len(lead: u8) -> Option<usize> {
437    match lead {
438        0x00..=0x7F => Some(1),
439        0xC0..=0xDF => Some(2),
440        0xE0..=0xEF => Some(3),
441        0xF0..=0xF4 => Some(4),
442        _ => None,
443    }
444}

464/// Result of rendering a slot.
465#[derive(Clone, Copy, Debug, PartialEq, Eq)]
466pub struct RenderResult {
467    pub len: usize,
468    pub complete: bool,
469    pub had_invalid: bool,
470    pub truncated: bool,
471}
472
473/// Render a slot's fragments into `out` as UTF-8, inserting one sentinel per
474/// absent run and skipping code points damaged by missing byte boundaries.
475pub fn render_slot<const SLOTS: usize, const PAGES: usize>(
476    pool: &ReassemblyPool<SLOTS, PAGES>,
477    slot_index: usize,
478    sentinels: &RenderSentinels,
479    final_render: bool,
480    out: &mut [u8],
481) -> RenderResult {
482    let slot = pool.slots[slot_index].as_ref().expect("occupied slot");
483    let mut writer = LossyWriter::new(out);
484    let mut index = 0u8;
485    let mut after_gap = false;
486    while index < slot.count {
487        let bit = 1u16 << index;
488        if slot.present & bit != 0 {
489            let mut buffer = [0u8; FRAGMENT_BODY_MAX];
490            let len = pool.read_fragment(slot_index, index, &mut buffer);
491            let mut bytes = &buffer[..len];
492            if after_gap {
493                // Skip continuation bytes orphaned by the missing boundary.
494                while let Some((first, rest)) = bytes.split_first() {
495                    if first & 0xC0 == 0x80 {
496                        bytes = rest;
497                    } else {
498                        break;
499                    }
500                }
501                after_gap = false;
502            }
503            writer.push(bytes);
504            index += 1;
505            continue;
506        }
507        // A gap: absent fragments sharing one repair state render as one
508        // sentinel, and the run splits where that state changes, so a
509        // disclaimed portion reads [UNAVAILABLE] immediately even when it
510        // borders a still-repairable one. Sentinel count is bounded by the
511        // fragment count.
512        writer.end_run(true);
513        while index < slot.count && slot.present & (1u16 << index) == 0 {
514            let run_unavailable = slot.unavailable & (1u16 << index) != 0;
515            while index < slot.count
516                && slot.present & (1u16 << index) == 0
517                && (slot.unavailable & (1u16 << index) != 0) == run_unavailable
518            {
519                index += 1;
520            }
521            let sentinel = if run_unavailable {
522                sentinels.unavailable
523            } else if final_render {
524                sentinels.missing
525            } else {
526                sentinels.pending
527            };
528            writer.emit(sentinel.as_bytes());
529        }
530        after_gap = true;
531    }
532    writer.end_run(false);
533    RenderResult {
534        len: writer.pos,
535        complete: slot.is_complete(),
536        had_invalid: writer.had_invalid,
537        truncated: writer.truncated,
538    }
539}

541/// The body exceeds the wire maximum of 10 fragments × 160 bytes.
542#[derive(Clone, Copy, Debug, PartialEq, Eq)]
543pub struct BodyTooLarge;
544
545/// Plan for splitting an outbound body into fragments.
546#[derive(Clone, Copy, Debug, PartialEq, Eq)]
547pub struct FragmentPlan {
548    pub count: u8,
549    body_len: usize,
550}
551
552impl FragmentPlan {
553    /// Plan fragmentation for a body. Returns `None` when the body fits in a
554    /// single frame (given its encoded option overhead) and `Some` plan
555    /// otherwise. Errors when the body exceeds the wire maximum.
556    pub fn plan(body_len: usize, single_frame_budget: usize) -> Result<Option<Self>, BodyTooLarge> {
557        if body_len <= single_frame_budget && body_len <= FRAGMENT_BODY_MAX {
558            return Ok(None);
559        }
560        let count = body_len.div_ceil(FRAGMENT_BODY_MAX);
561        if count > FRAGMENT_COUNT_MAX as usize || count < 2 {
562            if count < 2 {
563                // Options alone exceed the frame; still send as two fragments.
564                return Ok(Some(Self { count: 2, body_len }));
565            }
566            return Err(BodyTooLarge);
567        }
568        Ok(Some(Self {
569            count: count as u8,
570            body_len,
571        }))
572    }
573
574    /// Byte range of fragment `index` within the body.
575    pub fn range(&self, index: u8) -> core::ops::Range<usize> {
576        let per = self
577            .body_len
578            .div_ceil(self.count as usize)
579            .min(FRAGMENT_BODY_MAX);
580        let start = per * index as usize;
581        let end = (start + per).min(self.body_len);
582        start..end
583    }
584}

667        pool.insert_fragment(index, 4, b"z").unwrap_stored();
668        // Fragments 1..=3 are absent; 2 is disclaimed, its neighbors pending.
669        pool.slots[index].as_mut().unwrap().unavailable = 1 << 2;
670        let mut out = [0u8; 128];
671        let result = render_slot(&pool, index, &RenderSentinels::default(), false, &mut out);
672        let text = core::str::from_utf8(&out[..result.len]).unwrap();
673        assert_eq!(text, "a[PENDING][UNAVAILABLE][PENDING]z");
674
675        // At final render, pending sub-runs become missing.
676        let result = render_slot(&pool, index, &RenderSentinels::default(), true, &mut out);
677        let text = core::str::from_utf8(&out[..result.len]).unwrap();
678        assert_eq!(text, "a[MISSING][UNAVAILABLE][MISSING]z");
679    }
