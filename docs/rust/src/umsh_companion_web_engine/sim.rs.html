17const WIRE_CAPACITY: usize = umsh_companion::gatt::MAX_FRAME;
18
19type NcpSession = Session<SoftwareAes, SoftwareSha256>;
20
21/// A deterministic, RAM-backed NCP that speaks HDLC-Lite exactly like USB.
22///
23/// The storage, radio, RSSI sampler, and entropy source are small stand-ins;
24/// every protocol decision is made by the production NCP [`Session`].
25pub struct SimulatedNcp {
26    session: NcpSession,
27    decoder: hdlc::Decoder<WIRE_CAPACITY>,
28    outbound: VecDeque<Vec<u8>>,
29    snapshot: Option<Vec<u8>>,
30    identity: Option<([u8; 32], [u8; 32])>,
31    identity_seed: u8,
32    pairing_pin: Option<u32>,
33    air: Vec<Vec<u8>>,
34    now_ms: u64,
35}
36
37impl Default for SimulatedNcp {
38    fn default() -> Self {
39        Self::new()
40    }
41}
42
43impl SimulatedNcp {
44    pub fn new() -> Self {
45        Self {
46            session: NcpSession::new(
47                session_config(),
48                Status::RESET_POWER_ON,
49                CryptoEngine::new(SoftwareAes, SoftwareSha256),
50            ),
51            decoder: hdlc::Decoder::new(),
52            outbound: VecDeque::new(),
53            snapshot: None,
54            identity: None,
55            identity_seed: 0,
56            pairing_pin: None,
57            air: Vec::new(),
58            now_ms: 0,
59        }
60    }
61
62    /// Attach over the virtual equivalent of a physically secure serial link.
63    pub fn attach(&mut self) {
64        self.decoder.reset();
65        self.outbound.clear();
66        self.session.attach(true);
67    }
68
69    pub fn detach(&mut self) {
70        self.decoder.reset();
71        self.outbound.clear();
72        self.session.detach();
73    }
74
75    /// Feed an arbitrary serial byte chunk into the virtual USB link.
76    pub fn ingest(&mut self, bytes: &[u8], now_ms: u64) -> Result<(), String> {
77        self.now_ms = now_ms;
78        for &byte in bytes {
79            let outcome = self
80                .decoder
81                .push(byte)
82                .map(|result| result.map(<[u8]>::to_vec));
83            if let Some(outcome) = outcome {
84                let frame = outcome.map_err(|error| format!("HDLC decode error: {error:?}"))?;
85                self.handle_frame(&frame);
86            }
87        }
88        Ok(())
89    }
90
91    pub fn take_outbound(&mut self) -> Option<Vec<u8>> {
92        self.outbound.pop_front()
93    }
94
95    /// Put a canned radio frame through the real NCP receive path.
96    pub fn inject_radio_rx(&mut self, bytes: &[u8], now_ms: u64) {
97        self.now_ms = now_ms;
98        let mut emitted = Vec::new();
99        let effect = self
100            .session
101            .on_radio_rx(bytes, -82, 35, None, now_ms, &mut |frame| {
102                emitted.push(frame.to_vec())
103            });
104        self.execute(effect, &mut emitted);
105        self.queue_emitted(emitted);
106    }
107
108    /// Build and inject a small valid UMSH packet for interactive UI demos.
109    pub fn inject_demo_rx(&mut self, now_ms: u64) {
110        let mut bytes = [0; 64];
111        let packet = PacketBuilder::new(&mut bytes)
112            .broadcast()
113            .source_hint(NodeHint([0x11, 0x22, 0x33]))
114            .flood_hops(3)
115            .payload(b"hello from the simulated radio")
116            .build()
117            .expect("fixed demo packet fits")
118            .to_vec();
119        self.inject_radio_rx(&packet, now_ms);
120    }
121
122    #[cfg(test)]
123    fn transmitted_frames(&self) -> &[Vec<u8>] {
124        &self.air
125    }
126
127    fn handle_frame(&mut self, frame: &[u8]) {
128        let mut emitted = Vec::new();
129        let effect = self.session.handle_frame(frame, self.now_ms, &mut |bytes| {
130            emitted.push(bytes.to_vec())
131        });
132        self.execute(effect, &mut emitted);
133        self.queue_emitted(emitted);
134    }
135
136    fn execute(&mut self, effect: Option<Effect>, emitted: &mut Vec<Vec<u8>>) {
137        let mut emit = |frame: &[u8]| emitted.push(frame.to_vec());
138        match effect {
139            None | Some(Effect::ApplyRadio(_)) | Some(Effect::DeviceNameChanged) => {}
140            Some(Effect::StartTransmit) => {
141                self.air.push(self.session.tx_data().to_vec());
142                self.session.on_tx_result(true, self.now_ms, &mut emit);
143            }
144            Some(Effect::SampleRssi { tid }) => {
145                self.session.respond_rssi(tid, Ok(-77), &mut emit);
146            }
147            Some(Effect::SampleBattery { tid }) => {
148                // Stable, human-recognizable simulated measurement
149                // matching the configured field set below.
150                self.session.respond_battery(
151                    tid,
152                    Ok(BatteryStatus {
153                        voltage_mv: Some(4111),
154                        level_percent: Some(87),
155                        charge_state: Some(BatteryChargeState::Charging),
156                    }),
157                    &mut emit,
158                );
159            }
160            Some(Effect::SetPairingPin { tid, pin }) => {
161                self.pairing_pin = pin;
162                self.session.respond_pin_set(tid, Ok(()), &mut emit);
163            }
164            Some(Effect::WipeHostDomain { tid }) => {
165                let mut buf = [0u8; SNAPSHOT_MAX];
166                if let Some(len) = self.session.encode_wiped_snapshot(&mut buf) {
167                    self.snapshot = Some(buf[..len].to_vec());
168                }
169                self.session.respond_host_wipe(tid, Ok(()), &mut emit);
170            }
171            Some(Effect::DrainQueue) => while self.session.drain_step(self.now_ms, &mut emit) {},
172            Some(Effect::SaveSnapshot { tid }) => {
173                let mut buf = [0u8; SNAPSHOT_MAX];
174                let result = match self.session.encode_snapshot(&mut buf) {
175                    Some(len) => {
176                        self.snapshot = Some(buf[..len].to_vec());
177                        Ok(())
178                    }
179                    None => Err(()),
180                };
181                self.session.respond_save(tid, result, &mut emit);
182            }
183            Some(Effect::ClearSaved { tid }) => {
184                self.snapshot = None;
185                self.identity = None;
186                self.session.respond_clear(tid, Ok(()), &mut emit);
187            }
188            Some(Effect::FactoryReset) => {
189                // Emulate the platform wipe-and-reboot: drop every persisted
190                // artifact (snapshot, identity, bonds/PIN) and bring the
191                // session back up factory-fresh as from a power cycle. No
192                // reply is emitted — on hardware the reboot drops the link.
193                self.snapshot = None;
194                self.identity = None;
195                self.identity_seed = 0;
196                self.pairing_pin = None;
197                self.session = NcpSession::new(
198                    session_config(),
199                    Status::RESET_POWER_ON,
200                    CryptoEngine::new(SoftwareAes, SoftwareSha256),
201                );
202            }
203            Some(Effect::ProvisionIdentity { tid }) => {
204                let result = match self.session.identity_request() {
205                    Some(source) => {
206                        let secret = match source {
207                            IdentitySource::Install(secret) => secret,
208                            IdentitySource::Generate => {
209                                self.identity_seed = self.identity_seed.wrapping_add(1).max(1);
210                                [self.identity_seed; 32]
211                            }
212                        };
213                        let public = SoftwareIdentity::from_secret_bytes(&secret).public_key().0;
214                        self.identity = Some((secret, public));
215                        Ok(public)
216                    }
217                    None => Err(()),
218                };
219                self.session.respond_identity(tid, result, &mut emit);
220            }
221        }
222    }
223
224    fn queue_emitted(&mut self, emitted: Vec<Vec<u8>>) {
225        for frame in emitted {
226            let mut encoded = vec![0; hdlc::max_encoded_len(frame.len())];
227            let len = hdlc::encode_frame(&frame, &mut encoded)
228                .expect("simulated NCP output buffer uses HDLC worst-case size");
229            encoded.truncate(len);
230            self.outbound.push_back(encoded);
231        }
232    }
233}

269    fn exchange(sim: &mut SimulatedNcp, request: &[u8]) -> Vec<Vec<u8>> {
270        let mut wire = vec![0; hdlc::max_encoded_len(request.len())];
271        let len = hdlc::encode_frame(request, &mut wire).unwrap();
272        sim.ingest(&wire[..len], 100).unwrap();
273
274        let mut frames = Vec::new();
275        while let Some(wire) = sim.take_outbound() {
276            let mut decoder = hdlc::Decoder::<WIRE_CAPACITY>::new();
277            frames.push(
278                wire.into_iter()
279                    .find_map(|byte| decoder.push(byte).map(|frame| frame.unwrap().to_vec()))
280                    .unwrap(),
281            );
282        }
283        frames
284    }

286    #[test]
287    fn real_session_answers_attach_property_over_hdlc() {
288        let mut sim = SimulatedNcp::new();
289        sim.attach();
290        let mut request = [0; 16];
291        let len = frame::prop_get(&mut request, 1, prop::NCP_VERSION).unwrap();
292        let responses = exchange(&mut sim, &request[..len]);
293        assert_eq!(responses.len(), 1);
294        let response = Frame::parse(&responses[0]).unwrap();
295        let payload = PropPayload::parse(response.payload).unwrap();
296        assert_eq!(payload.key, prop::NCP_VERSION);
297        assert_eq!(payload.value, b"umsh-web-sim/0.1\0");
298    }
299
300    #[test]
301    fn real_session_executes_radio_transmit_effect() {
302        let mut sim = SimulatedNcp::new();
303        sim.attach();
304        let mut request = [0; 64];
305        let len = frame::prop_set(&mut request, 1, prop::PHY_ENABLED, &[1]).unwrap();
306        exchange(&mut sim, &request[..len]);
307        let len = frame::str_send(
308            &mut request,
309            2,
310            umsh_companion::ids::stream::PHY_RAW,
311            b"demo",
312            &[],
313        )
314        .unwrap();
315        exchange(&mut sim, &request[..len]);
316        assert_eq!(sim.transmitted_frames(), &[b"demo".to_vec()]);
317    }
