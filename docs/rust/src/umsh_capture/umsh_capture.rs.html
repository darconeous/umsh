41#[derive(Clone, Copy, Debug, PartialEq, Eq)]
42enum CaptureLayers {
43    Radio,
44    Companion,
45    Both,
46}
47
48impl CaptureLayers {
49    fn parse(value: &str) -> Result<Self, String> {
50        match value {
51            "radio" => Ok(Self::Radio),
52            "companion" => Ok(Self::Companion),
53            "both" => Ok(Self::Both),
54            _ => Err(format!(
55                "invalid capture layer {value:?}; expected radio, companion, or both"
56            )),
57        }
58    }
59
60    fn radio(self) -> bool {
61        matches!(self, Self::Radio | Self::Both)
62    }
63
64    fn companion(self) -> bool {
65        matches!(self, Self::Companion | Self::Both)
66    }
67}

108impl RfArgs {
109    fn parse(args: &[String]) -> Result<Self, String> {
110        let mut rf = Self::default();
111        let mut capture_selection_explicit = false;
112        for arg in args {
113            if let Some(value) = arg.strip_prefix("--freq-khz=") {
114                rf.freq_khz = parse_u32(value)?;
115            } else if let Some(value) = arg.strip_prefix("--bw-hz=") {
116                rf.bandwidth_hz = parse_u32(value)?;
117            } else if let Some(value) = arg.strip_prefix("--sf=") {
118                rf.spreading_factor = parse_u32(value)?
119                    .try_into()
120                    .map_err(|_| format!("invalid spreading factor: {value}"))?;
121            } else if let Some(value) = arg.strip_prefix("--cr=") {
122                rf.coding_rate_denom = parse_u32(value)?
123                    .try_into()
124                    .map_err(|_| format!("invalid coding rate: {value}"))?;
125            } else if let Some(value) = arg.strip_prefix("--sync-word=") {
126                rf.sync_word = parse_u32(value)?
127                    .try_into()
128                    .map_err(|_| format!("invalid sync word: {value}"))?;
129            } else if let Some(value) = arg.strip_prefix("--tx-power=") {
130                rf.tx_power_dbm = value
131                    .parse()
132                    .map_err(|_| format!("invalid TX power: {value}"))?;
133            } else if let Some(value) = arg.strip_prefix("--idle-probe-secs=") {
134                rf.idle_probe_secs = value
135                    .parse()
136                    .map_err(|_| format!("invalid idle probe interval: {value}"))?;
137                if rf.idle_probe_secs == 0 {
138                    return Err("idle probe interval must be greater than zero".into());
139                }
140            } else if let Some(value) = arg.strip_prefix("--reconnect-delay-secs=") {
141                rf.reconnect_delay_secs = value
142                    .parse()
143                    .map_err(|_| format!("invalid reconnect delay: {value}"))?;
144                if rf.reconnect_delay_secs == 0 {
145                    return Err("reconnect delay must be greater than zero".into());
146                }
147            } else if arg == "--no-reconnect" {
148                rf.reconnect = false;
149            } else if arg == "--umsh-only" {
150                rf.umsh_only = true;
151            } else if let Some(value) = arg.strip_prefix("--pcap=") {
152                if value.is_empty() {
153                    return Err("pcap path must not be empty".into());
154                }
155                rf.pcap_path = Some(PathBuf::from(value));
156            } else if let Some(value) = arg.strip_prefix("--capture=") {
157                rf.capture_layers = CaptureLayers::parse(value)?;
158                capture_selection_explicit = true;
159            } else if arg == "--pcap-raw" {
160                rf.pcap_raw = true;
161            } else if let Some(value) = arg.strip_prefix("--pcap-linktype=") {
162                rf.pcap_linktype = Some(parse_u32(value)?);
163            } else if arg.starts_with('-') {
164                return Err(format!("unknown option: {arg}"));
165            }
166        }
167        if capture_selection_explicit && rf.pcap_path.is_none() {
168            return Err("--capture requires --pcap=PATH".into());
169        }
170        if rf.pcap_raw {
171            if rf.pcap_path.is_none() {
172                return Err("--pcap-raw requires --pcap=PATH".into());
173            }
174            if rf.capture_layers != CaptureLayers::Radio {
175                return Err("--pcap-raw requires --capture=radio".into());
176            }
177            if rf.pcap_linktype.is_none() {
178                return Err("--pcap-raw requires --pcap-linktype=N".into());
179            }
180        } else if rf.pcap_linktype.is_some() {
181            return Err("--pcap-linktype requires --pcap-raw".into());
182        }
183        Ok(rf)
184    }
185
186    fn config(&self) -> CompanionRadioConfig {
187        let mut config = CompanionRadioConfig::new(
188            self.freq_khz,
189            self.bandwidth_hz,
190            self.spreading_factor,
191            self.coding_rate_denom,
192        );
193        config.sync_word = self.sync_word;
194        config.tx_power_dbm = self.tx_power_dbm;
195        config
196    }
197}

199fn parse_u32(value: &str) -> Result<u32, String> {
200    if let Some(hex) = value
201        .strip_prefix("0x")
202        .or_else(|| value.strip_prefix("0X"))
203    {
204        u32::from_str_radix(hex, 16).map_err(|_| format!("invalid number: {value}"))
205    } else {
206        value
207            .parse()
208            .map_err(|_| format!("invalid number: {value}"))
209    }
210}

218/// Shared classic-pcap sink. Frames use the repository's established
219/// synthetic Ethernet/IPv4/UDP encapsulation so stock Wireshark can open a
220/// single file containing both capture layers.
221type SharedCapture = Rc<RefCell<PcapWriter>>;

223const PCAP_LINKTYPE_ETHERNET: u32 = 1;
224const RADIO_UDP_PORT: u16 = 4242;
225const COMPANION_HOST_UDP_PORT: u16 = 4243;
226const COMPANION_NCP_UDP_PORT: u16 = 4244;
227
228#[derive(Clone, Copy)]
229enum PcapEncapsulation {
230    Ethernet,
231    RawLoRa { linktype: u32 },
232}
233
234struct PcapWriter {
235    output: BufWriter<File>,
236    layers: CaptureLayers,
237    encapsulation: PcapEncapsulation,
238    packet_id: u16,
239}
240
241impl PcapWriter {
242    fn create(
243        path: &PathBuf,
244        layers: CaptureLayers,
245        encapsulation: PcapEncapsulation,
246    ) -> std::io::Result<Self> {
247        let mut output = BufWriter::new(File::create(path)?);
248        output.write_all(&0xa1b2_c3d4u32.to_le_bytes())?;
249        output.write_all(&2u16.to_le_bytes())?;
250        output.write_all(&4u16.to_le_bytes())?;
251        output.write_all(&0i32.to_le_bytes())?;
252        output.write_all(&0u32.to_le_bytes())?;
253        output.write_all(&65_535u32.to_le_bytes())?;
254        let linktype = match encapsulation {
255            PcapEncapsulation::Ethernet => PCAP_LINKTYPE_ETHERNET,
256            PcapEncapsulation::RawLoRa { linktype } => linktype,
257        };
258        output.write_all(&linktype.to_le_bytes())?;
259        output.flush()?;
260        Ok(Self {
261            output,
262            layers,
263            encapsulation,
264            packet_id: 0,
265        })
266    }
267
268    fn write_radio(&mut self, frame: &[u8]) -> std::io::Result<()> {
269        if self.layers.radio() {
270            match self.encapsulation {
271                PcapEncapsulation::Ethernet => self.write_udp(
272                    CaptureDirection::NcpToHost,
273                    RADIO_UDP_PORT,
274                    RADIO_UDP_PORT,
275                    frame,
276                )?,
277                PcapEncapsulation::RawLoRa { .. } => self.write_record(frame)?,
278            }
279        }
280        Ok(())
281    }
282
283    fn write_companion(
284        &mut self,
285        direction: CaptureDirection,
286        frame: &[u8],
287    ) -> std::io::Result<()> {
288        if !self.layers.companion() {
289            return Ok(());
290        }
291        debug_assert!(matches!(self.encapsulation, PcapEncapsulation::Ethernet));
292        let (src_port, dst_port) = match direction {
293            CaptureDirection::HostToNcp => (COMPANION_HOST_UDP_PORT, COMPANION_NCP_UDP_PORT),
294            CaptureDirection::NcpToHost => (COMPANION_NCP_UDP_PORT, COMPANION_HOST_UDP_PORT),
295        };
296        self.write_udp(direction, src_port, dst_port, frame)
297    }
298
299    fn write_udp(
300        &mut self,
301        direction: CaptureDirection,
302        src_port: u16,
303        dst_port: u16,
304        payload: &[u8],
305    ) -> std::io::Result<()> {
306        let udp_len = 8usize
307            .checked_add(payload.len())
308            .and_then(|len| u16::try_from(len).ok())
309            .ok_or_else(|| std::io::Error::other("capture payload exceeds IPv4 UDP size"))?;
310        let ip_len = 20u16
311            .checked_add(udp_len)
312            .ok_or_else(|| std::io::Error::other("capture packet exceeds IPv4 size"))?;
313        let frame_len = 14usize + usize::from(ip_len);
314        let mut packet = Vec::with_capacity(frame_len);
315
316        // Synthetic Ethernet and loopback IPv4 endpoints. Direction remains
317        // visible in both endpoint addresses and companion UDP ports.
318        packet.extend_from_slice(&[0x02, 0, 0, 0, 0, 2]);
319        packet.extend_from_slice(&[0x02, 0, 0, 0, 0, 1]);
320        packet.extend_from_slice(&0x0800u16.to_be_bytes());
321        let (src_ip, dst_ip) = match direction {
322            CaptureDirection::HostToNcp => ([127, 0, 0, 1], [127, 0, 0, 2]),
323            CaptureDirection::NcpToHost => ([127, 0, 0, 2], [127, 0, 0, 1]),
324        };
325        let ip_start = packet.len();
326        packet.extend_from_slice(&[
327            0x45,
328            0,
329            (ip_len >> 8) as u8,
330            ip_len as u8,
331            (self.packet_id >> 8) as u8,
332            self.packet_id as u8,
333            0,
334            0,
335            64,
336            17,
337            0,
338            0,
339            src_ip[0],
340            src_ip[1],
341            src_ip[2],
342            src_ip[3],
343            dst_ip[0],
344            dst_ip[1],
345            dst_ip[2],
346            dst_ip[3],
347        ]);
348        let checksum = ipv4_checksum(&packet[ip_start..ip_start + 20]);
349        packet[ip_start + 10..ip_start + 12].copy_from_slice(&checksum.to_be_bytes());
350        packet.extend_from_slice(&src_port.to_be_bytes());
351        packet.extend_from_slice(&dst_port.to_be_bytes());
352        packet.extend_from_slice(&udp_len.to_be_bytes());
353        packet.extend_from_slice(&0u16.to_be_bytes());
354        packet.extend_from_slice(payload);
355        self.packet_id = self.packet_id.wrapping_add(1);
356
357        self.write_record(&packet)
358    }
359
360    fn write_record(&mut self, packet: &[u8]) -> std::io::Result<()> {
361        let timestamp = SystemTime::now()
362            .duration_since(UNIX_EPOCH)
363            .unwrap_or_default();
364        let seconds = u32::try_from(timestamp.as_secs()).unwrap_or(u32::MAX);
365        let captured_len = u32::try_from(packet.len())
366            .map_err(|_| std::io::Error::other("capture record exceeds pcap size"))?;
367        self.output.write_all(&seconds.to_le_bytes())?;
368        self.output
369            .write_all(&timestamp.subsec_micros().to_le_bytes())?;
370        self.output.write_all(&captured_len.to_le_bytes())?;
371        self.output.write_all(&captured_len.to_le_bytes())?;
372        self.output.write_all(&packet)?;
373        // Keep the file usable by Wireshark during a long-running capture.
374        self.output.flush()
375    }
376}

378fn ipv4_checksum(header: &[u8]) -> u16 {
379    let mut sum = 0u32;
380    for word in header.chunks_exact(2) {
381        sum += u32::from(u16::from_be_bytes([word[0], word[1]]));
382    }
383    while sum > 0xffff {
384        sum = (sum & 0xffff) + (sum >> 16);
385    }
386    !(sum as u16)
387}

389struct CapturingFrameLink<L> {
390    inner: L,
391    capture: Option<SharedCapture>,
392}
393
394impl<L> CapturingFrameLink<L> {
395    fn new(inner: L, capture: Option<SharedCapture>) -> Self {
396        Self { inner, capture }
397    }
398
399    fn record(&self, direction: CaptureDirection, frame: &[u8]) -> std::io::Result<()> {
400        if let Some(capture) = &self.capture {
401            capture.borrow_mut().write_companion(direction, frame)?;
402        }
403        Ok(())
404    }
405}
406
407impl<L: FrameLink> FrameLink for CapturingFrameLink<L> {
408    async fn send_frame(
409        &mut self,
410        frame: &[u8],
411    ) -> Result<(), umsh::companion_radio::CompanionRadioError> {
412        self.record(CaptureDirection::HostToNcp, frame)?;
413        self.inner.send_frame(frame).await
414    }
415
416    fn poll_recv_frame(
417        &mut self,
418        cx: &mut core::task::Context<'_>,
419    ) -> core::task::Poll<Result<Vec<u8>, umsh::companion_radio::CompanionRadioError>> {
420        match self.inner.poll_recv_frame(cx) {
421            core::task::Poll::Ready(Ok(frame)) => {
422                match self.record(CaptureDirection::NcpToHost, &frame) {
423                    Ok(()) => core::task::Poll::Ready(Ok(frame)),
424                    Err(error) => core::task::Poll::Ready(Err(error.into())),
425                }
426            }
427            other => other,
428        }
429    }
430}

432#[tokio::main(flavor = "current_thread")]
433async fn main() -> Result<(), Box<dyn std::error::Error>> {
434    let mut args = std::env::args().skip(1);
435    let Some(first) = args.next() else {
436        return Err(USAGE.into());
437    };
438    if first == "--help" || first == "-h" {
439        print!("{USAGE}");
440        return Ok(());
441    }
442
443    let mut rest: Vec<String> = args.collect();
444    let selector = if first == "--ble" {
445        rest.first().filter(|arg| !arg.starts_with('-')).cloned()
446    } else {
447        None
448    };
449    if selector.is_some() {
450        rest.remove(0);
451    }
452    let rf = RfArgs::parse(&rest).map_err(|error| format!("{error}\n\n{USAGE}"))?;
453    let config = rf.config();
454    let capture = if let Some(path) = &rf.pcap_path {
455        let encapsulation = if rf.pcap_raw {
456            PcapEncapsulation::RawLoRa {
457                linktype: rf.pcap_linktype.expect("validated raw pcap linktype"),
458            }
459        } else {
460            PcapEncapsulation::Ethernet
461        };
462        let writer = PcapWriter::create(path, rf.capture_layers, encapsulation)?;
463        println!(
464            "capture: {} layer={:?} encoding={}",
465            path.display(),
466            rf.capture_layers,
467            if rf.pcap_raw {
468                "raw LoRa"
469            } else {
470                "Ethernet/IPv4/UDP"
471            },
472        );
473        Some(Rc::new(RefCell::new(writer)))
474    } else {
475        None
476    };
477
478    if first == "--ble" {
479        #[cfg(feature = "ble-radio")]
480        {
481            return run_ble_dump(selector.as_deref(), config, rf, capture).await;
482        }
483        #[cfg(not(feature = "ble-radio"))]
484        return Err("the ble-radio feature is required for --ble".into());
485    }
486
487    #[cfg(feature = "serial-radio")]
488    {
489        use tokio_serial::SerialPortBuilderExt;
490        use umsh::companion_radio::SerialFrameLink;
491
492        println!("attaching to {first} ...");
493        let stream = tokio_serial::new(&first, 115_200).open_native_async()?;
494        let link = CapturingFrameLink::new(SerialFrameLink::new(stream), capture.clone());
495        let radio = CompanionRadio::new(link, config).await?;
496        let mut stats = DumpStats::new();
497        stats.sessions = 1;
498        return run_dump(radio, &rf, &mut stats, capture.as_ref()).await;
499    }
500    #[cfg(not(feature = "serial-radio"))]
501    Err("the serial-radio feature is required for a serial path".into())
502}

526#[cfg(feature = "ble-radio")]
527async fn run_ble_dump(
528    selector: Option<&str>,
529    config: CompanionRadioConfig,
530    rf: RfArgs,
531    capture: Option<SharedCapture>,
532) -> Result<(), Box<dyn std::error::Error>> {
533    use umsh::companion_radio::{BleFrameLink, BleFrameLinkConfig};
534
535    let mut stats = DumpStats::new();
536    loop {
537        println!("discovering BLE companion radio ...");
538        let failure: Box<dyn std::error::Error> =
539            match BleFrameLink::connect(selector, BleFrameLinkConfig::default()).await {
540                Ok(link) => match CompanionRadio::new(
541                    CapturingFrameLink::new(link, capture.clone()),
542                    config.clone(),
543                )
544                .await
545                {
546                    Ok(radio) => {
547                        stats.sessions += 1;
548                        match run_dump(radio, &rf, &mut stats, capture.as_ref()).await {
549                            Ok(()) => return Ok(()),
550                            Err(error) => error,
551                        }
552                    }
553                    Err(error) => Box::new(error),
554                },
555                Err(error) => Box::new(error),
556            };
557
558        eprintln!(
559            "BLE session failure +{:.3}s after {} packets: {failure}",
560            stats.started.elapsed().as_secs_f64(),
561            stats.sequence,
562        );
563        if !rf.reconnect {
564            return Err(failure);
565        }
566        println!(
567            "recovery: rediscovering in {} s (ctrl-c to exit) ...",
568            rf.reconnect_delay_secs,
569        );
570        tokio::time::sleep(Duration::from_secs(rf.reconnect_delay_secs)).await;
571    }
572}

574async fn run_dump<L: FrameLink>(
575    mut radio: CompanionRadio<L>,
576    rf: &RfArgs,
577    stats: &mut DumpStats,
578    capture: Option<&SharedCapture>,
579) -> Result<(), Box<dyn std::error::Error>> {
580    println!(
581        "session #{}: ncp={} boot_status={:?}",
582        stats.sessions,
583        radio.ncp_version(),
584        radio.boot_status(),
585    );
586    println!(
587        "radio: {} kHz, BW {} Hz, SF{}, CR 4/{}, sync 0x{:04x}",
588        rf.freq_khz, rf.bandwidth_hz, rf.spreading_factor, rf.coding_rate_denom, rf.sync_word,
589    );
590    // A capture is a promiscuous listener. An NCP with a provisioned
591    // (or saved-and-restored) host domain filters receptions, so the
592    // factory deliver-everything rule cannot be relied on; bypass the
593    // filtering for this session (`PROP_MAC_PROMISCUOUS` is
594    // session-scoped and reverts on detach). An NCP that predates the
595    // property refuses the set — capture then sees only frames
596    // matching its receive filtering.
597    match radio
598        .set_prop(umsh::companion::ids::prop::MAC_PROMISCUOUS, &[1])
599        .await
600    {
601        Ok(_) => println!("promiscuous mode enabled"),
602        Err(umsh::companion_radio::CompanionRadioError::Status(status)) => eprintln!(
603            "warning: NCP refused promiscuous mode ({status:?}); \
604             capture is limited to the NCP's receive filtering"
605        ),
606        Err(error) => return Err(error.into()),
607    }
608    println!("dumping packets (ctrl-c to exit) ...");
609
610    let mut buf = [0u8; 256];
611    let idle_probe_interval = Duration::from_secs(rf.idle_probe_secs);
612    loop {
613        let receive = core::future::poll_fn(|cx| radio.poll_receive(cx, &mut buf));
614        let info = match tokio::time::timeout(idle_probe_interval, receive).await {
615            Ok(result) => result?,
616            Err(_) => {
617                let value = radio
618                    .get_prop(umsh::companion::ids::prop::PHY_RSSI)
619                    .await
620                    .map_err(|error| {
621                        format!(
622                            "idle health probe failed after {} packets: {error}",
623                            stats.sequence,
624                        )
625                    })?;
626                let [rssi] = value.as_slice() else {
627                    return Err(format!(
628                        "idle health probe returned malformed PHY_RSSI value: {value:02x?}"
629                    )
630                    .into());
631                };
632                println!(
633                    "idle +{:.3}s  received={}  displayed={}  filtered={}  session={}  link=ok  channel RSSI={} dBm",
634                    stats.started.elapsed().as_secs_f64(),
635                    stats.sequence,
636                    stats.displayed,
637                    stats.filtered,
638                    stats.sessions,
639                    *rssi as i8,
640                );
641                stats.last_progress = Instant::now();
642                continue;
643            }
644        };
645        stats.sequence += 1;
646        let packet = &buf[..info.len];
647        if let Some(capture) = capture {
648            capture.borrow_mut().write_radio(packet)?;
649        }
650        if !should_display(packet, rf.umsh_only) {
651            stats.filtered += 1;
652            if stats.last_progress.elapsed() >= idle_probe_interval {
653                println!(
654                    "filter +{:.3}s  received={}  displayed={}  filtered={}  session={}  link=ok",
655                    stats.started.elapsed().as_secs_f64(),
656                    stats.sequence,
657                    stats.displayed,
658                    stats.filtered,
659                    stats.sessions,
660                );
661                stats.last_progress = Instant::now();
662            }
663            continue;
664        }
665        stats.displayed += 1;
666        println!(
667            "\n#{} +{:.3}s  session={}  len={}  RSSI={} dBm  SNR={}  LQI={}",
668            stats.sequence,
669            stats.started.elapsed().as_secs_f64(),
670            stats.sessions,
671            info.len,
672            info.rssi,
673            info.snr,
674            info.lqi
675                .map_or_else(|| "n/a".into(), |lqi| lqi.get().to_string()),
676        );
677        print!("raw:");
678        for byte in packet {
679            print!(" {byte:02x}");
680        }
681        println!();
682        print_classification(packet);
683    }
684}
685
686fn should_display(packet: &[u8], umsh_only: bool) -> bool {
687    !umsh_only || PacketHeader::parse(packet).is_ok()
688}
689
690fn print_classification(packet: &[u8]) {
691    let header = match PacketHeader::parse(packet) {
692        Ok(header) => header,
693        Err(error) => {
694            println!("decode: not a valid UMSH packet ({error:?})");
695            return;
696        }
697    };
698    let options = ParsedOptions::extract(packet, header.options_range.clone());
699    let encrypted = header
700        .sec_info
701        .is_some_and(|security| security.scf.encrypted());
702    let frame_counter = header.sec_info.map(|security| security.frame_counter);
703    let payload_type = (!encrypted && header.packet_type() != PacketType::MacAck)
704        .then(|| {
705            packet
706                .get(header.body_range.start)
707                .and_then(|byte| PayloadType::from_byte(*byte))
708        })
709        .flatten();
710
711    let src = match header.source {
712        SourceAddrRef::Hint(hint) => format!("hint:{hint}"),
713        SourceAddrRef::FullKeyAt { offset } => packet.get(offset..offset + 32).map_or_else(
714            || "<truncated>".to_owned(),
715            |bytes| {
716                let mut key = [0u8; 32];
717                key.copy_from_slice(bytes);
718                PublicKey(key).to_string()
719            },
720        ),
721        SourceAddrRef::Encrypted { .. } => "<encrypted>".to_owned(),
722        SourceAddrRef::None => "-".to_owned(),
723    };
724    let dst = header
725        .dst
726        .map_or_else(|| "-".to_owned(), |hint| hint.to_string());
727    println!(
728        "decode: UMSH {:?} src={} dst={} channel={:?}",
729        header.packet_type(),
730        src,
731        dst,
732        header.channel,
733    );
734    println!(
735        "        encrypted={encrypted} frame_counter={frame_counter:?} ack_requested={} body={} mic={} payload={payload_type:?}",
736        header.ack_requested(),
737        header.body_range.len(),
738        header.mic_range.len(),
739    );
740    match options {
741        Ok(options) => println!("        options={options:?}"),
742        Err(error) => println!("        options=<decode error: {error:?}>"),
743    }
744}
