1//! The board-agnostic companion-NCP session driver (Phase 5, increment C).
2//!
3//! This is the extraction of the nRF NCP's `ncp_task` select loop, its
4//! `apply_effect` radio-effect dispatcher, and the `Emitter` frame stager —
5//! the one copy of the session-driving logic shared by every companion
6//! firmware (T-Echo, T-1000E, Heltec V3). The board personalities that used
7//! to be `cfg(feature = "t1000e")` forks inside the loop are expressed as
8//! [`NcpEnv`] hooks with no-op defaults, so a new board supplies exactly the
9//! couplings it has and nothing else.
10//!
11//! The split of responsibilities:
12//!
13//! - **This module** owns the protocol loop: transport arbitration, frame
14//!   handling, radio RX/TX-completion processing, every deferred `Effect`
15//!   arm (save/clear/wipe/provision/PIN/RSSI/battery/drain), and the
16//!   device-domain mirror.
17//! - **The board** owns the edges: transport tasks feeding [`InEvent`]s and
18//!   draining [`TransportChannels`], the radio runner + mux serving the
19//!   session's virtual [`Channels`] bundle, and an [`NcpEnv`] implementation
20//!   wiring persistence, entropy, pairing, and indicators to its hardware.
21

42/// Largest raw companion frame accepted from a transport.
43pub const FRAME_IN_MAX: usize = 300;
44/// Largest companion frame the session emits (CMD_STR_RECV around a
45/// full-MTU payload).
46pub const FRAME_OUT_MAX: usize = 300;
47
48/// One raw companion frame moving through the driver.
49pub type FrameBuf = heapless::Vec<u8, FRAME_IN_MAX>;
50
51/// Framing-free receive path and connection edges into the driver.
52pub enum InEvent {
53    Attached(Transport),
54    Detached(Transport),
55    Frame(Transport, FrameBuf),
56}
57
58/// The inbound event channel every transport task feeds.
59pub type InputChannel<M> = Channel<M, InEvent, 8>;
60
61/// One raw companion frame in a transport output queue, stamped with the
62/// session generation that produced it so a displaced session's frames
63/// are dropped at the transport edge (`transport_policy::generation_checked`).
64pub struct OutFrame {
65    pub generation: u32,
66    pub frame: FrameBuf,
67}
68
69/// The per-transport outbound frame queues, drained by the board's
70/// transport output tasks. `wired` is the physical-possession transport
71/// (USB-CDC or UART), `ble` the bonded GATT transport — the same pairing
72/// `transport_policy::Transport` names.
73pub struct TransportChannels<M: RawMutex> {
74    pub wired: Channel<M, OutFrame, 4>,
75    pub ble: Channel<M, OutFrame, 4>,
76}
77
78impl<M: RawMutex> TransportChannels<M> {
79    pub const fn new() -> Self {
80        Self {
81            wired: Channel::new(),
82            ble: Channel::new(),
83        }
84    }
85
86    fn for_transport(&self, transport: Transport) -> &Channel<M, OutFrame, 4> {
87        match transport {
88            Transport::Usb => &self.wired,
89            Transport::Ble => &self.ble,
90        }
91    }
92}

102/// Board couplings of the session driver. Everything the loop needs from
103/// the platform, expressed as one trait so the driver itself stays free
104/// of HAL types and `cfg` board forks. Hooks a board doesn't have keep
105/// their no-op defaults (e.g. only the T-1000E implements the attention
106/// indicator and transmit-load hooks today).
107// Single-executor embedded consumers; `Send` futures are irrelevant here,
108// same as the embassy ecosystem's own async traits.
109#[allow(async_fn_in_trait)]
110pub trait NcpEnv {
111    /// Durably persist the encoded protocol snapshot (CMD_SAVE / host wipe).
112    async fn persist_snapshot(&mut self, bytes: &[u8]) -> Result<(), ()>;
113    /// Tombstone the snapshot journal (CMD_CLEAR).
114    async fn clear_snapshot(&mut self) -> Result<(), ()>;
115    /// Durably persist the encoded device identity.
116    async fn persist_identity(&mut self, bytes: &[u8]) -> Result<(), ()>;
117    /// Tombstone the identity journal (CMD_CLEAR).
118    async fn clear_identity(&mut self) -> Result<(), ()>;
119    /// Drop persisted frame-counter boundaries after a successful
120    /// identity clear. Boards without a device node keep the default.
121    async fn clear_counters(&mut self) {}
122    /// Fill `secret` from the platform's cryptographic RNG. Fails closed:
123    /// an error refuses identity generation rather than degrading.
124    fn fill_secret(&mut self, secret: &mut [u8; 32]) -> Result<(), ()>;
125    /// One fresh battery measurement (`Effect::SampleBattery`). Only
126    /// emitted when the board's `SessionConfig::battery` advertises
127    /// fields, so the default refuses.
128    async fn sample_battery(&mut self) -> Result<umsh_companion::battery::BatteryStatus, ()> {
129        Err(())
130    }
131    /// Apply a `PROP_BLE_PAIRING_PIN` write against the bond journal and
132    /// the live BLE stack; `true` when it took effect.
133    async fn apply_pairing_pin(&mut self, pin: Option<u32>) -> bool;
134    /// `CMD_FACTORY_RESET`: erase EVERY piece of persistent state the
135    /// platform owns — saved snapshot, device identity, frame-counter
136    /// boundaries, BLE bonds, pairing PIN, and any other journal — then
137    /// reboot. Never returns: the reset discards in-RAM state and the
138    /// board comes back factory-fresh. There is no separate "clear bonds"
139    /// hook because a reboot reloads bonds from the (now-erased) journal,
140    /// so the live BLE stack never has to be touched.
141    async fn factory_reset(&mut self) -> !;
142    /// Publish the transport-arbitration advertising policy (a wired
143    /// attach suppresses BLE advertising). Diagnostic builds may
144    /// deliberately ignore `allowed`.
145    fn set_advertising_allowed(&mut self, allowed: bool);
146    /// Publish the session's device name to the board's consumers
147    /// (advertising data, device node, UI).
148    async fn publish_device_name(&mut self, name: &str);
149    /// Deliver a device-domain mirror to the board's device node.
150    fn publish_dev_domain(&mut self, snapshot: DevDomainSnapshot);
151    /// A covered frame was queued for an attached-or-future host
152    /// (T-1000E: request the attention LED).
153    fn request_attention(&mut self) {}
154    /// The host-facing queue drained to empty (T-1000E: clear it).
155    fn clear_attention(&mut self) {}
156    /// A transmit is about to start; boards with a battery-level
157    /// estimator mark the load spike.
158    fn note_transmit_load(&mut self) {}
159    /// Diagnostic trace line (routed to the board's debug channel; the
160    /// default discards).
161    fn trace(&mut self, args: core::fmt::Arguments<'_>) {
162        let _ = args;
163    }
164}

166/// The driver's `'static` wiring: the channels and control blocks the
167/// loop shares with the board's transport and radio tasks.
168pub struct NcpRuntime<M: RawMutex + 'static, const RX: usize, const TX: usize> {
169    /// Inbound frames and connection edges from every transport task.
170    pub input: &'static InputChannel<M>,
171    /// The session's radio endpoint — its private virtual `Channels`
172    /// bundle served by the board's radio mux (never the real radio
173    /// bundle directly).
174    pub radio: &'static Channels<M, RX, TX>,
175    /// Runtime radio settings / RSSI sampling into the radio runner.
176    pub ctl: &'static NcpControl<M>,
177    /// Outbound frame queues drained by the transport output tasks.
178    pub out: &'static TransportChannels<M>,
179    /// Published session epoch, checked by each transport at framing
180    /// edges (`transport_policy::generation_checked`).
181    pub session_gen: &'static AtomicU32,
182}

184/// Collects frames emitted synchronously by the session, then flushes
185/// them to the active transport's output queue asynchronously. The
186/// session emits at most one frame per call; two slots give headroom.
187struct Emitter {
188    bufs: [[u8; FRAME_OUT_MAX]; 2],
189    lens: [usize; 2],
190    count: usize,
191}
192
193impl Emitter {
194    const fn new() -> Self {
195        Self {
196            bufs: [[0; FRAME_OUT_MAX]; 2],
197            lens: [0; 2],
198            count: 0,
199        }
200    }
201
202    /// Copy one raw companion frame into the next slot.
203    ///
204    /// The session is expected to emit at most `bufs.len()` frames per call
205    /// and every frame is expected to fit `FRAME_OUT_MAX`. Both invariants are
206    /// asserted in debug builds so a future session change that violates
207    /// them is caught rather than silently dropping a response.
208    fn push(&mut self, frame: &[u8]) {
209        if self.count >= self.bufs.len() {
210            debug_assert!(
211                false,
212                "Emitter overflow: session emitted more frames per call than staging slots"
213            );
214            return;
215        }
216        if frame.len() <= FRAME_OUT_MAX {
217            self.bufs[self.count][..frame.len()].copy_from_slice(frame);
218            self.lens[self.count] = frame.len();
219            self.count += 1;
220        } else {
221            debug_assert!(false, "Emitter: companion frame exceeds FRAME_OUT_MAX");
222        }
223    }
224
225    /// Queue all staged frames for the active transport output task.
226    async fn flush<M: RawMutex>(
227        &mut self,
228        destination: Option<(Transport, u32)>,
229        out: &TransportChannels<M>,
230    ) {
231        if let Some((transport, generation)) = destination {
232            for index in 0..self.count {
233                let mut frame: FrameBuf = heapless::Vec::new();
234                if frame
235                    .extend_from_slice(&self.bufs[index][..self.lens[index]])
236                    .is_err()
237                {
238                    // FRAME_OUT_MAX == FrameBuf capacity, so this cannot
239                    // happen; assert in debug rather than silently drop.
240                    debug_assert!(false, "Emitter frame copy exceeded FrameBuf capacity");
241                    continue;
242                }
243                out.for_transport(transport)
244                    .send(OutFrame { generation, frame })
245                    .await;
246            }
247        }
248        self.count = 0;
249    }
250}

252/// Execute a radio side effect requested by the session.
253async fn apply_effect<A, S, const TXQ: usize, M, const RX: usize, const TX: usize, E>(
254    session: &Session<A, S, TXQ>,
255    effect: Option<Effect>,
256    rt: &NcpRuntime<M, RX, TX>,
257    env: &mut E,
258) where
259    A: AesProvider,
260    S: Sha256Provider,
261    M: RawMutex,
262    E: NcpEnv,
263{
264    match effect {
265        Some(Effect::ApplyRadio(settings)) => {
266            env.publish_device_name(session.device_name()).await;
267            // The session validates values against the same discrete
268            // sets these converters accept, so None here is
269            // unreachable; bail out defensively rather than panic.
270            let (Some(sf), Some(bw), Some(cr)) = (
271                spreading_factor_from_u8(settings.sf),
272                bandwidth_from_hz(settings.bw_hz),
273                coding_rate_from_denom(settings.cr_denom),
274            ) else {
275                return;
276            };
277            rt.ctl.apply(NcpSettings {
278                enabled: settings.enabled,
279                freq_hz: settings.freq_khz.saturating_mul(1_000),
280                sf,
281                bw,
282                cr,
283                power_dbm: i32::from(settings.tx_power_dbm),
284            });
285        }
286        Some(Effect::StartTransmit) => {
287            let mut data: heapless::Vec<u8, MAX_PAYLOAD> = heapless::Vec::new();
288            if data.extend_from_slice(session.tx_data()).is_err() {
289                env.trace(format_args!(
290                    "radio tx staging=FAILED len={}",
291                    session.tx_data().len()
292                ));
293                return;
294            }
295            let power_dbm = match session.tx_power() {
296                TxPower::Default => None,
297                TxPower::Max => Some(i32::from(session.max_tx_power_dbm())),
298                TxPower::Dbm(dbm) => Some(i32::from(dbm)),
299            };
300            // Mark the load for the board's battery level estimator (the
301            // radio runner transmits within milliseconds of this).
302            env.note_transmit_load();
303            rt.radio.tx.send(TxRequest { data, power_dbm }).await;
304        }
305        Some(Effect::DeviceNameChanged) => {
306            env.publish_device_name(session.device_name()).await;
307        }
308        // Deferred effects needing `&mut Session` + the emitter are
309        // handled inline in the run loop rather than here.
310        Some(Effect::SampleRssi { .. })
311        | Some(Effect::SampleBattery { .. })
312        | Some(Effect::SetPairingPin { .. })
313        | Some(Effect::WipeHostDomain { .. })
314        | Some(Effect::DrainQueue)
315        | Some(Effect::SaveSnapshot { .. })
316        | Some(Effect::ClearSaved { .. })
317        | Some(Effect::ProvisionIdentity { .. })
318        | Some(Effect::FactoryReset)
319        | None => {}
320    }
321}

323/// Mirror the session's device-domain node tables to the device node
324/// when their generation moved (device-node plan increment 3).
325/// `synced_version` is the caller's cache of the last published
326/// generation. Cheap when nothing changed — one u32 compare — so the
327/// loop runs it after every session interaction.
328fn sync_dev_domain<A, S, const TXQ: usize, E>(
329    session: &Session<A, S, TXQ>,
330    synced_version: &mut u32,
331    env: &mut E,
332) where
333    A: AesProvider,
334    S: Sha256Provider,
335    E: NcpEnv,
336{
337    if session.dev_domain_version() == *synced_version {
338        return;
339    }
340    *synced_version = session.dev_domain_version();
341    let mut snapshot = DevDomainSnapshot {
342        channel_keys: heapless::Vec::new(),
343        peers: heapless::Vec::new(),
344        identity_present: session.dev_key().is_some(),
345    };
346    for key in session.dev_channel_keys() {
347        let _ = snapshot.channel_keys.push(key);
348    }
349    for public_key in session.dev_peers() {
350        let _ = snapshot.peers.push(public_key);
351    }
352    env.publish_dev_domain(snapshot);
353}

355/// Drive the companion session forever: restore persisted state, then
356/// select over host frames, radio receptions, and transmit completions,
357/// executing every session effect through the board's [`NcpEnv`].
358///
359/// The caller constructs the [`Session`] with its board profile
360/// (`SessionConfig`) and boot status, mounts its journals, and hands the
361/// stored payloads in; the driver owns everything after that.
362pub async fn run<A, S, const TXQ: usize, M, const RX: usize, const TX: usize, E>(
363    mut session: Session<A, S, TXQ>,
364    boot_snapshot: Option<&[u8]>,
365    boot_identity: Option<[u8; 32]>,
366    rt: NcpRuntime<M, RX, TX>,
367    mut env: E,
368) -> !
369where
370    A: AesProvider,
371    S: Sha256Provider,
372    M: RawMutex,
373    E: NcpEnv,
374{
375    let mut emitter = Emitter::new();
376    let mut arbitration = SessionArbitration::new(rt.session_gen.load(Ordering::Acquire));
377    // Last device-domain generation mirrored to the device node.
378    // Matches the session's initial value; the first mutation (or a
379    // boot restore) publishes the first snapshot.
380    let mut dev_domain_synced: u32 = session.dev_domain_version();
381    // Shared staging buffer for the durable-write effect arms
382    // (save/wipe). Held across their persist awaits, so as a
383    // loop-lifetime local it costs one future slot instead of one
384    // per arm.
385    let mut snapshot_buf = [0u8; SNAPSHOT_MAX];
386
387    // The device identity is persisted independently of snapshots;
388    // its post-reset value is whatever the identity journal holds.
389    if let Some(public_key) = boot_identity {
390        session.set_boot_identity(public_key);
391    }
392
393    // Restore a stored snapshot before processing any host command:
394    // the saved configuration is applied, the PHY re-enabled if it
395    // was enabled when saved, and detached operation begins
396    // immediately. A snapshot that fails to decode is ignored.
397    if let Some(payload) = boot_snapshot {
398        let effect = session.restore_at_boot(payload);
399        env.trace(format_args!(
400            "proto-store boot-restore={}",
401            if effect.is_some() { "ok" } else { "IGNORED" }
402        ));
403        apply_effect(&session, effect, &rt, &mut env).await;
404        // Replay the restored device-domain tables into the device
405        // node before any host interaction: detached multicast
406        // processing must not wait for an attach.
407        sync_dev_domain(&session, &mut dev_domain_synced, &mut env);
408    }
409
410    loop {
411        // Only wait for a TX completion while one is outstanding,
412        // so a spurious tx_done can never be consumed early.
413        let tx_done = async {
414            if session.has_pending_tx() {
415                rt.radio.tx_done.wait().await
416            } else {
417                core::future::pending().await
418            }
419        };
420
421        match select3(rt.input.receive(), rt.radio.rx.receive(), tx_done).await {
422            Either3::First(InEvent::Attached(transport)) => {
423                // Fresh session state for the new host session; the
424                // device domain (PHY configuration and enable state,
425                // device name, duty accounting) is deliberately
426                // untouched and nothing is emitted (full-protocol
427                // attach semantics).
428                arbitration.attach(transport);
429                rt.session_gen
430                    .store(arbitration.generation(), Ordering::Release);
431                env.set_advertising_allowed(arbitration.advertising_allowed());
432                // Both transports meet their provisioning-security
433                // binding here: the wired transport by physical
434                // possession, BLE because the companion GATT service
435                // refuses any access outside an encrypted LESC-bonded
436                // link.
437                session.attach(true);
438            }
439            Either3::First(InEvent::Detached(transport)) => {
440                // Only the active transport's detach ends the
441                // session; a displaced transport's stale detach
442                // must not clear the successor's session state.
443                if arbitration.detach(transport) {
444                    env.set_advertising_allowed(true);
445                    session.detach();
446                }
447            }
448            Either3::First(InEvent::Frame(transport, frame_bytes)) => {
449                if arbitration.accepts_frame(transport) {
450                    let now_ms = Instant::now().as_millis();
451                    let effect = session.handle_frame(&frame_bytes, now_ms, &mut |frame: &[u8]| {
452                        emitter.push(frame)
453                    });
454                    emitter.flush(arbitration.destination(), rt.out).await;
455                    match effect {
456                        Some(Effect::SampleRssi { tid }) => {
457                            // Round-trip to the radio runner for an
458                            // instantaneous RSSI sample, then answer the
459                            // deferred PROP_PHY_RSSI get.
460                            rt.ctl.request_rssi();
461                            let sample = rt.ctl.wait_rssi().await;
462                            session.respond_rssi(tid, sample, &mut |frame: &[u8]| {
463                                emitter.push(frame)
464                            });
465                            emitter.flush(arbitration.destination(), rt.out).await;
466                        }
467                        Some(Effect::SampleBattery { tid }) => {
468                            // Round-trip to the platform battery
469                            // source for a fresh measurement, then
470                            // answer the deferred PROP_BATTERY get.
471                            let sample = env.sample_battery().await;
472                            session.respond_battery(tid, sample, &mut |frame: &[u8]| {
473                                emitter.push(frame)
474                            });
475                            emitter.flush(arbitration.destination(), rt.out).await;
476                        }
477                        Some(Effect::DrainQueue) => {
478                            // Deliver the covered frames one per
479                            // step, flushing between steps so the
480                            // two-slot emitter never overflows and
481                            // the transport applies backpressure.
482                            loop {
483                                let more = session.drain_step(
484                                    Instant::now().as_millis(),
485                                    &mut |frame: &[u8]| emitter.push(frame),
486                                );
487                                emitter.flush(arbitration.destination(), rt.out).await;
488                                if !more {
489                                    break;
490                                }
491                            }
492                            env.clear_attention();
493                        }
494                        Some(Effect::WipeHostDomain { tid }) => {
495                            // Durably wipe the host-domain portion of
496                            // any saved snapshot before the new host
497                            // key takes effect; with nothing saved the
498                            // wipe is trivially satisfied.
499                            let result = match session.encode_wiped_snapshot(&mut snapshot_buf) {
500                                Some(len) => env.persist_snapshot(&snapshot_buf[..len]).await,
501                                None => Ok(()),
502                            };
503                            session.respond_host_wipe(tid, result, &mut |frame: &[u8]| {
504                                emitter.push(frame)
505                            });
506                            emitter.flush(arbitration.destination(), rt.out).await;
507                        }
508                        Some(Effect::SaveSnapshot { tid }) => {
509                            let result = match session.encode_snapshot(&mut snapshot_buf) {
510                                Some(len) => env.persist_snapshot(&snapshot_buf[..len]).await,
511                                None => Err(()),
512                            };
513                            session.respond_save(tid, result, &mut |frame: &[u8]| {
514                                emitter.push(frame)
515                            });
516                            emitter.flush(arbitration.destination(), rt.out).await;
517                        }
518                        Some(Effect::ClearSaved { tid }) => {
519                            // CMD_CLEAR covers all persisted
520                            // provisioning: the snapshot and the
521                            // independently persisted device
522                            // identity. Each journal's tombstone is
523                            // individually atomic; an interruption
524                            // between them reports failure and the
525                            // host's retry completes the erase.
526                            let result = match env.clear_snapshot().await {
527                                Ok(()) => env.clear_identity().await,
528                                Err(()) => Err(()),
529                            };
530                            // With the identity durably gone, its
531                            // counter boundaries are dead weight;
532                            // drop them with it. (Kept if the
533                            // identity clear failed — the identity
534                            // then survives the reboot and still
535                            // needs its TX boundary.)
536                            if result.is_ok() {
537                                env.clear_counters().await;
538                            }
539                            session.respond_clear(tid, result, &mut |frame: &[u8]| {
540                                emitter.push(frame)
541                            });
542                            emitter.flush(arbitration.destination(), rt.out).await;
543                        }
544                        Some(Effect::ProvisionIdentity { tid }) => {
545                            // Build the keypair (drawing a fresh
546                            // secret from the platform RNG for
547                            // on-device generation), persist it, and
548                            // only then report the public key.
549                            let result = match session.identity_request() {
550                                Some(source) => {
551                                    let secret = match source {
552                                        IdentitySource::Install(secret) => Ok(secret),
553                                        IdentitySource::Generate => {
554                                            let mut secret = [0u8; 32];
555                                            env.fill_secret(&mut secret).map(|()| secret)
556                                        }
557                                    };
558                                    match secret {
559                                        Ok(secret) => {
560                                            let identity =
561                                                SoftwareIdentity::from_secret_bytes(&secret);
562                                            let public_key = identity.public_key().0;
563                                            let payload =
564                                                proto::encode_identity(&secret, &public_key);
565                                            env.persist_identity(&payload)
566                                                .await
567                                                .map(|()| public_key)
568                                        }
569                                        Err(()) => Err(()),
570                                    }
571                                }
572                                None => Err(()),
573                            };
574                            session.respond_identity(tid, result, &mut |frame: &[u8]| {
575                                emitter.push(frame)
576                            });
577                            emitter.flush(arbitration.destination(), rt.out).await;
578                        }
579                        Some(Effect::SetPairingPin { tid, pin }) => {
580                            let applied = env.apply_pairing_pin(pin).await;
581                            session.respond_pin_set(
582                                tid,
583                                applied.then_some(()).ok_or(()),
584                                &mut |frame: &[u8]| emitter.push(frame),
585                            );
586                            emitter.flush(arbitration.destination(), rt.out).await;
587                        }
588                        Some(Effect::FactoryReset) => {
589                            // Hand off to the platform, which erases every
590                            // persistent journal and reboots. This never
591                            // returns; no acknowledgement is sent because the
592                            // reset drops the link. Any frames the session
593                            // already staged were flushed above.
594                            env.trace(format_args!("CMD_FACTORY_RESET: wiping all state + reboot"));
595                            env.factory_reset().await
596                        }
597                        other => apply_effect(&session, other, &rt, &mut env).await,
598                    }
599                    if session.queued_frame_count() == 0 {
600                        env.clear_attention();
601                    }
602                }
603            }
604            Either3::Second(RxFrame { data, info }) => {
605                // While detached this may stage a delegated MAC
606                // acknowledgement (Effect::StartTransmit).
607                let queued_before = session.queued_frame_count();
608                let effect = session.on_radio_rx(
609                    &data,
610                    info.rssi,
611                    info.snr.as_centibels(),
612                    info.lqi,
613                    Instant::now().as_millis(),
614                    &mut |frame: &[u8]| emitter.push(frame),
615                );
616                if session.queued_frame_count() > queued_before {
617                    env.request_attention();
618                }
619                emitter.flush(arbitration.destination(), rt.out).await;
620                apply_effect(&session, effect, &rt, &mut env).await;
621            }
622            Either3::Third(result) => {
623                let now_ms = Instant::now().as_millis();
624                let effect = session.on_tx_result(result.is_ok(), now_ms, &mut |frame: &[u8]| {
625                    emitter.push(frame)
626                });
627                emitter.flush(arbitration.destination(), rt.out).await;
628                apply_effect(&session, effect, &rt, &mut env).await;
629            }
630        }
631        // Any of the arms may have moved the device-domain tables
632        // (property mutation, CMD_RST, CMD_RESTORE); one u32
633        // compare when they did not.
634        sync_dev_domain(&session, &mut dev_domain_synced, &mut env);
635    }
636}
