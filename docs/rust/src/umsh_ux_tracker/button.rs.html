1//! Button event recognition state machine.
2//!
3//! Resolves a stream of raw [`ButtonEdge`] events plus a monotonic
4//! millisecond clock into the high-level [`ButtonEvent`]s the UX is
5//! defined in terms of: single click, double click, triple click, and
6//! long-press.
7//!
8//! The machine is **pure logic** — no embassy, no hardware, no I/O — so
9//! it can be exhaustively unit-tested with synthetic time. Callers are
10//! expected to:
11//!
12//! 1. Call [`ButtonFsm::on_edge`] for each debounced press / release.
13//! 2. Call [`ButtonFsm::poll`] when the deadline reported by
14//!    [`ButtonFsm::next_deadline`] elapses.
15//!
16//! Both methods can produce a [`ButtonEvent`] when one becomes
17//! resolvable.
18//!
19//! # Recognition rules
20//!
21//! - A "click" is a press-then-release where the hold duration is at
22//!   most `max_click_hold`.
23//! - One, two, or three clicks within `inter_click_gap` of each other
24//!   produce [`ButtonEvent::Single`], [`ButtonEvent::Double`], or
25//!   [`ButtonEvent::Triple`] respectively. Triple fires immediately on
26//!   the third release without waiting for the gap; single / double
27//!   fire after the gap elapses with no further press.
28//! - Holding the button continuously for `long_press` produces
29//!   [`ButtonEvent::Long`] *while the button is still pressed* and
30//!   consumes any clicks accumulated before the hold began.
31//! - Releases longer than `max_click_hold` but shorter than
32//!   `long_press` are *discarded clicks*; if there are accumulated
33//!   prior clicks they are emitted, otherwise nothing fires. This
34//!   matches the rule: a press that's "too long to be a click but too
35//!   short to be a long-press" is a user error and should not silently
36//!   become either.
37
38use core::time::Duration;
39
40/// Raw debounced button transition.
41#[derive(Debug, Clone, Copy, PartialEq, Eq)]
42pub enum ButtonEdge {
43    /// Button transitioned from released to pressed.
44    Press,
45    /// Button transitioned from pressed to released.
46    Release,
47}
48
49/// High-level button event recognized by the FSM.
50#[derive(Debug, Clone, Copy, PartialEq, Eq)]
51pub enum ButtonEvent {
52    Single,
53    Double,
54    Triple,
55    Long,
56}
57
58/// Tunable timings. Defaults match the values in
59/// `docs/firmware-plan-t1000e.md`; tune on real hardware.
60#[derive(Debug, Clone, Copy)]
61pub struct ButtonTimings {
62    /// Longest hold that still counts as a click (rather than a discard
63    /// or a long-press).
64    pub max_click_hold: Duration,
65    /// Maximum gap from a release to the next press to count as part of
66    /// the same click sequence.
67    pub inter_click_gap: Duration,
68    /// Continuous hold duration that triggers [`ButtonEvent::Long`].
69    pub long_press: Duration,
70}
71
72impl Default for ButtonTimings {
73    fn default() -> Self {
74        Self {
75            max_click_hold: Duration::from_millis(250),
76            inter_click_gap: Duration::from_millis(400),
77            long_press: Duration::from_secs(5),
78        }
79    }
80}
81
82#[derive(Debug, Clone, Copy, PartialEq, Eq)]
83enum State {
84    /// No press in progress, no pending click sequence.
85    Idle,
86    /// Button currently held. `pressed_at` is the press timestamp,
87    /// `prior_clicks` is the count of completed clicks before this press.
88    Pressed { pressed_at: u64, prior_clicks: u8 },
89    /// Button released after a valid click, waiting to see whether the
90    /// next press arrives within `inter_click_gap`.
91    WaitingForNext { released_at: u64, clicks: u8 },
92    /// Long-press has already fired; suppress everything until release.
93    LongFired,
94}
95
96/// Button event recognition state machine.
97#[derive(Debug)]
98pub struct ButtonFsm {
99    timings: ButtonTimings,
100    state: State,
101}
102
103impl ButtonFsm {
104    pub fn new(timings: ButtonTimings) -> Self {
105        Self {
106            timings,
107            state: State::Idle,
108        }
109    }
110
111    /// Feed a debounced edge. Returns an event if one is resolvable
112    /// purely from the edge (without waiting for a timeout).
113    pub fn on_edge(&mut self, edge: ButtonEdge, now_ms: u64) -> Option<ButtonEvent> {
114        match (self.state, edge) {
115            // First press of a new sequence.
116            (State::Idle, ButtonEdge::Press) => {
117                self.state = State::Pressed {
118                    pressed_at: now_ms,
119                    prior_clicks: 0,
120                };
121                None
122            }
123
124            // Subsequent press in an in-progress click chord.
125            (State::WaitingForNext { clicks, .. }, ButtonEdge::Press) => {
126                self.state = State::Pressed {
127                    pressed_at: now_ms,
128                    prior_clicks: clicks,
129                };
130                None
131            }
132
133            // Release after a press: classify the hold duration.
134            (
135                State::Pressed {
136                    pressed_at,
137                    prior_clicks,
138                },
139                ButtonEdge::Release,
140            ) => self.classify_release(pressed_at, prior_clicks, now_ms),
141
142            // Release after long-press: reset.
143            (State::LongFired, ButtonEdge::Release) => {
144                self.state = State::Idle;
145                None
146            }
147
148            // Glitchy duplicate edges — ignore.
149            (State::Pressed { .. }, ButtonEdge::Press) => None,
150            (State::WaitingForNext { .. }, ButtonEdge::Release) => None,
151            (State::Idle, ButtonEdge::Release) => None,
152            (State::LongFired, ButtonEdge::Press) => None,
153        }
154    }
155
156    /// Advance time without an edge. Returns an event if a timeout
157    /// resolves one (long-press while held, or single/double after the
158    /// inter-click gap).
159    pub fn poll(&mut self, now_ms: u64) -> Option<ButtonEvent> {
160        match self.state {
161            State::Pressed { pressed_at, .. }
162                if elapsed(pressed_at, now_ms) >= self.timings.long_press =>
163            {
164                self.state = State::LongFired;
165                Some(ButtonEvent::Long)
166            }
167
168            State::WaitingForNext { released_at, clicks }
169                if elapsed(released_at, now_ms) >= self.timings.inter_click_gap =>
170            {
171                self.state = State::Idle;
172                click_count_to_event(clicks)
173            }
174
175            _ => None,
176        }
177    }
178
179    /// Returns the absolute monotonic-millisecond deadline at which
180    /// [`poll`](Self::poll) should next be called, if any.
181    pub fn next_deadline(&self) -> Option<u64> {
182        match self.state {
183            State::Pressed { pressed_at, .. } => {
184                Some(pressed_at + self.timings.long_press.as_millis() as u64)
185            }
186            State::WaitingForNext { released_at, .. } => {
187                Some(released_at + self.timings.inter_click_gap.as_millis() as u64)
188            }
189            State::Idle | State::LongFired => None,
190        }
191    }
192
193    fn classify_release(
194        &mut self,
195        pressed_at: u64,
196        prior_clicks: u8,
197        now_ms: u64,
198    ) -> Option<ButtonEvent> {
199        let hold = elapsed(pressed_at, now_ms);
200
201        if hold >= self.timings.long_press {
202            // Long-press should have already fired in poll; on the off
203            // chance it didn't (e.g. poll wasn't called), fire it now.
204            self.state = State::Idle;
205            return Some(ButtonEvent::Long);
206        }
207
208        if hold > self.timings.max_click_hold {
209            // Held too long for a click, too short for a long-press.
210            // Emit any accumulated prior clicks and reset.
211            self.state = State::Idle;
212            return click_count_to_event(prior_clicks);
213        }
214
215        // Valid click.
216        let clicks = prior_clicks.saturating_add(1);
217        if clicks >= 3 {
218            self.state = State::Idle;
219            return Some(ButtonEvent::Triple);
220        }
221        self.state = State::WaitingForNext {
222            released_at: now_ms,
223            clicks,
224        };
225        None
226    }
227}
228
229fn elapsed(since_ms: u64, now_ms: u64) -> Duration {
230    Duration::from_millis(now_ms.saturating_sub(since_ms))
231}
232
233fn click_count_to_event(clicks: u8) -> Option<ButtonEvent> {
234    match clicks {
235        1 => Some(ButtonEvent::Single),
236        2 => Some(ButtonEvent::Double),
237        3 => Some(ButtonEvent::Triple),
238        _ => None,
239    }
240}

246    fn fsm() -> ButtonFsm {
247        ButtonFsm::new(ButtonTimings::default())
248    }

250    /// Simulate a press at `down_ms` and release at `up_ms` with no
251    /// intervening polls. Returns the edge events plus the result of a
252    /// final `poll` at `up_ms + 500ms` (which is past the inter-click
253    /// gap, so any pending single/double/triple should fire).
254    fn click(fsm: &mut ButtonFsm, down_ms: u64, up_ms: u64) -> (Option<ButtonEvent>, Option<ButtonEvent>) {
255        let on_press = fsm.on_edge(ButtonEdge::Press, down_ms);
256        let on_release = fsm.on_edge(ButtonEdge::Release, up_ms);
257        (on_press, on_release)
258    }

260    #[test]
261    fn idle_release_is_ignored() {
262        let mut fsm = fsm();
263        assert_eq!(fsm.on_edge(ButtonEdge::Release, 0), None);
264    }

266    #[test]
267    fn single_click_fires_after_gap() {
268        let mut fsm = fsm();
269        let (p, r) = click(&mut fsm, 0, 100);
270        assert_eq!(p, None);
271        assert_eq!(r, None);
272
273        // Before the gap, nothing.
274        assert_eq!(fsm.poll(300), None);
275
276        // After the gap, single fires.
277        assert_eq!(fsm.poll(600), Some(ButtonEvent::Single));
278        assert_eq!(fsm.poll(700), None);
279    }

284        click(&mut fsm, 0, 100);
285        click(&mut fsm, 200, 300);

287        assert_eq!(fsm.poll(400), None);
288        assert_eq!(fsm.poll(800), Some(ButtonEvent::Double));
289    }

300    #[test]
301    fn long_press_fires_while_held() {
302        let mut fsm = fsm();
303        assert_eq!(fsm.on_edge(ButtonEdge::Press, 0), None);
304
305        // Before the threshold, nothing.
306        assert_eq!(fsm.poll(4_999), None);
307
308        // At the threshold, long fires.
309        assert_eq!(fsm.poll(5_000), Some(ButtonEvent::Long));
310
311        // Subsequent polls while still held don't re-fire.
312        assert_eq!(fsm.poll(6_000), None);
313
314        // Eventual release returns to idle without producing anything.
315        assert_eq!(fsm.on_edge(ButtonEdge::Release, 7_000), None);
316        assert_eq!(fsm.poll(8_000), None);
317    }

319    #[test]
320    fn long_press_consumes_prior_clicks() {
321        let mut fsm = fsm();
322        click(&mut fsm, 0, 100); // accumulate one click
323        assert_eq!(fsm.on_edge(ButtonEdge::Press, 200), None);
324
325        // Hold for 5 seconds → Long, prior click is lost.
326        assert_eq!(fsm.poll(5_200), Some(ButtonEvent::Long));
327        assert_eq!(fsm.on_edge(ButtonEdge::Release, 5_300), None);
