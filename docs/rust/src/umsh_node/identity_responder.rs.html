1//! Built-in responder for the [Identity Request](crate::mac_command) MAC command.
2//!
3//! When enabled on a [`LocalNode`](crate::LocalNode), the node answers a
4//! matching Identity Request with a **targeted authenticated unicast** carrying
5//! its own [`NodeIdentityPayload`], echoing any request `NONCE` into identity
6//! option 5. Responses are never signed and never fall back to broadcast: the
7//! pairwise MIC, frame counter, and echoed nonce already make the reply
8//! authentic and fresh, and a request whose source cannot be resolved to a key
9//! is simply dropped by the MAC before the responder ever runs.
10//!
11//! The application supplies a [`NodeIdentityProfile`] (its role, capabilities,
12//! and descriptive fields) and, optionally, a respond **policy** — a
13//! registerable discriminator that inspects the [`IdentityRequestContext`] and
14//! decides whether, and how, to answer (e.g. "known peers only", "only on a
15//! given channel", "repeater → always"). No signing key is required.
16

28/// This node's own identity, used to answer Identity Requests.
29///
30/// Holds descriptive fields only — **no signing key**. Config-like fields
31/// (`role`, `capabilities`, `name`, `supported_regions`) are typically set once
32/// at bring-up; live fields (`location`, `altitude_m`) can be refreshed at any
33/// time via [`LocalNode::update_identity_profile`](crate::LocalNode::update_identity_profile),
34/// e.g. from a GPS task.
35#[derive(Clone, Debug)]
36pub struct NodeIdentityProfile {
37    /// This node's public key. Its [hint](PublicKey::hint) is matched against a
38    /// request's `FILTER_NODE_HINT`; the key itself reaches the requester via
39    /// the reply's MAC source address, not the identity payload.
40    pub public_key: PublicKey,
41    pub role: NodeRole,
42    pub capabilities: NodeCapabilities,
43    pub name: Option<String>,
44    pub location: Option<NodeLocation>,
45    pub altitude_m: Option<i32>,
46    pub supported_regions: Option<Vec<u8>>,
47}
48
49impl NodeIdentityProfile {
50    /// Create a minimal profile (role + capabilities), no descriptive options.
51    pub fn new(public_key: PublicKey, role: NodeRole, capabilities: NodeCapabilities) -> Self {
52        Self {
53            public_key,
54            role,
55            capabilities,
56            name: None,
57            location: None,
58            altitude_m: None,
59            supported_regions: None,
60        }
61    }
62
63    /// Set the display name (builder style).
64    pub fn with_name(mut self, name: impl Into<String>) -> Self {
65        self.name = Some(name.into());
66        self
67    }
68
69    /// Set the geographic location (builder style).
70    pub fn with_location(mut self, location: NodeLocation) -> Self {
71        self.location = Some(location);
72        self
73    }
74
75    /// This node's hint, derived from its public key.
76    pub fn hint(&self) -> NodeHint {
77        self.public_key.hint()
78    }
79
80    /// Build the (unsigned) identity payload for a response, stamping `nonce`
81    /// into option 5.
82    fn to_payload(&self, nonce: Option<u32>) -> NodeIdentityPayload {
83        NodeIdentityPayload {
84            role: self.role,
85            capabilities: self.capabilities,
86            name: self.name.clone(),
87            location: self.location,
88            altitude_m: self.altitude_m,
89            timestamp: None,
90            supported_regions: self.supported_regions.clone(),
91            nonce,
92            signature: None,
93        }
94    }
95}

97/// Reception context for an incoming Identity Request, handed to the respond
98/// policy so it can decide whether — and how — to answer.
99///
100/// The request has already passed the filter gate (its `FILTER_*` options
101/// select this node) and its source has already been resolved to a key, so the
102/// policy only governs the "do I want to answer *this sender*?" decision.
103pub struct IdentityRequestContext<'a> {
104    /// Resolved sender key. Always present: an unresolvable source is dropped
105    /// before the policy runs.
106    pub from_key: PublicKey,
107    /// Sender hint, when the frame carried one.
108    pub from_hint: Option<NodeHint>,
109    /// Whether the request frame was authenticated (pairwise or channel MIC).
110    pub source_authenticated: bool,
111    /// Whether the request carried the sender's full 32-byte key.
112    pub has_full_source: bool,
113    /// The channel the request arrived on, if any (`None` for plain
114    /// broadcast/unicast).
115    pub channel: Option<ChannelId>,
116    /// Coarse packet family (Unicast / Broadcast / Multicast / BlindUnicast).
117    pub family: PacketFamily,
118    /// The request's filter/option block, for policies that inspect it further.
119    pub filters: IdentityRequestFilters<'a>,
120    /// Received signal strength of the request, if measured.
121    pub rssi: Option<i16>,
122    /// Signal-to-noise ratio of the request, if measured.
123    pub snr: Option<Snr>,
124}
125
126/// A respond policy's verdict for one Identity Request.
127#[derive(Clone, Copy, Debug, PartialEq, Eq)]
128pub enum RespondDecision {
129    /// Do not answer this request.
130    Ignore,
131    /// Answer with an authenticated unicast identity response.
132    Respond {
133        /// Include our full 32-byte key in the reply's source address, so a
134        /// requester that only had our hint can authenticate the reply without
135        /// a prior key exchange. Set `false` only when the requester already
136        /// holds our key.
137        full_source: bool,
138    },
139}
140
141/// A registerable respond policy: given the request context, decide the verdict.
142pub type RespondPolicy = dyn FnMut(&IdentityRequestContext<'_>) -> RespondDecision;
143
144/// The default respond policy: answer every request that reached the policy.
145///
146/// Includes our full source key unless the request was authenticated to us
147/// (an authenticated pairwise request implies the sender already holds our
148/// key). The blind-unicast-over-channel case, where a channel-authenticated
149/// sender may still lack our key, is a known edge a custom policy can override.
150pub fn default_respond_policy(ctx: &IdentityRequestContext<'_>) -> RespondDecision {
151    RespondDecision::Respond {
152        full_source: !ctx.source_authenticated,
153    }
154}
155
156/// Installed responder state: the profile plus the active policy.
157pub(crate) struct IdentityResponder {
158    pub(crate) profile: NodeIdentityProfile,
159    pub(crate) policy: Box<RespondPolicy>,
160}
161
162/// A resolved plan to answer one Identity Request, produced synchronously while
163/// the node state is borrowed and executed later by the async pump.
164pub(crate) struct IdentityResponsePlan {
165    /// Destination (the requester).
166    pub(crate) to: PublicKey,
167    /// Whether the reply should carry our full source key.
168    pub(crate) full_source: bool,
169    /// The framed reply payload: `PayloadType::NodeIdentity` + encoded identity.
170    pub(crate) framed: Vec<u8>,
171}
172
173impl IdentityResponder {
174    /// Evaluate an incoming request against the profile and policy.
175    ///
176    /// Returns `Some(plan)` when the node should answer: the request's filters
177    /// select this node **and** the policy returns `Respond`. Returns `None`
178    /// otherwise (not selected, policy said `Ignore`, or the reply could not be
179    /// encoded).
180    pub(crate) fn evaluate(
181        &mut self,
182        ctx: &IdentityRequestContext<'_>,
183    ) -> Option<IdentityResponsePlan> {
184        // Filter gate: does this request target a node like us?
185        let our_hint = self.profile.hint();
186        if !ctx
187            .filters
188            .selects(self.profile.role, self.profile.capabilities, &our_hint)
189            .unwrap_or(false)
190        {
191            return None;
192        }
193
194        // Policy gate: do we want to answer this particular sender?
195        let full_source = match (self.policy)(ctx) {
196            RespondDecision::Ignore => return None,
197            RespondDecision::Respond { full_source } => full_source,
198        };
199
200        // Build the framed reply, echoing the request nonce into option 5.
201        let nonce = ctx.filters.nonce().ok().flatten();
202        let payload = self.profile.to_payload(nonce);
203        let mut buf = [0u8; 192];
204        buf[0] = PayloadType::NodeIdentity as u8;
205        let len = 1 + payload.encode(&mut buf[1..]).ok()?;
206        Some(IdentityResponsePlan {
207            to: ctx.from_key,
208            full_source,
209            framed: Vec::from(&buf[..len]),
210        })
211    }
