4#[derive(Clone, Copy, Debug, PartialEq, Eq)]
5#[repr(u8)]
6pub enum CommandId {
7    BeaconRequest = 0,
8    IdentityRequest = 1,
9    SignalReportRequest = 2,
10    SignalReportResponse = 3,
11    EchoRequest = 4,
12    EchoResponse = 5,
13    PfsSessionRequest = 6,
14    PfsSessionResponse = 7,
15    EndPfsSession = 8,

18#[derive(Clone, Copy, Debug, PartialEq, Eq)]
19pub enum MacCommand<'a> {
20    BeaconRequest { nonce: Option<u32> },
21    IdentityRequest,
22    SignalReportRequest,
23    SignalReportResponse { rssi: u8, snr: i8 },
24    EchoRequest { data: &'a [u8] },
25    EchoResponse { data: &'a [u8] },
26    PfsSessionRequest {
27        ephemeral_key: umsh_core::PublicKey,
28        duration_minutes: u16,
29    },
30    PfsSessionResponse {
31        ephemeral_key: umsh_core::PublicKey,
32        duration_minutes: u16,
33    },
34    EndPfsSession,
35}

37pub fn parse(payload: &[u8]) -> Result<MacCommand<'_>, AppParseError> {
38    let (&command_id, body) = payload
39        .split_first()
40        .ok_or(AppParseError::Core(umsh_core::ParseError::Truncated))?;
41
42    match command_id {
43        0 => match body {
44            [] => Ok(MacCommand::BeaconRequest { nonce: None }),
45            [a, b, c, d] => Ok(MacCommand::BeaconRequest {
46                nonce: Some(u32::from_be_bytes([*a, *b, *c, *d])),
47            }),
48            _ => Err(AppParseError::InvalidLength {
49                expected: 4,
50                actual: body.len(),
51            }),
52        },
53        1 => {
54            if body.is_empty() {
55                Ok(MacCommand::IdentityRequest)
56            } else {
57                Err(AppParseError::InvalidOptionValue)
58            }
59        }
60        2 => {
61            if body.is_empty() {
62                Ok(MacCommand::SignalReportRequest)
63            } else {
64                Err(AppParseError::InvalidOptionValue)
65            }
66        }
67        3 => match body {
68            [rssi, snr] => Ok(MacCommand::SignalReportResponse {
69                rssi: *rssi,
70                snr: *snr as i8,
71            }),
72            _ => Err(AppParseError::InvalidLength {
73                expected: 2,
74                actual: body.len(),
75            }),
76        },
77        4 => Ok(MacCommand::EchoRequest { data: body }),
78        5 => Ok(MacCommand::EchoResponse { data: body }),
79        6 => parse_pfs(body, true),
80        7 => parse_pfs(body, false),
81        8 => {
82            if body.is_empty() {
83                Ok(MacCommand::EndPfsSession)
84            } else {
85                Err(AppParseError::InvalidOptionValue)
86            }
87        }
88        other => Err(AppParseError::InvalidCommandId(other)),
89    }
90}

92fn parse_pfs(payload: &[u8], request: bool) -> Result<MacCommand<'_>, AppParseError> {
93    if payload.len() != 34 {
94        return Err(AppParseError::InvalidLength {
95            expected: 34,
96            actual: payload.len(),
97        });
98    }
99    let ephemeral_key = umsh_core::PublicKey(*fixed(&payload[..32])?);
100    let duration_minutes = u16::from_be_bytes(*fixed(&payload[32..34])?);
101    Ok(if request {
102        MacCommand::PfsSessionRequest {
103            ephemeral_key,
104            duration_minutes,
105        }
106    } else {
107        MacCommand::PfsSessionResponse {
108            ephemeral_key,
109            duration_minutes,
110        }
111    })
112}

114pub fn encode(cmd: &MacCommand<'_>, buf: &mut [u8]) -> Result<usize, AppEncodeError> {
115    let mut pos = 0usize;
116    match cmd {
117        MacCommand::BeaconRequest { nonce } => {
118            push_byte(buf, &mut pos, CommandId::BeaconRequest as u8)?;
119            if let Some(nonce) = nonce {
120                copy_into(buf, &mut pos, &nonce.to_be_bytes())?;
121            }
122        }
123        MacCommand::IdentityRequest => push_byte(buf, &mut pos, CommandId::IdentityRequest as u8)?,
124        MacCommand::SignalReportRequest => {
125            push_byte(buf, &mut pos, CommandId::SignalReportRequest as u8)?;
126        }
127        MacCommand::SignalReportResponse { rssi, snr } => {
128            push_byte(buf, &mut pos, CommandId::SignalReportResponse as u8)?;
129            push_byte(buf, &mut pos, *rssi)?;
130            push_byte(buf, &mut pos, *snr as u8)?;
131        }
132        MacCommand::EchoRequest { data } => {
133            push_byte(buf, &mut pos, CommandId::EchoRequest as u8)?;
134            copy_into(buf, &mut pos, data)?;
135        }
136        MacCommand::EchoResponse { data } => {
137            push_byte(buf, &mut pos, CommandId::EchoResponse as u8)?;
138            copy_into(buf, &mut pos, data)?;
139        }
140        MacCommand::PfsSessionRequest {
141            ephemeral_key,
142            duration_minutes,
143        } => {
144            push_byte(buf, &mut pos, CommandId::PfsSessionRequest as u8)?;
145            copy_into(buf, &mut pos, &ephemeral_key.0)?;
146            copy_into(buf, &mut pos, &duration_minutes.to_be_bytes())?;
147        }
148        MacCommand::PfsSessionResponse {
149            ephemeral_key,
150            duration_minutes,
151        } => {
152            push_byte(buf, &mut pos, CommandId::PfsSessionResponse as u8)?;
153            copy_into(buf, &mut pos, &ephemeral_key.0)?;
154            copy_into(buf, &mut pos, &duration_minutes.to_be_bytes())?;
155        }
156        MacCommand::EndPfsSession => push_byte(buf, &mut pos, CommandId::EndPfsSession as u8)?,
157    }
158    Ok(pos)
159}
