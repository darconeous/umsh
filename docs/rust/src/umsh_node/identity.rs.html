19#[derive(Clone, Copy, Debug, PartialEq, Eq)]
20pub enum NodeRole {
21    Unspecified,
22    Repeater,
23    Chat,
24    Tracker,
25    Sensor,
26    Bridge,
27    ChatRoom,
28    TemporarySession,
29    /// A role value not recognized by this implementation; preserved for round-tripping.
30    Unknown(u8),
31}
32
33impl NodeRole {
34    pub fn from_byte(value: u8) -> Self {
35        match value {
36            0 => Self::Unspecified,
37            1 => Self::Repeater,
38            2 => Self::Chat,
39            3 => Self::Tracker,
40            4 => Self::Sensor,
41            5 => Self::Bridge,
42            6 => Self::ChatRoom,
43            7 => Self::TemporarySession,
44            n => Self::Unknown(n),
45        }
46    }
47
48    pub fn as_byte(self) -> u8 {
49        match self {
50            Self::Unspecified => 0,
51            Self::Repeater => 1,
52            Self::Chat => 2,
53            Self::Tracker => 3,
54            Self::Sensor => 4,
55            Self::Bridge => 5,
56            Self::ChatRoom => 6,
57            Self::TemporarySession => 7,
58            Self::Unknown(n) => n,
59        }
60    }
61}

75#[derive(Clone, Debug, PartialEq, Eq)]
76pub struct NodeIdentityPayload {
77    pub role: NodeRole,
78    pub capabilities: NodeCapabilities,
79    /// Option 0 — display name (UTF-8).
80    pub name: Option<String>,
81    /// Option 1 — geographic position.
82    pub location: Option<NodeLocation>,
83    /// Option 2 — altitude above mean sea level, in meters.
84    pub altitude_m: Option<i32>,
85    /// Option 3 — seconds since the Unix epoch (freshness marker).
86    pub timestamp: Option<u32>,
87    /// Option 4 — concatenated 2-byte region codes this repeater serves.
88    pub supported_regions: Option<Vec<u8>>,
89    /// EdDSA signature over ROLE..=0xFF, present when the identity stands alone.
90    ///
91    /// TODO: signing and verification belong outside `NodeIdentityPayload`
92    /// (the way the source-address routing hint lives outside the payload).
93    /// The eventual rework should remove this field and move the
94    /// signed-identity wrapper to a separate type that owns both the encoded
95    /// payload bytes and the signature, so the signed-byte range is never
96    /// reconstructed at the call site.
97    pub signature: Option<[u8; 64]>,
98}

100impl NodeIdentityPayload {
101    pub fn from_bytes(payload: &[u8]) -> Result<NodeIdentityPayload, AppParseError> {
102        if payload.len() < 2 {
103            return Err(AppParseError::Core(umsh_core::ParseError::Truncated));
104        }
105
106        let role = NodeRole::from_byte(payload[0]);
107        let capabilities = NodeCapabilities::from_bits_truncate(payload[1]);
108        let remaining = &payload[2..];
109
110        let mut name = None;
111        let mut location = None;
112        let mut altitude_m = None;
113        let mut timestamp = None;
114        let mut supported_regions = None;
115
116        let mut decoder = OptionDecoder::new(remaining);
117        for result in decoder.by_ref() {
118            let (number, value) = result?;
119            match number {
120                opt::NAME => name = Some(String::from(parse_utf8(value)?)),
121                opt::LOCATION => {
122                    // Spec: MUST ignore bytes after the 7th
123                    location = Some(NodeLocation::from_bytes(value));
124                }
125                opt::ALTITUDE => altitude_m = Some(parse_be_i32(value)?),
126                opt::TIMESTAMP => timestamp = Some(parse_be_u32(value)?),
127                opt::SUPPORTED_REGIONS => {
128                    if value.len() % 2 != 0 {
129                        return Err(AppParseError::InvalidOptionValue);
130                    }
131                    supported_regions = Some(Vec::from(value));
132                }
133                _ => {} // unknown options are silently skipped
134            }
135        }
136
137        let sig_bytes = decoder.remainder();
138        let signature = match sig_bytes.len() {
139            0 => None,
140            64 => Some(
141                sig_bytes
142                    .try_into()
143                    .map_err(|_| AppParseError::InvalidLength {
144                        expected: 64,
145                        actual: sig_bytes.len(),
146                    })?,
147            ),
148            n => {
149                return Err(AppParseError::InvalidLength {
150                    expected: 64,
151                    actual: n,
152                });
153            }
154        };
155
156        Ok(NodeIdentityPayload {
157            role,
158            capabilities,
159            name,
160            location,
161            altitude_m,
162            timestamp,
163            supported_regions,
164            signature,
165        })
166    }
167
168    pub fn encode(&self, buf: &mut [u8]) -> Result<usize, AppEncodeError> {
169        if buf.len() < 2 {
170            return Err(AppEncodeError::BufferTooSmall);
171        }
172        buf[0] = self.role.as_byte();
173        buf[1] = self.capabilities.bits();
174        let mut pos = 2;
175
176        {
177            let mut enc = OptionEncoder::new(&mut buf[pos..]);
178            if let Some(name) = self.name.as_deref() {
179                enc.put(opt::NAME, name.as_bytes())?;
180            }
181            if let Some(loc) = self.location {
182                enc.put(opt::LOCATION, loc.as_bytes())?;
183            }
184            if let Some(alt) = self.altitude_m {
185                enc.put_i32(opt::ALTITUDE, alt)?;
186            }
187            if let Some(ts) = self.timestamp {
188                enc.put_u32(opt::TIMESTAMP, ts)?;
189            }
190            if let Some(regions) = self.supported_regions.as_deref() {
191                enc.put(opt::SUPPORTED_REGIONS, regions)?;
192            }
193            if self.signature.is_some() {
194                enc.end_marker()?;
195            }
196            pos += enc.finish();
197        }
198
199        if let Some(sig) = &self.signature {
200            if pos + 64 > buf.len() {
201                return Err(AppEncodeError::BufferTooSmall);
202            }
203            buf[pos..pos + 64].copy_from_slice(sig);
204            pos += 64;
205        }
206
207        Ok(pos)
208    }
209}

215    fn round_trip(id: &NodeIdentityPayload) -> bool {
216        let mut buf = [0u8; 256];
217        let len = id.encode(&mut buf).expect("encode failed");
218        let decoded = NodeIdentityPayload::from_bytes(&buf[..len]).expect("parse failed");
219        decoded == *id
220    }

222    #[test]
223    fn minimal_two_bytes() {
224        let id = NodeIdentityPayload {
225            role: NodeRole::Chat,
226            capabilities: NodeCapabilities::TEXT_MESSAGES,
227            name: None,
228            location: None,
229            altitude_m: None,
230            timestamp: None,
231            supported_regions: None,
232            signature: None,
233        };
234        let mut buf = [0u8; 16];
235        let len = id.encode(&mut buf).unwrap();
236        assert_eq!(len, 2);
237        assert_eq!(buf[0], 2); // Chat role
238        assert_eq!(buf[1], NodeCapabilities::TEXT_MESSAGES.bits());
239        assert!(round_trip(&id));
240    }
