14pub(crate) fn copy_into(dst: &mut [u8], pos: &mut usize, src: &[u8]) -> Result<(), AppEncodeError> {
15    if dst.len().saturating_sub(*pos) < src.len() {
16        return Err(AppEncodeError::BufferTooSmall);
17    }
18    dst[*pos..*pos + src.len()].copy_from_slice(src);
19    *pos += src.len();
20    Ok(())
21}
