1// LilyGO T-Echo companion-radio NCP firmware.
2//
3// Exposes the SX1262 as a host-controlled PHY speaking the minimal
4// companion-radio protocol (docs/protocol/src/companion-radio-minimal.md)
5// with HDLC-Lite framing over USB-CDC. The UMSH MAC does not run here:
6// the host owns it and drives this device through
7// `umsh::companion_radio::CompanionRadio`.
8//
9// Protocol behavior lives in `umsh-companion-ncp::Session` (host-tested,
10// no I/O); this binary is only glue:
11//
12// Task layout (steady state):
13//   - main():            joins usb.run / heartbeat (LED + WDT pet)
14//   - radio_task:        owns lora_phy::LoRa via umsh_radio_loraphy::ncp_runner;
15//                        modulation/frequency/power are pushed at runtime
16//                        through NCP_CTL as the host sets properties
17//   - usb_in_task:       owns CdcAcmRescue + HDLC decoder; forwards frames and
18//                        attach/detach edges into INPUT_CH (keeps
19//                        read_packet out of any select, so cancel safety
20//                        never depends on the USB driver)
21//   - ncp_task:          owns the framing-free Session; sorts INPUT_CH,
22//                        radio RX, and TX completions into session calls
23//   - output_task:       owns the USB Sender + HDLC encoder, drains OUT_USB_CH
24//   - button_task:       2 s hold on the user button fires SHUTDOWN_SIGNAL
25//   - shutdown_task:     tri-states peripheral pins, drops the rail,
26//                        enters System OFF
27//
28// CMD_RST is a protocol-level reset: session state returns to post-reset
29// defaults and the radio is re-applied (disabled), but the MCU and the
30// USB link stay up. Host attach also resets the session, silently — the
31// reset notice is only emitted in response to CMD_RST so the host never
32// sees an unsolicited reset it didn't ask for mid-handshake.
33//
34// Safety primitives inherited from the BSP (see umsh-bsp-nrf52840):
35//   * Panic capture into reserved RAM (reported as STATUS_RESET_CRASH).
36//   * 1200-baud touchless reset and Ctrl-C × 3 + "dfu" escape to
37//     bootloader (baked into CdcAcmRescue).
38//   * Watchdog.
39

198    /// Largest companion frame the session emits (CMD_STR_RECV around a
199    /// full-MTU payload).
200    const FRAME_OUT_MAX: usize = 300;
201    const WIRE_MAX: usize = hdlc::max_encoded_len(300);
202
203    /// Collects frames emitted synchronously by the session, then
204    /// flushes them to OUT_USB_CH asynchronously. The session emits at
205    /// most one frame per call; two slots give headroom.
206    struct Emitter {
207        bufs: [[u8; FRAME_OUT_MAX]; 2],
208        lens: [usize; 2],
209        count: usize,
210    }
211
212    impl Emitter {
213        const fn new() -> Self {
214            Self {
215                bufs: [[0; FRAME_OUT_MAX]; 2],
216                lens: [0; 2],
217                count: 0,
218            }
219        }
220
221        /// Copy one raw companion frame into the next slot.
222        ///
223        /// The session is expected to emit at most `bufs.len()` frames per call
224        /// and every frame is expected to fit `FRAME_OUT_MAX`. Both invariants are
225        /// asserted in debug builds so a future session change that violates
226        /// them is caught rather than silently dropping a response.
227        fn push(&mut self, frame: &[u8]) {
228            if self.count >= self.bufs.len() {
229                debug_assert!(
230                    false,
231                    "Emitter overflow: session emitted more frames per call than staging slots"
232                );
233                return;
234            }
235            if frame.len() <= FRAME_OUT_MAX {
236                self.bufs[self.count][..frame.len()].copy_from_slice(frame);
237                self.lens[self.count] = frame.len();
238                self.count += 1;
239            } else {
240                debug_assert!(false, "Emitter: companion frame exceeds FRAME_OUT_MAX");
241            }
242        }
243
244        /// Queue all staged frames for the USB output task.
245        async fn flush(&mut self, destination: Option<(Transport, u32)>) {
246            if let Some((Transport::Usb, generation)) = destination {
247                for index in 0..self.count {
248                    let mut frame: FrameBuf = heapless::Vec::new();
249                    let _ = frame.extend_from_slice(&self.bufs[index][..self.lens[index]]);
250                    OUT_USB_CH.send(OutFrame { generation, frame }).await;
251                }
252            }
253            self.count = 0;
254        }
255    }

257    /// Execute a radio side effect requested by the session.
258    async fn apply_effect(session: &Session, effect: Option<Effect>) {
259        match effect {
260            Some(Effect::ApplyRadio(settings)) => {
261                // The session validates values against the same discrete
262                // sets these converters accept, so None here is
263                // unreachable; bail out defensively rather than panic.
264                let (Some(sf), Some(bw), Some(cr)) = (
265                    spreading_factor_from_u8(settings.sf),
266                    bandwidth_from_hz(settings.bw_hz),
267                    coding_rate_from_denom(settings.cr_denom),
268                ) else {
269                    return;
270                };
271                NCP_CTL.apply(NcpSettings {
272                    enabled: settings.enabled,
273                    freq_hz: settings.freq_khz.saturating_mul(1_000),
274                    sf,
275                    bw,
276                    cr,
277                    power_dbm: i32::from(settings.tx_power_dbm),
278                });
279            }
280            Some(Effect::StartTransmit) => {
281                let mut data: heapless::Vec<u8, MAX_PAYLOAD> = heapless::Vec::new();
282                let _ = data.extend_from_slice(session.tx_data());
283                let power_dbm = match session.tx_power() {
284                    TxPower::Default => None,
285                    TxPower::Max => Some(i32::from(MAX_TX_POWER_DBM)),
286                    TxPower::Dbm(dbm) => Some(i32::from(dbm)),
287                };
288                RADIO_CH.tx.send(TxRequest { data, power_dbm }).await;
289            }
290            // SampleRssi needs `&mut Session` + the emitter, so it is handled
291            // inline in ncp_task rather than here.
292            Some(Effect::SampleRssi { .. }) | Some(Effect::SetPairingPin { .. }) | None => {}
293        }
294    }

305    /// Owns the USB `Sender`, HDLC-encodes frames, and writes USB packets.
306    #[embassy_executor::task]
307    async fn output_task(mut tx: TechoSender) {
308        loop {
309            let outbound = OUT_USB_CH.receive().await;
310            if SESSION_GEN.load(Ordering::Acquire) != outbound.generation {
311                continue;
312            }
313            let mut wire = [0u8; WIRE_MAX];
314            let Ok(len) = hdlc::encode_frame(&outbound.frame, &mut wire) else {
315                continue;
316            };
317            for chunk in wire[..len].chunks(64) {
318                if SESSION_GEN.load(Ordering::Acquire) != outbound.generation {
319                    break;
320                }
321                let _ = tx.write_packet(chunk).await;
322            }
323        }
324    }

326    /// Owns the CDC receive half and HDLC decoder. Forwards frames and edges
327    /// into INPUT_CH; `wait_connection` always precedes the read loop so
328    /// a disconnected port never busy-loops.
329    #[embassy_executor::task]
330    async fn usb_in_task(mut rx: TechoRescue) {
331        loop {
332            rx.wait_connection().await;
333            let mut decoder: hdlc::Decoder<FRAME_IN_MAX> = hdlc::Decoder::new();
334            let mut local_generation = SESSION_GEN.load(Ordering::Acquire);
335            INPUT_CH.send(InEvent::Attached(Transport::Usb)).await;
336            loop {
337                let generation = SESSION_GEN.load(Ordering::Acquire);
338                if generation != local_generation {
339                    decoder.reset();
340                    local_generation = generation;
341                }
342                let mut packet = [0u8; 64];
343                match rx.read_packet(&mut packet).await {
344                    Ok(0) | Err(_) => break,
345                    Ok(len) => {
346                        for &byte in &packet[..len] {
347                            let Some(Ok(bytes)) = decoder.push(byte) else {
348                                continue;
349                            };
350                            let mut frame = heapless::Vec::new();
351                            let _ = frame.extend_from_slice(bytes);
352                            INPUT_CH.send(InEvent::Frame(Transport::Usb, frame)).await;
353                        }
354                    }
355                }
356            }
357            INPUT_CH.send(InEvent::Detached(Transport::Usb)).await;
358        }
359    }

361    /// Owns the framing-free protocol session. Sorts host frames,
362    /// radio receptions, and transmit completions into session calls and
363    /// executes the resulting radio effects.
364    #[embassy_executor::task]
365    async fn ncp_task(boot_reason: Status) {
366        let mut session = Session::new(session_config());
367        let mut emitter = Emitter::new();
368        let mut active = None;
369        let mut generation = SESSION_GEN.load(Ordering::Acquire);
370
371        loop {
372            // Only wait for a TX completion while one is outstanding,
373            // so a spurious tx_done can never be consumed early.
374            let tx_done = async {
375                if session.has_pending_tx() {
376                    RADIO_CH.tx_done.wait().await
377                } else {
378                    core::future::pending().await
379                }
380            };
381
382            match select3(INPUT_CH.receive(), RADIO_CH.rx.receive(), tx_done).await {
383                Either3::First(InEvent::Attached(transport)) => {
384                    // Fresh protocol state for the new host session.
385                    // Silent: the reset notice is only sent for CMD_RST,
386                    // so the host never races a stray notice during its
387                    // own reset handshake.
388                    generation = SESSION_GEN.fetch_add(1, Ordering::AcqRel).wrapping_add(1);
389                    active = Some(transport);
390                    let effect = session.reset(boot_reason, &mut |_frame: &[u8]| {});
391                    apply_effect(&session, Some(effect)).await;
392                }
393                Either3::First(InEvent::Detached(transport)) => {
394                    if active == Some(transport) {
395                        active = None;
396                    }
397                }
398                Either3::First(InEvent::Frame(transport, frame_bytes)) => {
399                    if active == Some(transport) {
400                        let now_ms = Instant::now().as_millis();
401                        let effect =
402                            session.handle_frame(&frame_bytes, now_ms, &mut |frame: &[u8]| {
403                                emitter.push(frame)
404                            });
405                        emitter
406                            .flush(active.map(|transport| (transport, generation)))
407                            .await;
408                        match effect {
409                            Some(Effect::SampleRssi { tid }) => {
410                                // Round-trip to the radio runner for an
411                                // instantaneous RSSI sample, then answer the
412                                // deferred PROP_PHY_RSSI get.
413                                NCP_CTL.request_rssi();
414                                let sample = NCP_CTL.wait_rssi().await;
415                                session.respond_rssi(tid, sample, &mut |frame: &[u8]| {
416                                    emitter.push(frame)
417                                });
418                                emitter
419                                    .flush(active.map(|transport| (transport, generation)))
420                                    .await;
421                            }
422                            Some(Effect::SetPairingPin { tid, pin }) => {
423                                PAIRING_PIN.store(pin.unwrap_or(u32::MAX), Ordering::Release);
424                                session.respond_pin_set(tid, Ok(()), &mut |frame: &[u8]| {
425                                    emitter.push(frame)
426                                });
427                                emitter
428                                    .flush(active.map(|transport| (transport, generation)))
429                                    .await;
430                            }
431                            other => apply_effect(&session, other).await,
432                        }
433                    }
434                }
435                Either3::Second(RxFrame { data, info }) => {
436                    session.on_radio_rx(
437                        &data,
438                        info.rssi,
439                        info.snr.as_centibels(),
440                        info.lqi,
441                        &mut |frame: &[u8]| emitter.push(frame),
442                    );
443                    emitter
444                        .flush(active.map(|transport| (transport, generation)))
445                        .await;
446                }
447                Either3::Third(result) => {
448                    let now_ms = Instant::now().as_millis();
449                    session.on_tx_result(result.is_ok(), now_ms, &mut |frame: &[u8]| {
450                        emitter.push(frame)
451                    });
452                    emitter
453                        .flush(active.map(|transport| (transport, generation)))
454                        .await;
455                }
456            }
457        }
458    }

460    /// Long-press watcher for the user button on P1.10 (active-low,
461    /// pull-up). Two-second hold fires SHUTDOWN_SIGNAL.
462    #[embassy_executor::task]
463    async fn button_task(mut button: Input<'static>) {
464        const HOLD: Duration = Duration::from_secs(2);
465        loop {
466            button.wait_for_low().await;
467            match select(button.wait_for_high(), Timer::after(HOLD)).await {
468                Either::First(()) => {
469                    // Released before HOLD — no-op.
470                }
471                Either::Second(()) => {
472                    SHUTDOWN_SIGNAL.signal(());
473                    button.wait_for_high().await;
474                }
475            }
476        }
477    }

479    /// Controlled power-off: tri-state peripheral signal pins, drop the
480    /// peripheral rail, enter System OFF. No MAC counters or display
481    /// handshake in this firmware. See companion-cli-techo for the
482    /// rationale behind each tri-stated pin.
483    #[embassy_executor::task]
484    async fn shutdown_task(peripheral_power: Output<'static>) -> ! {
485        SHUTDOWN_SIGNAL.wait().await;
486
487        // E-paper SPI bus (SPIM2): SCK=P0.31, MOSI=P1.07, MISO=P0.29
488        // E-paper control:         CS=P0.30, DC=P0.28, RST=P0.02, BUSY=P0.03
489        // Radio SPI bus (TWISPI1): SCK=P0.19, MOSI=P0.22, MISO=P0.23
490        // Radio control:           CS=P0.24, RST=P0.25, BUSY=P0.17, DIO1=P0.20
491        // (E-paper pins are unconfigured in this firmware; tri-stating
492        // them anyway is harmless and keeps the list identical to the
493        // CLI firmware.)
494        for (port, pin) in [
495            (Port::P0, 31u8),
496            (Port::P1, 7u8),
497            (Port::P0, 29u8),
498            (Port::P0, 30u8),
499            (Port::P0, 28u8),
500            (Port::P0, 2u8),
501            (Port::P0, 3u8),
502            (Port::P0, 19u8),
503            (Port::P0, 22u8),
504            (Port::P0, 23u8),
505            (Port::P0, 24u8),
506            (Port::P0, 25u8),
507            (Port::P0, 17u8),
508            (Port::P0, 20u8), // radio DIO1 ← has SENSE set by async radio wait
509        ] {
510            tristate_pin(port, pin);
511        }
512
513        drop(peripheral_power);
514
515        // P1.10 is the side user button. Active-low, pull-up → DETECT-low wakes.
516        power_off(&[WakePin {
517            port: Port::P1,
518            pin: 10,
519            sense: WakeSense::Low,
520        }])
521    }

525    #[embassy_executor::main]
526    async fn main(spawner: Spawner) {
527        let p = embassy_nrf::init(umsh_bsp_nrf52840::clocks::default_config());
528
529        // Peripheral power enable (P0.12). Must be high before the LoRa
530        // module is addressed. Ownership transfers to shutdown_task.
531        let peripheral_power = Output::new(p.P0_12, Level::High, OutputDrive::Standard);
532
533        // WDT: 8 s timeout, petted by the heartbeat task every ~2 s.
534        let mut wdt_config = WdtConfig::default();
535        wdt_config.timeout_ticks = 32768 * 8;
536        let (_wdt, [wdt_handle]) =
537            Watchdog::try_new::<_, 1>(p.WDT, wdt_config).unwrap_or_else(|_| panic!("wdt"));
538
539        // A message in the panic slot means the last reset was a crash;
540        // report that as the reset reason. The slot is cleared either way.
541        let boot_reason = {
542            let mut slot = PanicSlot::new(super::panic::panic_region());
543            if slot.read().is_some() {
544                slot.clear();
545                Status::RESET_CRASH
546            } else {
547                Status::RESET_POWER_ON
548            }
549        };
550
551        // ── SX1262 LoRa radio ────────────────────────────────────────────────
552        // Pin assignment (T-Echo hardware, firmware-confirmed):
553        //   SPI bus: SCK=P0.19, MOSI=P0.22, MISO=P0.23 (TWISPI1)
554        //   CS=P0.24, RST=P0.25, BUSY=P0.17, DIO1=P0.20
555        //   DIO2: internal RF switch; DIO3: 1.8 V TCXO.
556        {
557            let mut cfg = SpimConfig::default();
558            // SX1262 datasheet §8.2: max SCK = 16 MHz, Mode 0.
559            cfg.frequency = Frequency::M16;
560            let radio_bus = Spim::new(
561                p.TWISPI1, Irqs, p.P0_19, // SCK
562                p.P0_23, // MISO
563                p.P0_22, // MOSI
564                cfg,
565            );
566            let radio_cs = Output::new(p.P0_24, Level::High, OutputDrive::Standard);
567            let radio_spi = ExclusiveDevice::new(radio_bus, radio_cs, Delay).unwrap();
568
569            let radio_rst = Output::new(p.P0_25, Level::High, OutputDrive::Standard);
570            let radio_dio1 = Input::new(p.P0_20, Pull::None);
571            let radio_busy = Input::new(p.P0_17, Pull::None);
572
573            let iv = GenericSx126xInterfaceVariant::new(
574                radio_rst, radio_dio1, radio_busy,
575                None, // rf_switch_rx: DIO2 wired internally on the T-Echo module
576                None, // rf_switch_tx: same
577            )
578            .unwrap();
579
580            let lora_config = LoraConfig {
581                chip: Sx1262,
582                tcxo_ctrl: Some(TcxoCtrlVoltage::Ctrl1V8), // DIO3 → 1.8 V TCXO
583                use_dcdc: true,
584                rx_boost: true,
585            };
586
587            // enable_public_network=false → sync word 0x1424 (private).
588            // session_config().sync_word must match this choice.
589            let lora = LoRa::new(Sx126x::new(radio_spi, iv, lora_config), false, Delay)
590                .await
591                .unwrap_or_else(|_| panic!("radio init"));
592
593            spawner.spawn(radio_task(lora).unwrap());
594        }
595
596        // ── USB stack ────────────────────────────────────────────────────────
597        let led = Output::new(p.P0_14, Level::High, OutputDrive::Standard);
598        let driver = Driver::new(p.USBD, Irqs, HardwareVbusDetect::new(Irqs));
599
600        let mut config = Config::new(0x16c0, 0x27dd);
601        config.manufacturer = Some("UMSH");
602        config.product = Some("T-Echo UMSH NCP");
603        config.serial_number = Some("companion-ncp-techo");
604        config.max_power = 100;
605        config.max_packet_size_0 = 64;
606
607        static CONFIG_DESC: StaticCell<[u8; 256]> = StaticCell::new();
608        static BOS_DESC: StaticCell<[u8; 256]> = StaticCell::new();
609        static MSOS_DESC: StaticCell<[u8; 0]> = StaticCell::new();
610        static CONTROL_BUF: StaticCell<[u8; 64]> = StaticCell::new();
611        static STATE: StaticCell<State> = StaticCell::new();
612
613        let mut builder = Builder::new(
614            driver,
615            config,
616            CONFIG_DESC.init([0; 256]),
617            BOS_DESC.init([0; 256]),
618            MSOS_DESC.init([0; 0]),
619            CONTROL_BUF.init([0; 64]),
620        );
621
622        let class = CdcAcmClass::new(&mut builder, STATE.init(State::new()), 64);
623        let mut usb = builder.build();
624
625        let (tx, raw_rx, ctrl) = class.split_with_control();
626        let rx = CdcAcmRescue::new(raw_rx, ctrl);
627
628        spawner.spawn(output_task(tx).unwrap());
629        spawner.spawn(usb_in_task(rx).unwrap());
630        spawner.spawn(ncp_task(boot_reason).unwrap());
631
632        let button = Input::new(p.P1_10, Pull::Up);
633        spawner.spawn(button_task(button).unwrap());
634        spawner.spawn(shutdown_task(peripheral_power).unwrap());
635
636        join(usb.run(), heartbeat(led, wdt_handle)).await;
637    }

639    // ─── Heartbeat + WDT pet ─────────────────────────────────────────────────
640
641    async fn heartbeat(mut led: Output<'static>, mut wdt: WatchdogHandle) -> ! {
642        let mut engine = LedEngine::new(LedTimings::default(), Instant::now().as_millis());
643        loop {
644            wdt.pet();
645            let decision = engine.tick(Instant::now().as_millis());
646            // P0.14 is active-low: set_low() = LED on.
647            if decision.on {
648                led.set_low()
649            } else {
650                led.set_high()
651            }
652            Timer::at(Instant::from_millis(decision.next_deadline_ms)).await;
653        }
654    }
