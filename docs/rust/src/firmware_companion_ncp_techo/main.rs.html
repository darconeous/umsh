1// LilyGO T-Echo companion-radio NCP firmware.
2//
3// Exposes the SX1262 as a host-controlled PHY speaking the minimal
4// companion-radio protocol (docs/protocol/src/companion-radio-minimal.md)
5// over USB-CDC/HDLC-Lite and encrypted+bonded BLE GATT/SAR. The UMSH MAC does not run here:
6// the host owns it and drives this device through
7// `umsh::companion_radio::CompanionRadio`.
8//
9// Protocol behavior lives in `umsh-companion-ncp::Session` (host-tested,
10// no I/O); this binary is only glue:
11//
12// Task layout (steady state):
13//   - main():            initializes MPSL/SDC and joins BLE, USB, heartbeat
14//   - radio_task:        owns lora_phy::LoRa via umsh_radio_loraphy::ncp_runner;
15//                        modulation/frequency/power are pushed at runtime
16//                        through NCP_CTL as the host sets properties
17//   - usb_in_task:       owns CdcAcmRescue + HDLC decoder; forwards frames and
18//                        attach/detach edges into INPUT_CH (keeps
19//                        read_packet out of any select, so cancel safety
20//                        never depends on the USB driver)
21//   - ncp_task:          owns the framing-free Session; sorts INPUT_CH,
22//                        radio RX, and TX completions into session calls
23//   - output_task:       owns the USB Sender + HDLC encoder, drains OUT_USB_CH
24//   - ble_app:           advertising + encrypted/bond-gated GATT/SAR edges,
25//                        pairing policy, generation-tagged OUT_BLE_CH, and
26//                        MPSL-coordinated PIN/bond persistence
27//   - button_task:       resolves the side button into display-menu gestures
28//   - display_task:      owns the e-paper BLE menu
29//   - touch_task:        preserves the touch button's backlight control
30//   - shutdown_task:     tri-states peripheral pins, drops the rail,
31//                        enters System OFF
32//
33// CMD_RST is a protocol-level reset: session state returns to post-reset
34// defaults and the radio is re-applied (disabled), but the MCU and the
35// USB link stay up. Host attach also resets the session, silently — the
36// reset notice is only emitted in response to CMD_RST so the host never
37// sees an unsolicited reset it didn't ask for mid-handshake.
38//
39// Safety primitives inherited from the BSP (see umsh-bsp-nrf52840):
40//   * Panic capture into reserved RAM (reported as STATUS_RESET_CRASH).
41//   * 1200-baud touchless reset and Ctrl-C × 3 + "dfu" escape to
42//     bootloader (baked into CdcAcmRescue).
43//   * Watchdog.

283    impl BleStore {
284        fn mount(mut flash: nrf_mpsl::Flash<'static>) -> Self {
285            let mut latest: Option<(u32, Snapshot)> = None;
286            let mut read_failures = 0u8;
287            let mut valid_records = 0u8;
288            for page in [ble_store::PAGE0, ble_store::PAGE1] {
289                let mut address = page;
290                while address < page + ble_store::PAGE_SIZE {
291                    let mut bytes = [0u8; ble_store::SLOT_SIZE];
292                    if flash.read(address, &mut bytes).is_ok() {
293                        if Snapshot::decode(&bytes).is_some() {
294                            valid_records = valid_records.saturating_add(1);
295                        }
296                        latest = ble_store::consider_snapshot(latest, address, &bytes);
297                    } else {
298                        read_failures = read_failures.saturating_add(1);
299                    }
300                    address += ble_store::SLOT_SIZE as u32;
301                }
302            }
303            let (slot, snapshot) = latest
304                .map(|(slot, snapshot)| (Some(slot), snapshot))
305                .unwrap_or((None, Snapshot::empty()));
306            debug_log(format_args!(
307                "store mount valid-records={} read-failures={} selected-slot={:?} generation={} bonds={} pin={} local-irk={}",
308                valid_records,
309                read_failures,
310                slot,
311                snapshot.generation,
312                snapshot.bonds.len(),
313                snapshot.pin.is_some(),
314                snapshot.local_irk.is_some(),
315            ));
316            Self {
317                flash,
318                snapshot,
319                slot,
320            }
321        }
322
323        fn snapshot(&self) -> &Snapshot {
324            &self.snapshot
325        }
326
327        fn erased_slot(&mut self, start: u32, end: u32) -> Option<u32> {
328            let mut address = start;
329            while address < end {
330                let mut bytes = [0u8; ble_store::SLOT_SIZE];
331                match self.flash.read(address, &mut bytes) {
332                    Ok(()) if bytes.iter().all(|byte| *byte == 0xff) => return Some(address),
333                    Ok(()) => {}
334                    Err(_) => debug_log(format_args!(
335                        "store erased-slot read=FAILED address=0x{address:06x}"
336                    )),
337                }
338                address += ble_store::SLOT_SIZE as u32;
339            }
340            None
341        }
342
343        async fn persist(&mut self, mut snapshot: Snapshot) -> Result<(), ()> {
344            snapshot.generation = self.snapshot.generation.wrapping_add(1);
345            let target = if let Some(current) = self.slot {
346                let page = if current < ble_store::PAGE1 {
347                    ble_store::PAGE0
348                } else {
349                    ble_store::PAGE1
350                };
351                self.erased_slot(
352                    current + ble_store::SLOT_SIZE as u32,
353                    page + ble_store::PAGE_SIZE,
354                )
355            } else {
356                self.erased_slot(ble_store::PAGE0, ble_store::PAGE0 + ble_store::PAGE_SIZE)
357            };
358            let target = match target {
359                Some(target) => target,
360                None => {
361                    let page = if self.slot.is_some_and(|slot| slot < ble_store::PAGE1) {
362                        ble_store::PAGE1
363                    } else {
364                        ble_store::PAGE0
365                    };
366                    debug_log(format_args!("store erase begin page=0x{page:06x}"));
367                    if ble_store::erase_journal_page(&mut self.flash, page)
368                        .await
369                        .is_err()
370                    {
371                        debug_log(format_args!("store erase=FAILED page=0x{page:06x}"));
372                        return Err(());
373                    }
374                    debug_log(format_args!("store erase=ok page=0x{page:06x}"));
375                    page
376                }
377            };
378
379            let bytes = snapshot.encode();
380            debug_log(format_args!(
381                "store body-write begin generation={} target=0x{target:06x}",
382                snapshot.generation
383            ));
384            match ble_store::write_committed_record(&mut self.flash, target, &bytes).await {
385                Ok(()) => debug_log(format_args!(
386                    "store body-write=ok commit-write=ok target=0x{target:06x}"
387                )),
388                Err(ble_store::CommitError::Body(())) => {
389                    debug_log(format_args!(
390                        "store body-write=FAILED target=0x{target:06x}"
391                    ));
392                    return Err(());
393                }
394                Err(ble_store::CommitError::Commit(())) => {
395                    debug_log(format_args!(
396                        "store body-write=ok commit-write=FAILED target=0x{target:06x}"
397                    ));
398                    return Err(());
399                }
400            }
401            self.snapshot = snapshot;
402            self.slot = Some(target);
403            debug_log(format_args!(
404                "store commit generation={} slot=0x{:06x} bonds={} pin={} local_irk={}",
405                self.snapshot.generation,
406                target,
407                self.snapshot.bonds.len(),
408                self.snapshot.pin.is_some(),
409                self.snapshot.local_irk.is_some(),
410            ));
411            Ok(())
412        }
413
414        async fn set_pin(&mut self, pin: Option<u32>) -> Result<(), ()> {
415            let mut next = self.snapshot.clone();
416            next.pin = pin;
417            self.persist(next).await
418        }

420        async fn set_local_irk(&mut self, local_irk: [u8; 16]) -> Result<(), ()> {
421            if self.snapshot.local_irk == Some(local_irk) {
422                return Ok(());
423            }
424            let mut next = self.snapshot.clone();
425            next.local_irk = Some(local_irk);
426            self.persist(next).await
427        }

429        async fn add_bond(&mut self, bond: &BondInformation) -> Result<(), ()> {
430            let stored = stored_bond(bond);
431            let mut next = self.snapshot.clone();
432            if let Some(existing) = next.bonds.iter_mut().find(|existing| {
433                existing.address_kind == stored.address_kind && existing.address == stored.address
434            }) {
435                if *existing == stored {
436                    return Ok(());
437                }
438                *existing = stored;
439            } else {
440                next.bonds.push(stored).map_err(|_| ())?;
441            }
442            self.persist(next).await
443        }
444
445        async fn clear_security(&mut self) -> Result<(), ()> {
446            let mut next = Snapshot::empty();
447            next.generation = self.snapshot.generation;
448            next.local_irk = self.snapshot.local_irk;
449            self.persist(next).await
450        }
451    }

453    fn stored_bond(bond: &BondInformation) -> StoredBond {
454        let address = bond.identity.addr.to_bytes();
455        StoredBond {
456            address_kind: address[0],
457            address: address[1..].try_into().unwrap(),
458            irk: bond.identity.irk.map(IdentityResolvingKey::to_le_bytes),
459            ltk: bond.ltk.to_le_bytes(),
460            security_level: match bond.security_level {
461                SecurityLevel::NoEncryption => 0,
462                SecurityLevel::Encrypted => 1,
463                SecurityLevel::EncryptedAuthenticated => 2,
464            },
465            is_bonded: bond.is_bonded,
466        }
467    }

469    fn bond_identity_is_persistable(bond: &BondInformation) -> bool {
470        let address = bond.identity.addr.to_bytes();
471        let public = address[0] & 1 == 0;
472        let random_static = address[1] & 0xc0 == 0xc0;
473        public || random_static || bond.identity.irk.is_some()
474    }

476    fn trouble_bond(bond: &StoredBond) -> Option<BondInformation> {
477        let mut raw = bond.address;
478        raw.reverse();
479        let identity = Identity {
480            addr: Address::new(AddrKind::new(bond.address_kind), BdAddr::new(raw)),
481            irk: bond.irk.and_then(IdentityResolvingKey::from_le_bytes),
482        };
483        let security_level = match bond.security_level {
484            0 => SecurityLevel::NoEncryption,
485            1 => SecurityLevel::Encrypted,
486            2 => SecurityLevel::EncryptedAuthenticated,
487            _ => return None,
488        };
489        Some(BondInformation::new(
490            identity,
491            LongTermKey::from_le_bytes(bond.ltk),
492            security_level,
493            bond.is_bonded,
494        ))
495    }

616    /// Largest companion frame the session emits (CMD_STR_RECV around a
617    /// full-MTU payload).
618    const FRAME_OUT_MAX: usize = 300;
619    const WIRE_MAX: usize = hdlc::max_encoded_len(300);

621    /// Collects frames emitted synchronously by the session, then
622    /// flushes them to OUT_USB_CH asynchronously. The session emits at
623    /// most one frame per call; two slots give headroom.
624    struct Emitter {
625        bufs: [[u8; FRAME_OUT_MAX]; 2],
626        lens: [usize; 2],
627        count: usize,
628    }
629
630    impl Emitter {
631        const fn new() -> Self {
632            Self {
633                bufs: [[0; FRAME_OUT_MAX]; 2],
634                lens: [0; 2],
635                count: 0,
636            }
637        }
638
639        /// Copy one raw companion frame into the next slot.
640        ///
641        /// The session is expected to emit at most `bufs.len()` frames per call
642        /// and every frame is expected to fit `FRAME_OUT_MAX`. Both invariants are
643        /// asserted in debug builds so a future session change that violates
644        /// them is caught rather than silently dropping a response.
645        fn push(&mut self, frame: &[u8]) {
646            if self.count >= self.bufs.len() {
647                debug_assert!(
648                    false,
649                    "Emitter overflow: session emitted more frames per call than staging slots"
650                );
651                return;
652            }
653            if frame.len() <= FRAME_OUT_MAX {
654                self.bufs[self.count][..frame.len()].copy_from_slice(frame);
655                self.lens[self.count] = frame.len();
656                self.count += 1;
657            } else {
658                debug_assert!(false, "Emitter: companion frame exceeds FRAME_OUT_MAX");
659            }
660        }
661
662        /// Queue all staged frames for the active transport output task.
663        async fn flush(&mut self, destination: Option<(Transport, u32)>) {
664            if let Some((transport, generation)) = destination {
665                for index in 0..self.count {
666                    let mut frame: FrameBuf = heapless::Vec::new();
667                    if frame
668                        .extend_from_slice(&self.bufs[index][..self.lens[index]])
669                        .is_err()
670                    {
671                        debug_log(format_args!(
672                            "emitter frame copy=FAILED index={} len={}",
673                            index, self.lens[index]
674                        ));
675                        continue;
676                    }
677                    let outbound = OutFrame { generation, frame };
678                    match transport {
679                        Transport::Usb => OUT_USB_CH.send(outbound).await,
680                        Transport::Ble => OUT_BLE_CH.send(outbound).await,
681                    }
682                }
683            }
684            self.count = 0;
685        }
686    }

762    /// Execute a radio side effect requested by the session.
763    async fn apply_effect(session: &Session, effect: Option<Effect>) {
764        match effect {
765            Some(Effect::ApplyRadio(settings)) => {
766                // The session validates values against the same discrete
767                // sets these converters accept, so None here is
768                // unreachable; bail out defensively rather than panic.
769                let (Some(sf), Some(bw), Some(cr)) = (
770                    spreading_factor_from_u8(settings.sf),
771                    bandwidth_from_hz(settings.bw_hz),
772                    coding_rate_from_denom(settings.cr_denom),
773                ) else {
774                    return;
775                };
776                NCP_CTL.apply(NcpSettings {
777                    enabled: settings.enabled,
778                    freq_hz: settings.freq_khz.saturating_mul(1_000),
779                    sf,
780                    bw,
781                    cr,
782                    power_dbm: i32::from(settings.tx_power_dbm),
783                });
784            }
785            Some(Effect::StartTransmit) => {
786                let mut data: heapless::Vec<u8, MAX_PAYLOAD> = heapless::Vec::new();
787                if data.extend_from_slice(session.tx_data()).is_err() {
788                    debug_log(format_args!(
789                        "radio tx staging=FAILED len={}",
790                        session.tx_data().len()
791                    ));
792                    return;
793                }
794                let power_dbm = match session.tx_power() {
795                    TxPower::Default => None,
796                    TxPower::Max => Some(i32::from(MAX_TX_POWER_DBM)),
797                    TxPower::Dbm(dbm) => Some(i32::from(dbm)),
798                };
799                RADIO_CH.tx.send(TxRequest { data, power_dbm }).await;
800            }
801            // SampleRssi needs `&mut Session` + the emitter, so it is handled
802            // inline in ncp_task rather than here.
803            Some(Effect::SampleRssi { .. }) | Some(Effect::SetPairingPin { .. }) | None => {}
804        }
805    }

838    async fn pairing_config_task<C: Controller, P: PacketPool>(
839        stack: &Stack<'_, C, P>,
840        store: &BleStoreMutex,
841    ) -> ! {
842        loop {
843            match select3(
844                PAIRING_CONFIG_CH.receive(),
845                PAIRING_MODE_REQUEST.wait(),
846                BLE_WIPE_REQUEST.wait(),
847            )
848            .await
849            {
850                Either3::First(pin) => {
851                    debug_log(format_args!(
852                        "pin config begin configured={}",
853                        pin.is_some()
854                    ));
855                    let persisted = match store.lock().await.set_pin(pin).await {
856                        Ok(()) => {
857                            debug_log(format_args!("pin config persist=ok"));
858                            true
859                        }
860                        Err(()) => {
861                            debug_log(format_args!("pin config persist=FAILED"));
862                            false
863                        }
864                    };
865                    let applied = if persisted {
866                        match stack.set_fixed_passkey(pin) {
867                            Ok(()) => {
868                                debug_log(format_args!("pin config trouble-passkey=ok"));
869                                true
870                            }
871                            Err(error) => {
872                                debug_log(format_args!(
873                                    "pin config trouble-passkey=FAILED error={error:?}"
874                                ));
875                                false
876                            }
877                        }
878                    } else {
879                        debug_log(format_args!("pin config trouble-passkey=skipped"));
880                        false
881                    };
882                    let result = persisted && applied;
883                    if result {
884                        stack.set_io_capabilities(if pin.is_some() {
885                            IoCapabilities::DisplayOnly
886                        } else {
887                            IoCapabilities::NoInputNoOutput
888                        });
889                        PAIRING_PIN.store(pin.unwrap_or(u32::MAX), Ordering::Release);
890                        apply_pairing_gate(stack);
891                    }
892                    debug_log(format_args!(
893                        "pin config requested={} persisted={} applied={}",
894                        pin.is_some(),
895                        persisted,
896                        result,
897                    ));
898                    PAIRING_CONFIG_ACK.signal(result);
899                }
900                Either3::Second(()) => {
901                    debug_log(format_args!("pairing mode requested"));
902                    PAIRING_MODE.store(true, Ordering::Release);
903                    BLE_LED_MODE.store(1, Ordering::Release);
904                    let unavailable = PAIRING_LOCKED_OUT.load(Ordering::Acquire)
905                        || usize::from(BLE_BOND_COUNT.load(Ordering::Acquire))
906                            >= ble_store::MAX_BONDS;
907                    UI_NOTICE.signal(if unavailable {
908                        UiNotice::PairingUnavailable
909                    } else {
910                        UiNotice::PairingStarted
911                    });
912                    PAIRING_TIMER_RESET.signal(());
913                    apply_pairing_gate(stack);
914                }
915                Either3::Third(()) => {
916                    debug_log(format_args!("security wipe requested"));
917                    if store.lock().await.clear_security().await.is_ok() {
918                        debug_log(format_args!("security wipe flash=ok"));
919                        BLE_BOND_COUNT.store(0, Ordering::Release);
920                        let mut identities: heapless::Vec<Identity, { ble_store::MAX_BONDS }> =
921                            heapless::Vec::new();
922                        stack.with_bond_information(|bonds| {
923                            for bond in bonds {
924                                if identities.push(bond.identity).is_err() {
925                                    debug_log(format_args!("security wipe identity-list=FULL"));
926                                }
927                            }
928                        });
929                        debug_log(format_args!(
930                            "security wipe removing-bonds count={}",
931                            identities.len()
932                        ));
933                        for identity in identities {
934                            match stack.remove_bond_information(identity) {
935                                Ok(()) => debug_log(format_args!("security wipe remove-bond=ok")),
936                                Err(error) => debug_log(format_args!(
937                                    "security wipe remove-bond=FAILED error={error:?}"
938                                )),
939                            }
940                        }
941                        match stack.set_fixed_passkey(None) {
942                            Ok(()) => debug_log(format_args!("security wipe clear-passkey=ok")),
943                            Err(error) => debug_log(format_args!(
944                                "security wipe clear-passkey=FAILED error={error:?}"
945                            )),
946                        }
947                        stack.set_io_capabilities(IoCapabilities::NoInputNoOutput);
948                        PAIRING_PIN.store(u32::MAX, Ordering::Release);
949                        PAIRING_FAILURES.store(0, Ordering::Release);
950                        PAIRING_LOCKED_OUT.store(false, Ordering::Release);
951                        PAIRING_MODE.store(true, Ordering::Release);
952                        BLE_LED_MODE.store(2, Ordering::Release);
953                        PAIRING_TIMER_RESET.signal(());
954                        apply_pairing_gate(stack);
955                        debug_log(format_args!("security wipe complete"));
956                        UI_NOTICE.signal(UiNotice::BondsCleared);
957                    } else {
958                        debug_log(format_args!("security wipe flash=FAILED"));
959                        UI_NOTICE.signal(UiNotice::ClearFailed);
960                    }
961                }
962            }
963        }
964    }

1018    async fn send_ble_frame(
1019        server: &CompanionServer<'_>,
1020        conn: &GattConnection<'_, '_, DefaultPacketPool>,
1021        outbound: OutFrame,
1022    ) -> Result<(), trouble_host::Error> {
1023        if SESSION_GEN.load(Ordering::Acquire) != outbound.generation {
1024            debug_log(format_args!(
1025                "ble outbound dropped stale-generation frame-gen={} active-gen={}",
1026                outbound.generation,
1027                SESSION_GEN.load(Ordering::Acquire),
1028            ));
1029            return Ok(());
1030        }
1031        let segment_payload = usize::from(conn.raw().att_mtu())
1032            .saturating_sub(4)
1033            .clamp(1, BLE_VALUE_MAX - 1);
1034        let mut segments = generation_checked(
1035            gatt::segments(&outbound.frame, segment_payload),
1036            outbound.generation,
1037            || SESSION_GEN.load(Ordering::Acquire),
1038        );
1039        for segment in segments.by_ref() {
1040            let mut value: heapless09::Vec<u8, BLE_VALUE_MAX> = heapless09::Vec::new();
1041            value
1042                .push(segment.header())
1043                .map_err(|_| trouble_host::Error::InsufficientSpace)?;
1044            value
1045                .extend_from_slice(segment.payload())
1046                .map_err(|_| trouble_host::Error::InsufficientSpace)?;
1047            server
1048                .companion
1049                .frame_out
1050                .notify(conn, &value, false)
1051                .await?;
1052        }
1053        if segments.stale() {
1054            debug_log(format_args!(
1055                "ble outbound segmentation stopped generation-changed"
1056            ));
1057        }
1058        Ok(())
1059    }

1061    async fn gatt_connection<C: Controller, P: PacketPool>(
1062        stack: &Stack<'_, C, P>,
1063        store: &BleStoreMutex,
1064        server: &CompanionServer<'_>,
1065        conn: &GattConnection<'_, '_, DefaultPacketPool>,
1066    ) -> Result<(), trouble_host::Error> {
1067        match conn.raw().set_bondable(true) {
1068            Ok(()) => debug_log(format_args!("connection set-bondable=ok")),
1069            Err(error) => {
1070                debug_log(format_args!(
1071                    "connection set-bondable=FAILED error={error:?}"
1072                ));
1073                return Err(error);
1074            }
1075        }
1076        let peer = conn.raw().peer_identity();
1077        debug_log(format_args!(
1078            "connected peer={} kind={} irk={} table_match={} level={:?} mtu={}",
1079            peer.addr,
1080            peer.addr.to_bytes()[0],
1081            peer.irk.is_some(),
1082            conn.raw().is_bonded_peer(),
1083            conn.raw().security_level(),
1084            conn.raw().att_mtu(),
1085        ));
1086        let mut attached = false;
1087        let mut reassembler: gatt::Reassembler<{ gatt::MAX_FRAME }> = gatt::Reassembler::new();
1088
1089        loop {
1090            match select3(conn.next(), OUT_BLE_CH.receive(), ADV_POLICY_CHANGED.wait()).await {
1091                Either3::First(GattConnectionEvent::Disconnected { reason }) => {
1092                    debug_log(format_args!("disconnected reason={reason:?}"));
1093                    break;
1094                }
1095                Either3::First(GattConnectionEvent::PairingComplete { bond, .. }) => {
1096                    debug_log(format_args!(
1097                        "pairing-complete bond={} table_match={}",
1098                        bond.is_some(),
1099                        conn.raw().is_bonded_peer(),
1100                    ));
1101                    if let Some(bond) = bond {
1102                        if !bond_identity_is_persistable(&bond) {
1103                            debug_log(format_args!("pairing bond identity=incomplete"));
1104                            match stack.remove_bond_information(bond.identity) {
1105                                Ok(()) => {
1106                                    debug_log(format_args!("pairing incomplete bond remove=ok"))
1107                                }
1108                                Err(error) => debug_log(format_args!(
1109                                    "pairing incomplete bond remove=FAILED error={error:?}"
1110                                )),
1111                            }
1112                            debug_log(format_args!(
1113                                "disconnect initiated by incomplete pairing identity"
1114                            ));
1115                            conn.raw().disconnect();
1116                            break;
1117                        }
1118                        let persisted_bonds = match persist_bond(store, &bond).await {
1119                            Ok(count) => count,
1120                            Err(()) => {
1121                                debug_log(format_args!("pairing bond persist=FAILED"));
1122                                match stack.remove_bond_information(bond.identity) {
1123                                    Ok(()) => debug_log(format_args!(
1124                                        "pairing unpersisted bond remove=ok"
1125                                    )),
1126                                    Err(error) => debug_log(format_args!(
1127                                        "pairing unpersisted bond remove=FAILED error={error:?}"
1128                                    )),
1129                                }
1130                                debug_log(format_args!(
1131                                    "disconnect initiated by pairing persistence failure"
1132                                ));
1133                                conn.raw().disconnect();
1134                                break;
1135                            }
1136                        };
1137                        BLE_BOND_COUNT.store(persisted_bonds as u8, Ordering::Release);
1138                        UI_REFRESH.signal(());
1139                        debug_log(format_args!(
1140                            "pairing bond persist=ok peer={} kind={} irk={} bonded={} level={:?}",
1141                            bond.identity.addr,
1142                            bond.identity.addr.to_bytes()[0],

1204                            read.handle(),

1206                        )),

1218                            "gatt other handle={:?} level={:?}",
1219                            event.payload().handle(),
1220                            conn.raw().security_level(),
1221                        )),

1224                    let cccd = matches!(&event, GattEvent::Write(write) if Some(write.handle()) == server.companion.frame_out.cccd_handle);

1226                    let bonded = conn.raw().is_bonded_peer();

1229                    // every peripheral path.  The protected GATT edge is authoritative: if
1230                    // Trouble says this peer is bonded, find that exact live-table entry and
1231                    // make it durable before granting access.  add_bond is idempotent, so
1232                    // subsequent frames do not write flash.

1234                        let peer = conn.raw().peer_identity();
1235                        let bond = stack.with_bond_information(|bonds| {
1236                            bonds
1237                                .iter()
1238                                .find(|bond| bond.identity.match_identity(&peer))
1239                                .cloned()
1240                        });

1242                            Some(bond) if !bond_identity_is_persistable(&bond) => {
1243                                debug_log(format_args!("protected bond identity=pending"));
1244                                false
1245                            }
1246                            Some(bond) => match persist_bond(store, &bond).await {
1247                                Ok(count) => {
1248                                    debug_log(format_args!("protected bond persist=ok"));
1249                                    BLE_BOND_COUNT.store(count as u8, Ordering::Release);
1250                                    UI_REFRESH.signal(());
1251                                    apply_pairing_gate(stack);
1252                                    true
1253                                }
1254                                Err(()) => {
1255                                    debug_log(format_args!("protected bond persist=FAILED"));
1256                                    bond_persist_failed = true;
1257                                    match stack.remove_bond_information(bond.identity) {
1258                                        Ok(()) => debug_log(format_args!(
1259                                            "protected unpersisted bond remove=ok"
1260                                        )),
1261                                        Err(error) => debug_log(format_args!(
1262                                            "protected unpersisted bond remove=FAILED error={error:?}"
1263                                        )),
1264                                    }
1265                                    false
1266                                }
1267                            },
1268                            None => {
1269                                debug_log(format_args!("protected bond lookup=missing"));
1270                                false
1271                            }
1272                        }
1273                    } else {
1274                        !protected
1275                    };
1276                    if protected {
1277                        let peer = conn.raw().peer_identity();
1278                        debug_log(format_args!(
1279                            "gatt protected={} bonded={} durable={} peer={} kind={} irk={} level={:?}",
1280                            if cccd { "cccd" } else { "frame-in" },
1281                            bonded,
1282                            durable_bond,
1283                            peer.addr,
1284                            peer.addr.to_bytes()[0],
1285                            peer.irk.is_some(),
1286                            conn.raw().security_level(),
1287                        ));
1288                    }
1289                    let mut inbound: heapless09::Vec<u8, BLE_VALUE_MAX> = heapless09::Vec::new();

1291                        if let GattEvent::Write(write) = &event {
1292                            write.with_data(|_, data| {
1293                                if inbound.extend_from_slice(data).is_err() {
1294                                    debug_log(format_args!(
1295                                        "gatt frame-in staging=FAILED len={}",
1296                                        data.len()
1297                                    ));
1298                                }
1299                            });
1300                        }
1301                    }
1302
1303                    let server_permission_denied = matches!(&event, GattEvent::NotAllowed(_));
1304                    let reply = if protected && !(bonded && durable_bond) {
1305                        debug_log(format_args!(
1306                            "gatt decision=reject insufficient-authentication"
1307                        ));
1308                        event.reject(AttErrorCode::INSUFFICIENT_AUTHENTICATION)
1309                    } else if server_permission_denied {
1310                        // `NotAllowedEvent::accept()` preserves and returns the
1311                        // attribute server's permission error; it does not grant
1312                        // the operation. Make that non-obvious Trouble API
1313                        // behavior explicit in the hardware trace.
1314                        debug_log(format_args!("gatt decision=return-server-permission-error"));

1317                        debug_log(format_args!("gatt decision=accept"));
1318                        event.accept()
1319                    }?;

1323                        debug_log(format_args!(
1324                            "disconnect initiated by protected bond persistence failure"
1325                        ));
1326                        conn.raw().disconnect();
1327                        break;
1328                    }
1329
1330                    if frame_in && bonded {
1331                        match reassembler.push(&inbound) {
1332                            Some(Ok(frame)) => {
1333                                let mut value: FrameBuf = heapless::Vec::new();
1334                                match value.extend_from_slice(frame) {
1335                                    Ok(()) => {
1336                                        debug_log(format_args!(
1337                                            "gatt frame-in complete len={}",
1338                                            frame.len()
1339                                        ));
1340                                        INPUT_CH.send(InEvent::Frame(Transport::Ble, value)).await;
1341                                    }
1342                                    Err(()) => debug_log(format_args!(
1343                                        "gatt frame-in complete staging=FAILED len={}",

1345                                    )),
1346                                }
1347                            }
1348                            Some(Err(error)) => debug_log(format_args!(
1349                                "gatt frame-in decode=FAILED error={error:?} segment-len={}",
1350                                inbound.len()
1351                            )),
1352                            None => debug_log(format_args!(
1353                                "gatt frame-in segment accepted segment-len={} complete=false",
1354                                inbound.len()
1355                            )),
1356                        }
1357                    }
1358                    if cccd && bonded {
1359                        let subscribed = server.companion.frame_out.should_notify(conn);
1360                        match (attached, subscribed) {
1361                            (false, true) => {
1362                                debug_log(format_args!("cccd subscribed=true"));

1372                            (false, false) => debug_log(format_args!(
1373                                "cccd state unchanged attached=false subscribed=false"
1374                            )),
1375                            (true, true) => debug_log(format_args!(
1376                                "cccd state unchanged attached=true subscribed=true"
1377                            )),
1378                        }
1379                    }
1380                }
1381                Either3::First(GattConnectionEvent::PhyUpdated { tx_phy, rx_phy }) => {
1382                    debug_log(format_args!(
1383                        "connection phy-updated tx={tx_phy:?} rx={rx_phy:?}"
1384                    ));
1385                }
1386                Either3::First(GattConnectionEvent::ConnectionParamsUpdated {
1387                    conn_interval,
1388                    peripheral_latency,
1389                    supervision_timeout,
1390                }) => {
1391                    debug_log(format_args!(
1392                        "connection params-updated interval-us={} latency={} timeout-us={}",
1393                        conn_interval.as_micros(),
1394                        peripheral_latency,
1395                        supervision_timeout.as_micros(),
1396                    ));
1397                }
1398                Either3::First(GattConnectionEvent::RequestConnectionParams(request)) => {
1399                    debug_log(format_args!(
1400                        "connection params-requested params={:?}",
1401                        request.params()
1402                    ));
1403                    match request.accept(None, stack).await {
1404                        Ok(()) => debug_log(format_args!("connection params-response=accepted")),
1405                        Err(error) => debug_log(format_args!(
1406                            "connection params-response=FAILED error={error:?}"
1407                        )),
1408                    }
1409                }
1410                Either3::First(GattConnectionEvent::DataLengthUpdated {
1411                    max_tx_octets,
1412                    max_tx_time,
1413                    max_rx_octets,
1414                    max_rx_time,
1415                }) => debug_log(format_args!(
1416                    "connection data-length tx-octets={} tx-time={} rx-octets={} rx-time={}",
1417                    max_tx_octets, max_tx_time, max_rx_octets, max_rx_time,
1418                )),

1420                    frame_space,
1421                    initiator,
1422                    phys,
1423                    spacing_types,
1424                }) => debug_log(format_args!(
1425                    "connection frame-space us={} initiator={initiator:?} phys={phys:?} spacing={spacing_types:?}",

1427                )),
1428                Either3::First(GattConnectionEvent::ConnectionRateChanged {
1429                    conn_interval,
1430                    subrate_factor,
1431                    peripheral_latency,
1432                    continuation_number,
1433                    supervision_timeout,
1434                }) => debug_log(format_args!(
1435                    "connection rate-changed interval-us={} subrate={} latency={} continuation={} timeout-us={}",
1436                    conn_interval.as_micros(),
1437                    subrate_factor,
1438                    peripheral_latency,
1439                    continuation_number,
1440                    supervision_timeout.as_micros(),
1441                )),
1442                Either3::Second(outbound) => {
1443                    if attached && conn.raw().is_bonded_peer() {
1444                        send_ble_frame(server, conn, outbound).await?;
1445                    } else {
1446                        debug_log(format_args!(
1447                            "ble outbound dropped attached={} bonded={} level={:?}",
1448                            attached,
1449                            conn.raw().is_bonded_peer(),

1451                        ));
1452                    }
1453                }
1454                Either3::Third(()) => {

1457                            "disconnect initiated by transport arbitration"
1458                        ));
1459                        conn.raw().disconnect();
1460                        break;
1461                    }
1462                }
1463            }
1464        }
1465        if attached {
1466            INPUT_CH.send(InEvent::Detached(Transport::Ble)).await;
1467        }
1468        Ok(())
1469    }

1647    /// into INPUT_CH; `wait_connection` always precedes the read loop so

1653            debug_log(format_args!("usb debug attached"));
1654            let mut decoder: hdlc::Decoder<FRAME_IN_MAX> = hdlc::Decoder::new();
1655            let mut local_generation = SESSION_GEN.load(Ordering::Acquire);

1672                            let _ = frame.extend_from_slice(bytes);

1675                    }
1676                }
1677            }

1682    /// Owns the framing-free protocol session. Sorts host frames,
1683    /// radio receptions, and transmit completions into session calls and
1684    /// executes the resulting radio effects.
1685    #[embassy_executor::task]
1686    async fn ncp_task(boot_reason: Status) {
1687        let mut session = Session::new(session_config());
1688        let mut emitter = Emitter::new();
1689        let mut arbitration = SessionArbitration::new(SESSION_GEN.load(Ordering::Acquire));
1690
1691        loop {
1692            // Only wait for a TX completion while one is outstanding,
1693            // so a spurious tx_done can never be consumed early.
1694            let tx_done = async {
1695                if session.has_pending_tx() {
1696                    RADIO_CH.tx_done.wait().await
1697                } else {
1698                    core::future::pending().await
1699                }
1700            };
1701
1702            match select3(INPUT_CH.receive(), RADIO_CH.rx.receive(), tx_done).await {
1703                Either3::First(InEvent::Attached(transport)) => {
1704                    // Fresh protocol state for the new host session.
1705                    // Silent: the reset notice is only sent for CMD_RST,
1706                    // so the host never races a stray notice during its
1707                    // own reset handshake.
1708                    arbitration.attach(transport);
1709                    SESSION_GEN.store(arbitration.generation(), Ordering::Release);
1710                    #[cfg(not(feature = "ble-debug"))]
1711                    set_advertising_allowed(arbitration.advertising_allowed());
1712                    #[cfg(feature = "ble-debug")]
1713                    {
1714                        let _ = arbitration.advertising_allowed();
1715                        set_advertising_allowed(true);
1716                    }
1717                    let effect = session.reset(boot_reason, &mut |_frame: &[u8]| {});
1718                    apply_effect(&session, Some(effect)).await;
1719                }
1720                Either3::First(InEvent::Detached(transport)) => {
1721                    if arbitration.detach(transport) {
1722                        set_advertising_allowed(true);
1723                    }
1724                }
1725                Either3::First(InEvent::Frame(transport, frame_bytes)) => {
1726                    if arbitration.accepts_frame(transport) {
1727                        let now_ms = Instant::now().as_millis();
1728                        let effect =
1729                            session.handle_frame(&frame_bytes, now_ms, &mut |frame: &[u8]| {
1730                                emitter.push(frame)
1731                            });
1732                        emitter.flush(arbitration.destination()).await;
1733                        match effect {
1734                            Some(Effect::SampleRssi { tid }) => {
1735                                // Round-trip to the radio runner for an
1736                                // instantaneous RSSI sample, then answer the
1737                                // deferred PROP_PHY_RSSI get.
1738                                NCP_CTL.request_rssi();
1739                                let sample = NCP_CTL.wait_rssi().await;
1740                                session.respond_rssi(tid, sample, &mut |frame: &[u8]| {
1741                                    emitter.push(frame)
1742                                });
1743                                emitter.flush(arbitration.destination()).await;
1744                            }
1745                            Some(Effect::SetPairingPin { tid, pin }) => {
1746                                PAIRING_CONFIG_CH.send(pin).await;
1747                                let applied = PAIRING_CONFIG_ACK.wait().await;
1748                                session.respond_pin_set(
1749                                    tid,
1750                                    applied.then_some(()).ok_or(()),
1751                                    &mut |frame: &[u8]| emitter.push(frame),
1752                                );
1753                                emitter.flush(arbitration.destination()).await;
1754                            }
1755                            other => apply_effect(&session, other).await,
1756                        }
1757                    }
1758                }
1759                Either3::Second(RxFrame { data, info }) => {
1760                    session.on_radio_rx(
1761                        &data,
1762                        info.rssi,
1763                        info.snr.as_centibels(),
1764                        info.lqi,
1765                        &mut |frame: &[u8]| emitter.push(frame),
1766                    );
1767                    emitter.flush(arbitration.destination()).await;
1768                }
1769                Either3::Third(result) => {
1770                    let now_ms = Instant::now().as_millis();
1771                    session.on_tx_result(result.is_ok(), now_ms, &mut |frame: &[u8]| {
1772                        emitter.push(frame)
1773                    });
1774                    emitter.flush(arbitration.destination()).await;
1775                }
1776            }
1777        }
1778    }

1972    /// Resolves the P1.10 side button (active-low, pull-up) through the same
1973    /// tested state machine used by T-1000E. Single advances, double selects,
1974    /// a 1–4 second hold released by the user goes back, and a continuing
1975    /// four-second hold always powers off.
1976    #[embassy_executor::task]
1977    async fn button_task(mut button: Input<'static>) {
1978        const DEBOUNCE: Duration = Duration::from_millis(10);
1979        let mut fsm = ButtonFsm::new(techo_button_timings());
1980        let mut pressed = button.is_low();
1981        loop {
1982            let event = {
1983                let now_ms = Instant::now().as_millis();
1984                let edge_fut = async {
1985                    if pressed {
1986                        button.wait_for_high().await;
1987                        Timer::after(DEBOUNCE).await;
1988                        ButtonEdge::Release
1989                    } else {
1990                        button.wait_for_low().await;
1991                        Timer::after(DEBOUNCE).await;
1992                        ButtonEdge::Press
1993                    }
1994                };
1995                let deadline = fsm.next_deadline().unwrap_or(now_ms.saturating_add(60_000));
1996                match select(edge_fut, Timer::at(Instant::from_millis(deadline))).await {
1997                    Either::First(edge) => {
1998                        pressed = matches!(edge, ButtonEdge::Press);
1999                        fsm.on_edge(edge, Instant::now().as_millis())
2000                    }
2001                    Either::Second(()) => fsm.poll(Instant::now().as_millis()),
2002                }
2003            };
2004
2005            let input = match event {
2006                Some(ButtonEvent::Single) => Some(UiInput::Forward),
2007                Some(ButtonEvent::Double) => Some(UiInput::Select),
2008                Some(ButtonEvent::Long) => Some(UiInput::Backward),
2009                Some(ButtonEvent::VeryLong) => {
2010                    pressed = false;
2011                    fsm = ButtonFsm::new(techo_button_timings());
2012                    SHUTDOWN_SIGNAL.signal(());
2013                    None
2014                }
2015                Some(ButtonEvent::Triple | ButtonEvent::Quad) | None => None,
2016            };
2017            if let Some(input) = input {
2018                UI_INPUT_CH.send(input).await;
2019            }
2020        }
2021    }

2044    /// Controlled power-off: put the e-paper controller to sleep, tri-state
2045    /// peripheral signal pins, drop the rail, and enter System OFF.
2046    #[embassy_executor::task]
2047    async fn shutdown_task(peripheral_power: Output<'static>) -> ! {
2048        SHUTDOWN_SIGNAL.wait().await;
2049
2050        DISPLAY_SHUTDOWN.signal(());
2051        let _ = select(
2052            DISPLAY_SHUTDOWN_DONE.wait(),
2053            Timer::after(Duration::from_secs(5)),
2054        )
2055        .await;
2056
2057        // E-paper SPI bus (SPIM2): SCK=P0.31, MISO=P1.07, MOSI=P0.29
2058        // E-paper control:         CS=P0.30, DC=P0.28, RST=P0.02, BUSY=P0.03
2059        // Radio SPI bus (TWISPI1): SCK=P0.19, MOSI=P0.22, MISO=P0.23
2060        // Radio control:           CS=P0.24, RST=P0.25, BUSY=P0.17, DIO1=P0.20
2061        // The display and touch tasks still own these pins; direct PIN_CNF
2062        // writes are deliberate here because every task is about to lose power.
2063        for (port, pin) in [
2064            (Port::P0, 31u8),
2065            (Port::P1, 7u8),
2066            (Port::P0, 29u8),
2067            (Port::P0, 30u8),
2068            (Port::P0, 28u8),
2069            (Port::P0, 2u8),
2070            (Port::P0, 3u8),
2071            (Port::P0, 11u8), // touch input ← async wait may have set SENSE
2072            (Port::P1, 11u8), // e-paper backlight
2073            (Port::P0, 19u8),
2074            (Port::P0, 22u8),
2075            (Port::P0, 23u8),
2076            (Port::P0, 24u8),
2077            (Port::P0, 25u8),
2078            (Port::P0, 17u8),
2079            (Port::P0, 20u8), // radio DIO1 ← has SENSE set by async radio wait
2080        ] {
2081            tristate_pin(port, pin);
2082        }
2083
2084        drop(peripheral_power);
2085
2086        // P1.10 is the side user button. Active-low, pull-up → DETECT-low wakes.
2087        power_off(&[WakePin {
2088            port: Port::P1,
2089            pin: 10,
2090            sense: WakeSense::Low,
2091        }])
2092    }

2343    async fn heartbeat(mut led: Output<'static>, mut wdt: WatchdogHandle) -> ! {
2344        let mut engine = LedEngine::new(LedTimings::default(), Instant::now().as_millis());
2345        loop {
2346            wdt.pet();
2347            let ble_mode = BLE_LED_MODE.load(Ordering::Acquire);
2348            if ble_mode != 0 {
2349                let phase = Instant::now().as_millis() % 2_000;
2350                let on = if ble_mode == 1 {
2351                    phase < 100 || (500..600).contains(&phase)
2352                } else {
2353                    phase < 100 || (200..300).contains(&phase) || (400..500).contains(&phase)
2354                };
2355                if on {
2356                    led.set_low();
2357                } else {
2358                    led.set_high();
2359                }
2360                Timer::after_millis(50).await;
2361                continue;
2362            }
2363            let decision = engine.tick(Instant::now().as_millis());
2364            // P0.14 is active-low: set_low() = LED on.
2365            if decision.on {
2366                led.set_low()
2367            } else {
2368                led.set_high()
2369            }
2370            Timer::at(Instant::from_millis(decision.next_deadline_ms)).await;
2371        }
2372    }
