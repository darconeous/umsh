1// LilyGO T-Echo companion-radio NCP firmware.
2//
3// Exposes the SX1262 as a host-controlled PHY speaking the minimal
4// companion-radio protocol (docs/protocol/src/companion-radio-minimal.md)
5// with HDLC-Lite framing over USB-CDC. The UMSH MAC does not run here:
6// the host owns it and drives this device through
7// `umsh::companion_radio::CompanionRadio`.
8//
9// Protocol behavior lives in `umsh-companion-ncp::Session` (host-tested,
10// no I/O); this binary is only glue:
11//
12// Task layout (steady state):
13//   - main():            joins usb.run / heartbeat (LED + WDT pet)
14//   - radio_task:        owns lora_phy::LoRa via umsh_radio_loraphy::ncp_runner;
15//                        modulation/frequency/power are pushed at runtime
16//                        through NCP_CTL as the host sets properties
17//   - usb_in_task:       owns CdcAcmRescue; forwards packets and
18//                        connect/disconnect edges into INPUT_CH (keeps
19//                        read_packet out of any select, so cancel safety
20//                        never depends on the USB driver)
21//   - ncp_task:          owns the Session + HDLC decoder; sorts INPUT_CH,
22//                        radio RX, and TX completions into session calls
23//   - output_task:       owns the USB Sender, drains OUTPUT_CH
24//   - button_task:       2 s hold on the user button fires SHUTDOWN_SIGNAL
25//   - shutdown_task:     tri-states peripheral pins, drops the rail,
26//                        enters System OFF
27//
28// CMD_RST is a protocol-level reset: session state returns to post-reset
29// defaults and the radio is re-applied (disabled), but the MCU and the
30// USB link stay up. Host attach also resets the session, silently — the
31// reset notice is only emitted in response to CMD_RST so the host never
32// sees an unsolicited reset it didn't ask for mid-handshake.
33//
34// Safety primitives inherited from the BSP (see umsh-bsp-nrf52840):
35//   * Panic capture into reserved RAM (reported as STATUS_RESET_CRASH).
36//   * 1200-baud touchless reset and Ctrl-C × 3 + "dfu" escape to
37//     bootloader (baked into CdcAcmRescue).
38//   * Watchdog.
39

178    // ─── Outgoing frame staging ──────────────────────────────────────────────

180    /// Largest companion frame the session emits (CMD_STR_RECV around a
181    /// full-MTU payload), with HDLC escaping headroom.
182    const WIRE_MAX: usize = hdlc::max_encoded_len(300);
183
184    /// Collects frames emitted synchronously by the session, then
185    /// flushes them to OUTPUT_CH asynchronously. The session emits at
186    /// most one frame per call; two slots give headroom.
187    struct Emitter {
188        bufs: [[u8; WIRE_MAX]; 2],
189        lens: [usize; 2],
190        count: usize,
191    }
192
193    impl Emitter {
194        const fn new() -> Self {
195            Self {
196                bufs: [[0; WIRE_MAX]; 2],
197                lens: [0; 2],
198                count: 0,
199            }
200        }
201
202        /// HDLC-encode one companion frame into the next slot.
203        fn push(&mut self, frame: &[u8]) {
204            if self.count < self.bufs.len()
205                && let Ok(len) = hdlc::encode_frame(frame, &mut self.bufs[self.count])
206            {
207                self.lens[self.count] = len;
208                self.count += 1;
209            }
210        }
211
212        /// Queue all staged frames for the USB output task.
213        async fn flush(&mut self) {
214            for index in 0..self.count {
215                for chunk in self.bufs[index][..self.lens[index]].chunks(64) {
216                    let mut packet: OutChunk = heapless::Vec::new();
217                    let _ = packet.extend_from_slice(chunk);
218                    OUTPUT_CH.send(packet).await;
219                }
220            }
221            self.count = 0;
222        }
223    }

225    /// Execute a radio side effect requested by the session.
226    async fn apply_effect(session: &Session, effect: Option<Effect>) {
227        match effect {
228            Some(Effect::ApplyRadio(settings)) => {
229                // The session validates values against the same discrete
230                // sets these converters accept, so None here is
231                // unreachable; bail out defensively rather than panic.
232                let (Some(sf), Some(bw), Some(cr)) = (
233                    spreading_factor_from_u8(settings.sf),
234                    bandwidth_from_hz(settings.bw_hz),
235                    coding_rate_from_denom(settings.cr_denom),
236                ) else {
237                    return;
238                };
239                NCP_CTL.apply(NcpSettings {
240                    enabled: settings.enabled,
241                    freq_hz: settings.freq_khz.saturating_mul(1_000),
242                    sf,
243                    bw,
244                    cr,
245                    power_dbm: i32::from(settings.tx_power_dbm),
246                });
247            }
248            Some(Effect::StartTransmit) => {
249                let mut data: heapless::Vec<u8, MAX_PAYLOAD> = heapless::Vec::new();
250                let _ = data.extend_from_slice(session.tx_data());
251                let power_dbm = match session.tx_power() {
252                    TxPower::Default => None,
253                    TxPower::Max => Some(i32::from(MAX_TX_POWER_DBM)),
254                    TxPower::Dbm(dbm) => Some(i32::from(dbm)),
255                };
256                RADIO_CH.tx.send(TxRequest { data, power_dbm }).await;
257            }
258            None => {}
259        }
260    }

280    /// Owns the CDC receive half. Forwards packets and connection edges
281    /// into INPUT_CH; `wait_connection` always precedes the read loop so
282    /// a disconnected port never busy-loops.
283    #[embassy_executor::task]
284    async fn usb_in_task(mut rx: TechoRescue) {
285        loop {
286            rx.wait_connection().await;
287            INPUT_CH.send(InEvent::Connected).await;
288            loop {
289                let mut packet = [0u8; 64];
290                match rx.read_packet(&mut packet).await {
291                    Ok(0) | Err(_) => break,
292                    Ok(len) => {
293                        let mut data: heapless::Vec<u8, 64> = heapless::Vec::new();
294                        let _ = data.extend_from_slice(&packet[..len]);
295                        INPUT_CH.send(InEvent::Data(data)).await;
296                    }
297                }
298            }
299            INPUT_CH.send(InEvent::Disconnected).await;
300        }
301    }

303    /// Owns the protocol session and the HDLC decoder. Sorts host bytes,
304    /// radio receptions, and transmit completions into session calls and
305    /// executes the resulting radio effects.
306    #[embassy_executor::task]
307    async fn ncp_task(boot_reason: Status) {
308        let mut session = Session::new(session_config());
309        let mut decoder: hdlc::Decoder<300> = hdlc::Decoder::new();
310        let mut emitter = Emitter::new();
311
312        loop {
313            // Only wait for a TX completion while one is outstanding,
314            // so a spurious tx_done can never be consumed early.
315            let tx_done = async {
316                if session.has_pending_tx() {
317                    RADIO_CH.tx_done.wait().await
318                } else {
319                    core::future::pending().await
320                }
321            };
322
323            match select3(INPUT_CH.receive(), RADIO_CH.rx.receive(), tx_done).await {
324                Either3::First(InEvent::Connected) => {
325                    // Fresh protocol state for the new host session.
326                    // Silent: the reset notice is only sent for CMD_RST,
327                    // so the host never races a stray notice during its
328                    // own reset handshake.
329                    decoder.reset();
330                    let effect = session.reset(boot_reason, &mut |_frame: &[u8]| {});
331                    apply_effect(&session, Some(effect)).await;
332                }
333                Either3::First(InEvent::Disconnected) => {
334                    decoder.reset();
335                }
336                Either3::First(InEvent::Data(data)) => {
337                    for &byte in &data {
338                        let Some(Ok(frame_bytes)) = decoder.push(byte) else {
339                            continue;
340                        };
341                        let now_ms = Instant::now().as_millis();
342                        let effect =
343                            session.handle_frame(frame_bytes, now_ms, &mut |frame: &[u8]| {
344                                emitter.push(frame)
345                            });
346                        emitter.flush().await;
347                        apply_effect(&session, effect).await;
348                    }
349                }
350                Either3::Second(RxFrame { data, info }) => {
351                    session.on_radio_rx(
352                        &data,
353                        info.rssi,
354                        info.snr.as_centibels(),
355                        info.lqi,
356                        &mut |frame: &[u8]| emitter.push(frame),
357                    );
358                    emitter.flush().await;
359                }
360                Either3::Third(result) => {
361                    let now_ms = Instant::now().as_millis();
362                    session.on_tx_result(result.is_ok(), now_ms, &mut |frame: &[u8]| {
363                        emitter.push(frame)
364                    });
365                    emitter.flush().await;
366                }
367            }
368        }
369    }

371    /// Long-press watcher for the user button on P1.10 (active-low,
372    /// pull-up). Two-second hold fires SHUTDOWN_SIGNAL.
373    #[embassy_executor::task]
374    async fn button_task(mut button: Input<'static>) {
375        const HOLD: Duration = Duration::from_secs(2);
376        loop {
377            button.wait_for_low().await;
378            match select(button.wait_for_high(), Timer::after(HOLD)).await {
379                Either::First(()) => {
380                    // Released before HOLD — no-op.
381                }
382                Either::Second(()) => {
383                    SHUTDOWN_SIGNAL.signal(());
384                    button.wait_for_high().await;
385                }
386            }
387        }
388    }

390    /// Controlled power-off: tri-state peripheral signal pins, drop the
391    /// peripheral rail, enter System OFF. No MAC counters or display
392    /// handshake in this firmware. See companion-cli-techo for the
393    /// rationale behind each tri-stated pin.
394    #[embassy_executor::task]
395    async fn shutdown_task(peripheral_power: Output<'static>) -> ! {
396        SHUTDOWN_SIGNAL.wait().await;
397
398        // E-paper SPI bus (SPIM2): SCK=P0.31, MOSI=P1.07, MISO=P0.29
399        // E-paper control:         CS=P0.30, DC=P0.28, RST=P0.02, BUSY=P0.03
400        // Radio SPI bus (TWISPI1): SCK=P0.19, MOSI=P0.22, MISO=P0.23
401        // Radio control:           CS=P0.24, RST=P0.25, BUSY=P0.17, DIO1=P0.20
402        // (E-paper pins are unconfigured in this firmware; tri-stating
403        // them anyway is harmless and keeps the list identical to the
404        // CLI firmware.)
405        for (port, pin) in [
406            (Port::P0, 31u8),
407            (Port::P1, 7u8),
408            (Port::P0, 29u8),
409            (Port::P0, 30u8),
410            (Port::P0, 28u8),
411            (Port::P0, 2u8),
412            (Port::P0, 3u8),
413            (Port::P0, 19u8),
414            (Port::P0, 22u8),
415            (Port::P0, 23u8),
416            (Port::P0, 24u8),
417            (Port::P0, 25u8),
418            (Port::P0, 17u8),
419            (Port::P0, 20u8), // radio DIO1 ← has SENSE set by async radio wait
420        ] {
421            tristate_pin(port, pin);
422        }
423
424        drop(peripheral_power);
425
426        // P1.10 is the side user button. Active-low, pull-up → DETECT-low wakes.
427        power_off(&[WakePin {
428            port: Port::P1,
429            pin: 10,
430            sense: WakeSense::Low,
431        }])
432    }

436    #[embassy_executor::main]
437    async fn main(spawner: Spawner) {
438        let p = embassy_nrf::init(umsh_bsp_nrf52840::clocks::default_config());
439
440        // Peripheral power enable (P0.12). Must be high before the LoRa
441        // module is addressed. Ownership transfers to shutdown_task.
442        let peripheral_power = Output::new(p.P0_12, Level::High, OutputDrive::Standard);
443
444        // WDT: 8 s timeout, petted by the heartbeat task every ~2 s.
445        let mut wdt_config = WdtConfig::default();
446        wdt_config.timeout_ticks = 32768 * 8;
447        let (_wdt, [wdt_handle]) =
448            Watchdog::try_new::<_, 1>(p.WDT, wdt_config).unwrap_or_else(|_| panic!("wdt"));
449
450        // A message in the panic slot means the last reset was a crash;
451        // report that as the reset reason. The slot is cleared either way.
452        let boot_reason = {
453            let mut slot = PanicSlot::new(super::panic::panic_region());
454            if slot.read().is_some() {
455                slot.clear();
456                Status::RESET_CRASH
457            } else {
458                Status::RESET_POWER_ON
459            }
460        };
461
462        // ── SX1262 LoRa radio ────────────────────────────────────────────────
463        // Pin assignment (T-Echo hardware, firmware-confirmed):
464        //   SPI bus: SCK=P0.19, MOSI=P0.22, MISO=P0.23 (TWISPI1)
465        //   CS=P0.24, RST=P0.25, BUSY=P0.17, DIO1=P0.20
466        //   DIO2: internal RF switch; DIO3: 1.8 V TCXO.
467        {
468            let mut cfg = SpimConfig::default();
469            // SX1262 datasheet §8.2: max SCK = 16 MHz, Mode 0.
470            cfg.frequency = Frequency::M16;
471            let radio_bus = Spim::new(
472                p.TWISPI1, Irqs, p.P0_19, // SCK
473                p.P0_23, // MISO
474                p.P0_22, // MOSI
475                cfg,
476            );
477            let radio_cs = Output::new(p.P0_24, Level::High, OutputDrive::Standard);
478            let radio_spi = ExclusiveDevice::new(radio_bus, radio_cs, Delay).unwrap();
479
480            let radio_rst = Output::new(p.P0_25, Level::High, OutputDrive::Standard);
481            let radio_dio1 = Input::new(p.P0_20, Pull::None);
482            let radio_busy = Input::new(p.P0_17, Pull::None);
483
484            let iv = GenericSx126xInterfaceVariant::new(
485                radio_rst, radio_dio1, radio_busy,
486                None, // rf_switch_rx: DIO2 wired internally on the T-Echo module
487                None, // rf_switch_tx: same
488            )
489            .unwrap();
490
491            let lora_config = LoraConfig {
492                chip: Sx1262,
493                tcxo_ctrl: Some(TcxoCtrlVoltage::Ctrl1V8), // DIO3 → 1.8 V TCXO
494                use_dcdc: true,
495                rx_boost: true,
496            };
497
498            // enable_public_network=false → sync word 0x1424 (private).
499            // session_config().sync_word must match this choice.
500            let lora = LoRa::new(Sx126x::new(radio_spi, iv, lora_config), false, Delay)
501                .await
502                .unwrap_or_else(|_| panic!("radio init"));
503
504            spawner.spawn(radio_task(lora).unwrap());
505        }
506
507        // ── USB stack ────────────────────────────────────────────────────────
508        let led = Output::new(p.P0_14, Level::High, OutputDrive::Standard);
509        let driver = Driver::new(p.USBD, Irqs, HardwareVbusDetect::new(Irqs));
510
511        let mut config = Config::new(0x16c0, 0x27dd);
512        config.manufacturer = Some("UMSH");
513        config.product = Some("T-Echo UMSH NCP");
514        config.serial_number = Some("companion-ncp-techo");
515        config.max_power = 100;
516        config.max_packet_size_0 = 64;
517
518        static CONFIG_DESC: StaticCell<[u8; 256]> = StaticCell::new();
519        static BOS_DESC: StaticCell<[u8; 256]> = StaticCell::new();
520        static MSOS_DESC: StaticCell<[u8; 0]> = StaticCell::new();
521        static CONTROL_BUF: StaticCell<[u8; 64]> = StaticCell::new();
522        static STATE: StaticCell<State> = StaticCell::new();
523
524        let mut builder = Builder::new(
525            driver,
526            config,
527            CONFIG_DESC.init([0; 256]),
528            BOS_DESC.init([0; 256]),
529            MSOS_DESC.init([0; 0]),
530            CONTROL_BUF.init([0; 64]),
531        );
532
533        let class = CdcAcmClass::new(&mut builder, STATE.init(State::new()), 64);
534        let mut usb = builder.build();
535
536        let (tx, raw_rx, ctrl) = class.split_with_control();
537        let rx = CdcAcmRescue::new(raw_rx, ctrl);
538
539        spawner.spawn(output_task(tx).unwrap());
540        spawner.spawn(usb_in_task(rx).unwrap());
541        spawner.spawn(ncp_task(boot_reason).unwrap());
542
543        let button = Input::new(p.P1_10, Pull::Up);
544        spawner.spawn(button_task(button).unwrap());
545        spawner.spawn(shutdown_task(peripheral_power).unwrap());
546
547        join(usb.run(), heartbeat(led, wdt_handle)).await;
548    }

552    async fn heartbeat(mut led: Output<'static>, mut wdt: WatchdogHandle) -> ! {
553        let mut engine = LedEngine::new(LedTimings::default(), Instant::now().as_millis());
554        loop {
555            wdt.pet();
