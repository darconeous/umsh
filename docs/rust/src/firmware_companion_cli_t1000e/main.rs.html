1// Seeed SenseCAP T1000-E companion-radio CLI firmware — Phase 3 bringup.
2//
3// Phases 0-2 established: bootloader recon, USB-CDC, WDT, panic persist,
4// DFU rescue paths, button FSM (long-press → shutdown, triple-tap → DFU),
5// ordered System OFF with GPIO wake.
6//
7// Phase 2.5 proved the LR1110 radio receives MeshCore-US packets. The
8// root fixes were in the lora-phy fork (read_buffer offset+length framing,
9// calibration mask 0x3F, RadioLib-matching init sequence).
10//
11// Phase 3 wires CliSession over USB-CDC using the same MAC + radio_runner
12// pattern as companion-cli-techo (T-Echo Phase 6). Changes from Phase 2.5:
13//   - echo_task replaced by umsh_task (Host::run + CliSession::run)
14//   - radio_task replaced by radio_runner_task (umsh_radio_loraphy::runner)
15//   - output_task upgraded to cli_io::drain_to_sender (64-byte chunk drain)
16//   - NVMC identity persistence (first-boot TRNG key generation)
17//   - NVMC counter / peer / channel persistence
18//   - PowerSignaler connects /poweroff CLI command to SHUTDOWN_SIGNAL
19//
20// Boot sequence:
21//   1. Init heap allocator.
22//   2. Hold LR1110 RESET low (keeps radio quiet during USB init).
23//   3. Read button (P0.06); if held, enter serial DFU immediately.
24//   4. Arm the watchdog (8 s timeout, petted by heartbeat).
25//   5. Read any panic message left by the previous boot.
26//   6. Init NVMC storage (64 KB at 0xE4000..0xF4000).
27//   7. Load or TRNG-generate the local Ed25519 secret key.
28//   8. Build Mac<T1000EPlatform> and load persisted TX counter.
29//   9. Init LR1110 SPI + LoRa::new; derive MeshCore-US params.
30//  10. Set up USB-CDC with CdcAcmRescue (1200-baud touch + escape rescue).
31//  11. Spawn output_task, radio_runner_task, button_task, shutdown_task,
32//      umsh_task.
33//  12. Join usb.run / heartbeat in main.
34//
35// Task layout:
36//   - main():              joins usb.run / heartbeat
37//   - output_task:         owns USB Sender; drains cli_io::OUTPUT_CH
38//   - radio_runner_task:   owns LoRa<LR1110>; loops continuous RX ↔ TX
39//   - umsh_task:           host.run() + cli.run() via select; owns CdcInput
40//   - button_task:         owns P0.06 Input; runs ButtonFsm
41//   - shutdown_task:       awaits SHUTDOWN_SIGNAL; performs ordered System OFF
42//   - heartbeat (inline):  LED + WDT pet; runs in join with usb.run
43//
44// T1000-E pin notes (all confirmed against MeshCore variants/t1000-e):
45//   LED:    P0.24  active-HIGH  (set_high = on)
46//   Button: P0.06  active-HIGH  pull-down  (HIGH = pressed, WakeSense::High)
47//   LR1110: SCK=P0.11, CS=P0.12, MISO=P1.08, MOSI=P1.09, RST=P1.10
48//           DIO1/IRQ=P1.01, BUSY=P0.07
49//           DIO3: 1.6V TCXO control (set by lora-phy)
50//           DIO5-8: internal RF switch (set via RfSwitchConfig)
51

270    /// Runs the `CliSession` over USB-CDC. This is the only task that
271    /// blocks on a host terminal connection — everything else (radio, MAC,
272    /// button, buzzer, beacon) runs without it.
273    #[embassy_executor::task]
274    async fn cli_task(
275        node: T1000ENode,
276        local_key: PublicKey,
277        storage: &'static NvmcStorage,
278        rx: T1000eRescue,
279        prev_panic_buf: &'static [u8; 256],
280        prev_panic_len: usize,
281    ) {
282        use umsh_cli::CliSession;
283        use umsh_cli::io::CliOutput;
284        use umsh_cli::logger::NullLogger;
285
286        let mut input = cli_io::CdcInput::new(rx);
287        let mut out = cli_io::CdcOutput::new();
288
289        // Wait for the host to open the CDC port before emitting the banner.
290        input.wait_connection().await;
291
292        let _ = out.write_line("").await;
293        let _ = out.write_line("UMSH CLI (T1000-E)").await;
294        let _ = out.write_line("type /help for commands").await;
295        if prev_panic_len > 0 {
296            let _ = out.write_line("[PREV PANIC]:").await;
297            if let Ok(s) = core::str::from_utf8(&prev_panic_buf[..prev_panic_len]) {
298                let _ = out.write_line(s).await;
299            }
300        }
301
302        let peer_store = NvmcPeerStore::new(storage);
303        let channel_store = NvmcChannelStore::new(storage);
304        let mut cli: CliSession<_, _, _, _, _, _, 4, 4, 2, 8, 128> = CliSession::new(
305            node,
306            local_key,
307            out,
308            NullLogger::new(),
309            peer_store,
310            channel_store,
311            PowerSignaler,
312        );
313
314        // `run` loads peers/channels from storage and registers them with the
315        // MAC (idempotent) and the CLI display tables before entering the loop.
316        let _ = cli.run(&mut input).await;
317        panic!("cli exited");
318    }

320    /// Resolves raw GPIO edges on the user button (P0.06, active-high, pull-down)
321    /// into `ButtonFsm` events. `Long` raises `SHUTDOWN_SIGNAL`;
322    /// `Triple` stays inert (reserved for GPS power control).
323    #[embassy_executor::task]
324    async fn button_task(mut button: Input<'static>, storage: &'static NvmcStorage) {
325        let mut fsm = ButtonFsm::new(ButtonTimings::default());
326        let mut pressed = button.is_high();
327        loop {
328            let event = {
329                let now_ms = Instant::now().as_millis();
330                let edge_fut = async {
331                    if pressed {
332                        button.wait_for_low().await;
333                        Timer::after(DEBOUNCE).await;
334                        ButtonEdge::Release
335                    } else {
336                        button.wait_for_high().await;
337                        Timer::after(DEBOUNCE).await;
338                        ButtonEdge::Press
339                    }
340                };
341                let timeout_deadline_ms =
342                    fsm.next_deadline().unwrap_or(now_ms.saturating_add(60_000));
343                let timer_fut = Timer::at(Instant::from_millis(timeout_deadline_ms));
344                match select(edge_fut, timer_fut).await {
345                    Either::First(edge) => {
346                        pressed = matches!(edge, ButtonEdge::Press);
347                        fsm.on_edge(edge, Instant::now().as_millis())
348                    }
349                    Either::Second(()) => fsm.poll(Instant::now().as_millis()),
350                }
351            };
352
353            match event {
354                Some(ButtonEvent::Single) => {
355                    umsh_bsp_t1000e::indicator::LED_SEQUENCE_SIGNAL
356                        .signal(LedSequence::ActionConfirm);
357                    BEACON_SIGNAL.signal(());
358                }
359                Some(ButtonEvent::Double) => {
360                    let preferences = umsh_bsp_t1000e::preferences::toggle_silent();
361                    umsh_bsp_t1000e::BUZZER_SILENCE_SET.signal(preferences.silent);
362                    umsh_bsp_t1000e::indicator::LED_SEQUENCE_SIGNAL
363                        .signal(LedSequence::ActionConfirm);
364                    // Confirmation reflects the applied (in-RAM) toggle. A
365                    // failed flash write costs only durability across the
366                    // next reset; there is no user-facing fault channel.
367                    let mut durable = preferences;
368                    durable.battery_critical = false;
369                    let _ = storage.store_tracker_preferences(durable.encode()).await;
370                }
371                Some(ButtonEvent::Triple) => {
372                    // Reserved for GPS power control. This firmware does not
373                    // yet own a GNSS task, so the slot remains inert.
374                }
375                Some(ButtonEvent::Quad) => {
376                    // No action defined for Quad yet
377                }
378                Some(ButtonEvent::Long) => {
379                    pressed = false;
380                    fsm = ButtonFsm::new(ButtonTimings::default());
381                    // shutdown_task persists the Asleep preference for every
382                    // shutdown source (button, PowerSignaler, battery cutoff).
383                    umsh_bsp_t1000e::preferences::set_asleep(true);
384                    SHUTDOWN_SIGNAL.signal(());
385                }
386                _ => {}
387            }
388        }
389    }

421    async fn heartbeat(mut led: SimplePwm<'static>, mut wdt: WatchdogHandle) -> ! {
422        led.set_period(1_000);
423        led.enable();
424        let mut engine = T1000eLedEngine::new(Instant::now().as_millis());
425        loop {
426            wdt.pet();
427            engine.set_battery(umsh_bsp_t1000e::battery_state());
428            engine.set_attention(umsh_bsp_t1000e::indicator::attention_requested());
429            let decision = engine.tick(Instant::now().as_millis());
430            let duty =
431                ((u32::from(led.max_duty()) * u32::from(decision.brightness)) / 1_000) as u16;
432            led.set_duty(0, DutyCycle::inverted(duty));
433            match select4(
434                Timer::at(Instant::from_millis(decision.next_deadline_ms)),
435                umsh_bsp_t1000e::BATTERY_STATE_CHANGED.wait(),
436                umsh_bsp_t1000e::indicator::INDICATOR_CHANGED.wait(),
437                umsh_bsp_t1000e::indicator::LED_SEQUENCE_SIGNAL.wait(),
438            )
439            .await
440            {
441                Either4::First(()) | Either4::Second(_) | Either4::Third(()) => {}
442                Either4::Fourth(sequence) => {
443                    engine.play(sequence, Instant::now().as_millis());
444                }
445            }
446        }
447    }

449    #[embassy_executor::main]
450    async fn main(spawner: Spawner) {
451        // Init heap before any alloc-using code. 8 KiB is generous; runtime
452        // alloc is near-zero since all MAC state lives in static arrays.
453        {
454            const HEAP_SIZE: usize = 8192;
455            static mut HEAP: [MaybeUninit<u8>; HEAP_SIZE] = [MaybeUninit::uninit(); HEAP_SIZE];
456            unsafe { crate::ALLOCATOR.init(core::ptr::addr_of!(HEAP) as usize, HEAP_SIZE) }
457        }
458
459        let p = embassy_nrf::init(umsh_bsp_nrf52840::clocks::default_config());
460
461        // Mount durable user preferences before deciding whether normal
462        // application startup is permitted. GPREGRET2 remains only a fast
463        // mirror and a retained critical-shutdown reason.
464        let storage: &'static NvmcStorage = STORAGE.init(flash_store::new_storage(Nvmc::new(p.NVMC)));
465
466        let reset_reasons = embassy_nrf::pac::POWER.resetreas().read();
467        embassy_nrf::pac::POWER
468            .resetreas()
469            .write(|reasons| reasons.0 = u32::MAX);
470        // RESETREAS.OFF alone proves a button wake: P0.06 is the only GPIO
471        // DETECT source armed at System OFF entry (USB insertion wakes via
472        // the native VBUS detector and sets its own reason bit). The pin
473        // itself cannot be sampled this early — PIN_CNF resets to
474        // input-disconnected, so the IN register reads 0 regardless of the
475        // physical level.
476        let woke_from_system_off = reset_reasons.off();
477        let external_power_present = umsh_bsp_t1000e::power::usb_power_present();
478        let gpregret_state = umsh_bsp_t1000e::preferences::load_retained();
479        let retained_critical =
480            gpregret_state.is_some_and(|preferences| preferences.battery_critical);
481        let mut retained_state = storage
482            .load_tracker_preferences()
483            .await
484            .ok()
485            .flatten()
486            .and_then(umsh_ux_tracker::state::UserPreferences::try_decode)
487            .unwrap_or_default();
488        retained_state.battery_critical = retained_critical;
489        umsh_bsp_t1000e::preferences::store(retained_state);
490        if retained_state.battery_critical && !external_power_present {
491            umsh_bsp_t1000e::shutdown::resume_persisted_sleep().await;
492        }
493        if retained_state.battery_critical && external_power_present {
494            umsh_bsp_t1000e::preferences::set_battery_critical(false);
495        }
496        let charging_sleep_wake_requested =
497            reset_reasons.sreq() && gpregret_state.is_some_and(|preferences| !preferences.asleep);
498        if woke_from_system_off || charging_sleep_wake_requested {
499            let preferences = umsh_bsp_t1000e::preferences::set_asleep(false);
500            let mut durable = preferences;
501            durable.battery_critical = false;
502            let _ = storage.store_tracker_preferences(durable.encode()).await;
503        } else if umsh_bsp_t1000e::preferences::load().asleep {
504            if external_power_present {
505                let mut led_config = SimpleConfig::default();
506                led_config.prescaler = Prescaler::Div16;
507                let led_pwm = SimplePwm::new_1ch(p.PWM1, p.P0_24, &led_config);
508                let sleep_button = Input::new(p.P0_06, Pull::Down);
509                let sleep_external_power = Input::new(p.P0_05, Pull::Down);
510                let sleep_charge_active = Input::new(p.P1_03, Pull::Up);
511                umsh_bsp_t1000e::shutdown::run_charging_sleep(
512                    led_pwm,
513                    sleep_button,
514                    sleep_external_power,
515                    sleep_charge_active,
516                )
517                .await;
518            } else {
519                umsh_bsp_t1000e::shutdown::resume_persisted_sleep().await;
520            }
521        }
522
523        // Seize LR1110 RESET immediately and hold it low. The LR1110 can
524        // outlive nRF soft resets; if a previous image left it in a bad state
525        // (e.g. broken DCDC on this board), holding RESET prevents that state
526        // from destabilizing USB before LoRa::new() runs.
527        let radio_rst = Output::new(p.P1_10, Level::Low, OutputDrive::Standard);
528
529        // User button (active-HIGH, pull-down). DFU entry is the
530        // bootloader's 1200-baud touch, never a button gesture.
531        let button = Input::new(p.P0_06, Pull::Down);
532
533        // WDT: 8 s timeout, petted by heartbeat.
534        let mut wdt_config = WdtConfig::default();
535        wdt_config.timeout_ticks = 32768 * 8;
536        let (_wdt, [wdt_handle]) =
537            Watchdog::try_new::<_, 1>(p.WDT, wdt_config).unwrap_or_else(|_| panic!("wdt"));
538
539        // Read any panic message left by the previous boot, then clear it.
540        static PREV_PANIC_BUF: StaticCell<[u8; 256]> = StaticCell::new();
541        let mut prev_panic_tmp = [0u8; 256];
542        let prev_panic_len = {
543            let mut slot = PanicSlot::new(super::panic::panic_region());
544            if let Some(msg) = slot.read() {
545                let n = msg.len().min(prev_panic_tmp.len());
546                prev_panic_tmp[..n].copy_from_slice(&msg[..n]);
547                slot.clear();
548                n
549            } else {
550                0
551            }
552        };
553        let prev_panic_buf: &'static [u8; 256] = PREV_PANIC_BUF.init(prev_panic_tmp);
554
555        let mut led_config = SimpleConfig::default();
556        led_config.prescaler = Prescaler::Div16;
557        let led = SimplePwm::new_1ch(p.PWM1, p.P0_24, &led_config);
558
559        // ── Piezo buzzer ─────────────────────────────────────────────────────
560        // P0.25 = PWM, P1.05 = power-enable for the buzzer driver chip.
561        // Div16 prescaler gives a 1 MHz PWM clock — comfortably covers the
562        // 1–2 kHz melody range with max_duty 500–1000.
563        let buzzer_pwm = {
564            let mut cfg = SimpleConfig::default();
565            cfg.prescaler = Prescaler::Div16;
566            SimplePwm::new_1ch(p.PWM0, p.P0_25, &cfg)
567        };
568        let buzzer_enable = Output::new(p.P1_05, Level::Low, OutputDrive::Standard);
569        let initial_preferences = umsh_bsp_t1000e::preferences::load();
570        spawner.spawn(buzzer_task(buzzer_pwm, buzzer_enable, initial_preferences.silent).unwrap());
571        // Boot chirp — independent of USB, so headless boots also signal life.
572        umsh_bsp_t1000e::BUZZER_SIGNAL.signal(&buzzer_melodies::POWER_ON);
573
574        // ── Local identity ────────────────────────────────────────────────────
575        // The hardware-TRNG RNG built here is the single RNG path for this
576        // firmware — used for first-boot identity generation AND passed
577        // ownership-by-value into `Mac::new` below as `Platform::Rng`.
578        //
579        // Load identity from flash on subsequent boots; TRNG-generate on
580        // first boot. We do NOT fall back to any PRNG on failure — a
581        // predictable long-term key is worse than panicking.
582        let mut rng = Nrf52840Rng::new(p.RNG);
583        let sk_bytes: [u8; 32] = match storage.load_sk().await {
584            Ok(Some(sk)) => sk,
585            Ok(None) => {
586                let mut sk = [0u8; 32];
587                rng.fill_bytes(&mut sk);
588                storage
589                    .store_sk(&sk)
590                    .await
591                    .unwrap_or_else(|_| panic!("identity persist"));
592                sk
593            }
594            Err(_) => panic!("storage init failed"),
595        };
596        let identity = SoftwareIdentity::from_secret_bytes(&sk_bytes);
597        let local_key = *identity.public_key();
598
599        // ── LR1110 LoRa radio ─────────────────────────────────────────────────
600        // Pin map (confirmed against MeshCore variants/t1000-e):
601        //   SPI bus: SCK=P0.11, MISO=P1.08, MOSI=P1.09 (TWISPI0)
602        //   CS=P0.12, RST=P1.10, IRQ/DIO1=P1.01, BUSY=P0.07
603        //   DIO3: 1.6 V TCXO control (handled by lora-phy SetDIO3AsTCXOCtrl)
604        //   DIO5-8: internal RF switch (handled by RfSwitchConfig via SetDioAsRfSwitch)
605        let t_frame_ms = umsh_radio_loraphy::airtime_ms(
606            SpreadingFactor::_7,
607            Bandwidth::_62KHz,
608            umsh_radio_loraphy::MAX_PAYLOAD,
609        );
610        {
611            let mut cfg = SpimConfig::default();
612            cfg.frequency = Frequency::M8;
613            let radio_bus = Spim::new(
614                p.TWISPI0, Irqs, p.P0_11, // SCK
615                p.P1_08, // MISO
616                p.P1_09, // MOSI
617                cfg,
618            );
619            let radio_cs = Output::new(p.P0_12, Level::High, OutputDrive::Standard);
620            let radio_spi = ExclusiveDevice::new(radio_bus, radio_cs, Delay).unwrap();
621
622            // Pull::Down on DIO1: the LR1110's IRQ output is push-pull active-high,
623            // so a pull-down prevents floating reads when IRQ is de-asserted.
624            let radio_interrupt = Input::new(p.P1_01, Pull::Down);
625            let radio_busy = Input::new(p.P0_07, Pull::None);
626
627            let iv = GenericLr1110InterfaceVariant::new(
628                radio_rst,
629                radio_interrupt,
630                radio_busy,
631                None, // rf_switch_rx: not external — DIO5-8 handle it internally
632                None, // rf_switch_tx: same
633            )
634            .unwrap_or_else(|_| panic!("lr1110 iv"));
635
636            let lora_config = LoraConfig {
637                // HP PA — SetTx will route through tx_hp (0x0A = DIO6+DIO8)
638                // on our RF-switch table. Combined with TX_POWER_DBM=22 this
639                // is the maximum output the chip + board can produce.
640                chip: Lr1110Chip::with_pa(PaSelection::Hp),
641                tcxo_ctrl: Some(TcxoCtrlVoltage::Ctrl1V6),
642                use_dcdc: false, // T1000-E module has no external inductor for BST
643                rx_boost: true,
644                rf_switch: Some(RF_SWITCH),
645            };
646
647            // enable_public_network=false → private sync word 0x1424,
648            // matching MeshCore's RADIOLIB_SX126X_SYNC_WORD_PRIVATE = 0x12.
649            let mut lora = LoRa::new(Lr1110::new(radio_spi, iv, lora_config), false, Delay)
650                .await
651                .unwrap_or_else(|_| panic!("radio init"));
652
653            // MeshCore-US on-air parameters tuned for LR1110.
654            //
655            // Phase 2.5 RX bringup proved that preamble_length=16 (matching
656            // MeshCore's RadioLib TX preamble) reliably triggers
657            // SyncWordHeaderValid → RxDone on the LR1110. The shared
658            // `umsh_radio_loraphy::meshcore_us_params` helper uses 8 for RX,
659            // which is fine for the SX1262 on T-Echo but loses packets on
660            // LR1110. We pin both rx and tx to 16 here to match what worked
661            // on real hardware.
662            let mdltn = lora
663                .create_modulation_params(
664                    SpreadingFactor::_7,
665                    Bandwidth::_62KHz,
666                    CodingRate::_4_5,
667                    910_525_000,
668                )
669                .unwrap_or_else(|_| panic!("modulation params"));
670            let rx_pkt = lora
671                .create_rx_packet_params(
672                    16,    // preamble length: LR1110 needs 16 for MeshCore-US
673                    false, // explicit header
674                    255,   // max payload
675                    true,  // CRC on
676                    false, // IQ normal
677                    &mdltn,
678                )
679                .unwrap_or_else(|_| panic!("rx packet params"));
680            let tx_pkt = lora
681                .create_tx_packet_params(
682                    16,    // preamble length: matches MeshCore RadioLib
683                    false, // explicit header
684                    true,  // CRC on
685                    false, // IQ normal
686                    &mdltn,
687                )
688                .unwrap_or_else(|_| panic!("tx packet params"));
689
690            spawner.spawn(radio_runner_task(lora, mdltn, rx_pkt, tx_pkt).unwrap());
691        }
692
693        // ── MAC coordinator ───────────────────────────────────────────────────
694        let radio_handle = umsh_radio_loraphy::LoraphyRadio::new(&RADIO_CH, t_frame_ms);
695        let crypto = CryptoEngine::new(SoftwareAes, SoftwareSha256);
696        let mut mac = T1000EMac::new(
697            radio_handle,
698            crypto,
699            EmbassyClock,
700            rng,
701            NvmcCounterStore::new(storage),
702            RepeaterConfig::default(),
703            OperatingPolicy::default(),
704        );
705        let identity_id = mac
706            .add_identity(identity)
707            .unwrap_or_else(|_| panic!("identity"));
708        // Restore TX frame-counter boundary so the counter never rewinds.
709        mac.load_persisted_counter(identity_id)
710            .await
711            .unwrap_or_else(|_| panic!("tx counter load"));
712        let mac_cell: &'static AsyncRefCell<T1000EMac> = MAC_CELL.init(AsyncRefCell::new(mac));
713
714        // ── USB stack ─────────────────────────────────────────────────────────
715        let driver = Driver::new(p.USBD, Irqs, HardwareVbusDetect::new(Irqs));
716
717        let mut config = Config::new(0x2886, 0x0057);
718        config.manufacturer = Some("Seeed");
719        config.product = Some("T1000-E UMSH CLI");
720        config.serial_number = Some("umsh-t1000e");
721        config.max_power = 100;
722        config.max_packet_size_0 = 64;
723
724        static CONFIG_DESC: StaticCell<[u8; 256]> = StaticCell::new();
725        static BOS_DESC: StaticCell<[u8; 256]> = StaticCell::new();
726        static MSOS_DESC: StaticCell<[u8; 0]> = StaticCell::new();
727        static CONTROL_BUF: StaticCell<[u8; 64]> = StaticCell::new();
728        static STATE: StaticCell<State> = StaticCell::new();
729
730        let mut builder = Builder::new(
731            driver,
732            config,
733            CONFIG_DESC.init([0; 256]),
734            BOS_DESC.init([0; 256]),
735            MSOS_DESC.init([0; 0]),
736            CONTROL_BUF.init([0; 64]),
737        );
738
739        let class = CdcAcmClass::new(&mut builder, STATE.init(State::new()), 64);
740        let mut usb = builder.build();
741
742        let (tx, raw_rx, ctrl) = class.split_with_control();
743        let rx = CdcAcmRescue::new(raw_rx, ctrl);
744
745        // ── Battery ADC ───────────────────────────────────────────────────────
746        // P0.02 = AIN0 via 2:1 divider; sensor rail P1.06 gates the path.
747        let sensor_rail = Output::new(p.P1_06, Level::Low, OutputDrive::Standard);
748        let saadc = Saadc::new(
749            p.SAADC,
750            Irqs,
751            SaadcConfig::default(), // 12-bit, no oversample
752            [ChannelConfig::single_ended(p.P0_02)],
753        );
754        let external_power = Input::new(p.P0_05, Pull::Down);
755        let charge_active = Input::new(p.P1_03, Pull::Up);
756
757        // ── Host + LocalNode ──────────────────────────────────────────────────
758        // Build the Host and add the local identity's node here in main() so
759        // we can clone the node for the beacon task before moving Host into
760        // mac_task. The Host's internal node store and the cloned LocalNode
761        // share Rc state, so events route correctly regardless of which task
762        // holds which copy.
763        let handle = MacHandle::new(mac_cell);
764        let mut host: T1000EHost = Host::new(handle);
765        let node = host.add_node(identity_id);
766        let beacon_node = node.clone();
767
768        // Register persisted peers and channels into the MAC at boot, before
769        // spawning tasks. The CLI task must not be the first thing that
770        // registers these — it only runs after a host opens the CDC port, and
771        // the MAC needs the keys from the very first packet.
772        {
773            let mut peer_buf: heapless::Vec<([u8; 32], Option<heapless::String<16>>), 8> =
774                heapless::Vec::new();
775            let _ = storage.load_all_peers(&mut peer_buf).await;
776            let mut ch_buf: heapless::Vec<(heapless::String<16>, [u8; 32]), 2> =
777                heapless::Vec::new();
778            let _ = storage.load_all_channels(&mut ch_buf).await;
779            for (pk, _alias) in peer_buf.iter() {
780                let _ = node.peer(PublicKey(*pk)).await;
781            }
782            for (name, key_bytes) in ch_buf.iter() {
783                let channel =
784                    umsh_node::Channel::private(umsh_core::ChannelKey(*key_bytes), name.as_str());
785                let _ = node.join(&channel).await;
786            }
787        }
788
789        // Restore RX counter boundaries before the MAC starts processing
790        // packets so the replay window starts above the last accepted frame.
791        // Runs after the peer registration above so the persisted boundaries
792        // actually land on registered peers.
793        MacHandle::new(mac_cell)
794            .load_all_persisted_rx_counters()
795            .await
796            .ok();
797
798        spawner.spawn(output_task(tx).unwrap());
799        spawner.spawn(button_task(button, storage).unwrap());
800        spawner.spawn(shutdown_task(storage).unwrap());
801        spawner.spawn(power_task(saadc, sensor_rail, external_power, charge_active).unwrap());
802        spawner.spawn(mac_task(host).unwrap());
803        spawner.spawn(beacon_task(beacon_node).unwrap());
804        spawner
805            .spawn(cli_task(node, local_key, storage, rx, prev_panic_buf, prev_panic_len).unwrap());
806
807        join(usb.run(), heartbeat(led, wdt_handle)).await;
808    }
809}
