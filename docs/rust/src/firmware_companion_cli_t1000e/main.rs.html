1// Seeed SenseCAP T1000-E companion-radio CLI firmware — Phase 3 bringup.
2//
3// Phases 0-2 established: bootloader recon, USB-CDC, WDT, panic persist,
4// DFU rescue paths, button FSM (long-press → shutdown, triple-tap → DFU),
5// ordered System OFF with GPIO wake.
6//
7// Phase 2.5 proved the LR1110 radio receives MeshCore-US packets. The
8// root fixes were in the lora-phy fork (read_buffer offset+length framing,
9// calibration mask 0x3F, RadioLib-matching init sequence).
10//
11// Phase 3 wires CliSession over USB-CDC using the same MAC + radio_runner
12// pattern as companion-cli-techo (T-Echo Phase 6). Changes from Phase 2.5:
13//   - echo_task replaced by umsh_task (Host::run + CliSession::run)
14//   - radio_task replaced by radio_runner_task (umsh_radio_loraphy::runner)
15//   - output_task upgraded to cli_io::drain_to_sender (64-byte chunk drain)
16//   - NVMC identity persistence (first-boot TRNG key generation)
17//   - NVMC counter / peer / channel persistence
18//   - PowerSignaler connects /poweroff CLI command to SHUTDOWN_SIGNAL
19//
20// Boot sequence:
21//   1. Init heap allocator.
22//   2. Hold LR1110 RESET low (keeps radio quiet during USB init).
23//   3. Read button (P0.06); if held, enter serial DFU immediately.
24//   4. Arm the watchdog (8 s timeout, petted by heartbeat).
25//   5. Read any panic message left by the previous boot.
26//   6. Init NVMC storage (64 KB at 0xE4000..0xF4000).
27//   7. Load or TRNG-generate the local Ed25519 secret key.
28//   8. Build Mac<T1000EPlatform> and load persisted TX counter.
29//   9. Init LR1110 SPI + LoRa::new; derive MeshCore-US params.
30//  10. Set up USB-CDC with CdcAcmRescue (1200-baud touch + escape rescue).
31//  11. Spawn output_task, radio_runner_task, button_task, shutdown_task,
32//      umsh_task.
33//  12. Join usb.run / heartbeat in main.
34//
35// Task layout:
36//   - main():              joins usb.run / heartbeat
37//   - output_task:         owns USB Sender; drains cli_io::OUTPUT_CH
38//   - radio_runner_task:   owns LoRa<LR1110>; loops continuous RX ↔ TX
39//   - umsh_task:           host.run() + cli.run() via select; owns CdcInput
40//   - button_task:         owns P0.06 Input; runs ButtonFsm
41//   - shutdown_task:       awaits SHUTDOWN_SIGNAL; performs ordered System OFF
42//   - heartbeat (inline):  LED + WDT pet; runs in join with usb.run
43//
44// T1000-E pin notes (all confirmed against MeshCore variants/t1000-e):
45//   LED:    P0.24  active-HIGH  (set_high = on)
46//   Button: P0.06  active-HIGH  pull-down  (HIGH = pressed, WakeSense::High)
47//   LR1110: SCK=P0.11, CS=P0.12, MISO=P1.08, MOSI=P1.09, RST=P1.10
48//           DIO1/IRQ=P1.01, BUSY=P0.07
49//           DIO3: 1.6V TCXO control (set by lora-phy)
50//           DIO5-8: internal RF switch (set via RfSwitchConfig)

265    /// Runs the `CliSession` over USB-CDC. This is the only task that
266    /// blocks on a host terminal connection — everything else (radio, MAC,
267    /// button, buzzer, beacon) runs without it.
268    #[embassy_executor::task]
269    async fn cli_task(
270        node: T1000ENode,
271        local_key: PublicKey,
272        storage: &'static NvmcStorage,
273        rx: T1000eRescue,
274        prev_panic_buf: &'static [u8; 256],
275        prev_panic_len: usize,
276    ) {
277        use umsh_cli::CliSession;
278        use umsh_cli::io::CliOutput;
279        use umsh_cli::logger::NullLogger;
280
281        let mut input = cli_io::CdcInput::new(rx);
282        let mut out = cli_io::CdcOutput::new();
283
284        // Wait for the host to open the CDC port before emitting the banner.
285        input.wait_connection().await;
286
287        let _ = out.write_line("").await;
288        let _ = out.write_line("UMSH CLI (T1000-E)").await;
289        let _ = out.write_line("type /help for commands").await;
290        if prev_panic_len > 0 {
291            let _ = out.write_line("[PREV PANIC]:").await;
292            if let Ok(s) = core::str::from_utf8(&prev_panic_buf[..prev_panic_len]) {
293                let _ = out.write_line(s).await;
294            }
295        }
296
297        let peer_store = NvmcPeerStore::new(storage);
298        let channel_store = NvmcChannelStore::new(storage);
299        let mut cli: CliSession<_, _, _, _, _, _, 4, 4, 2, 8, 128> = CliSession::new(
300            node,
301            local_key,
302            out,
303            NullLogger::new(),
304            peer_store,
305            channel_store,
306            PowerSignaler,
307        );
308
309        // `run` loads peers/channels from storage and registers them with the
310        // MAC (idempotent) and the CLI display tables before entering the loop.
311        let _ = cli.run(&mut input).await;
312        panic!("cli exited");
313    }

315    /// Resolves raw GPIO edges on the user button (P0.06, active-high, pull-down)
316    /// into `ButtonFsm` events. `Long` raises `SHUTDOWN_SIGNAL`,
317    /// `Triple` enters UF2 DFU directly (diverges via system reset).
318    #[embassy_executor::task]
319    async fn button_task(mut button: Input<'static>) {
320        let mut fsm = ButtonFsm::new(ButtonTimings::default());
321        let mut pressed = button.is_high();
322        loop {
323            let event = {
324                let now_ms = Instant::now().as_millis();
325                let edge_fut = async {
326                    if pressed {
327                        button.wait_for_low().await;
328                        Timer::after(DEBOUNCE).await;
329                        ButtonEdge::Release
330                    } else {
331                        button.wait_for_high().await;
332                        Timer::after(DEBOUNCE).await;
333                        ButtonEdge::Press
334                    }
335                };
336                let timeout_deadline_ms =
337                    fsm.next_deadline().unwrap_or(now_ms.saturating_add(60_000));
338                let timer_fut = Timer::at(Instant::from_millis(timeout_deadline_ms));
339                match select(edge_fut, timer_fut).await {
340                    Either::First(edge) => {
341                        pressed = matches!(edge, ButtonEdge::Press);
342                        fsm.on_edge(edge, Instant::now().as_millis())
343                    }
344                    Either::Second(()) => fsm.poll(Instant::now().as_millis()),
345                }
346            };
347
348            match event {
349                Some(ButtonEvent::Single) => {
350                    BEACON_SIGNAL.signal(());
351                }
352                Some(ButtonEvent::Double) => {
353                    umsh_bsp_t1000e::BUZZER_SILENCE_TOGGLE.signal(());
354                }
355                Some(ButtonEvent::Triple) => {
356                    umsh_bsp_nrf52840::gpregret::enter_dfu_uf2();
357                }
358                Some(ButtonEvent::Quad) => {
359                    // No action defined for Quad yet
360                }
361                Some(ButtonEvent::Long) => {
362                    pressed = false;
363                    fsm = ButtonFsm::new(ButtonTimings::default());
364                    SHUTDOWN_SIGNAL.signal(());
365                }
366                _ => {}
367            }
368        }
369    }

386    async fn heartbeat(mut led: Output<'static>, mut wdt: WatchdogHandle) -> ! {
387        let mut engine = LedEngine::new(LedTimings::default(), Instant::now().as_millis());
388        loop {
389            wdt.pet();
390            let decision = engine.tick(Instant::now().as_millis());
391            if decision.on {
392                led.set_high()
393            } else {
394                led.set_low()
395            }
396            Timer::at(Instant::from_millis(decision.next_deadline_ms)).await;
397        }
398    }

400    #[embassy_executor::main]
401    async fn main(spawner: Spawner) {
402        // Init heap before any alloc-using code. 8 KiB is generous; runtime
403        // alloc is near-zero since all MAC state lives in static arrays.
404        {
405            const HEAP_SIZE: usize = 8192;
406            static mut HEAP: [MaybeUninit<u8>; HEAP_SIZE] = [MaybeUninit::uninit(); HEAP_SIZE];
407            unsafe { crate::ALLOCATOR.init(core::ptr::addr_of!(HEAP) as usize, HEAP_SIZE) }
408        }
409
410        let p = embassy_nrf::init(umsh_bsp_nrf52840::clocks::default_config());
411
412        // Seize LR1110 RESET immediately and hold it low. The LR1110 can
413        // outlive nRF soft resets; if a previous image left it in a bad state
414        // (e.g. broken DCDC on this board), holding RESET prevents that state
415        // from destabilizing USB before LoRa::new() runs.
416        let radio_rst = Output::new(p.P1_10, Level::Low, OutputDrive::Standard);
417
418        // Button-held-at-boot DFU check (active-HIGH, pull-down).
419        let button = Input::new(p.P0_06, Pull::Down);
420        cortex_m::asm::delay(640_000); // ~10 ms settle
421        if button.is_high() {
422            umsh_bsp_nrf52840::gpregret::enter_dfu_serial();
423        }
424
425        // WDT: 8 s timeout, petted by heartbeat.
426        let mut wdt_config = WdtConfig::default();
427        wdt_config.timeout_ticks = 32768 * 8;
428        let (_wdt, [wdt_handle]) =
429            Watchdog::try_new::<_, 1>(p.WDT, wdt_config).unwrap_or_else(|_| panic!("wdt"));
430
431        // Read any panic message left by the previous boot, then clear it.
432        static PREV_PANIC_BUF: StaticCell<[u8; 256]> = StaticCell::new();
433        let mut prev_panic_tmp = [0u8; 256];
434        let prev_panic_len = {
435            let mut slot = PanicSlot::new(super::panic::panic_region());
436            if let Some(msg) = slot.read() {
437                let n = msg.len().min(prev_panic_tmp.len());
438                prev_panic_tmp[..n].copy_from_slice(&msg[..n]);
439                slot.clear();
440                n
441            } else {
442                0
443            }
444        };
445        let prev_panic_buf: &'static [u8; 256] = PREV_PANIC_BUF.init(prev_panic_tmp);
446
447        let led = Output::new(p.P0_24, Level::Low, OutputDrive::Standard);
448
449        // ── Piezo buzzer ─────────────────────────────────────────────────────
450        // P0.25 = PWM, P1.05 = power-enable for the buzzer driver chip.
451        // Div16 prescaler gives a 1 MHz PWM clock — comfortably covers the
452        // 1–2 kHz melody range with max_duty 500–1000.
453        let buzzer_pwm = {
454            let mut cfg = SimpleConfig::default();
455            cfg.prescaler = Prescaler::Div16;
456            SimplePwm::new_1ch(p.PWM0, p.P0_25, &cfg)
457        };
458        let buzzer_enable = Output::new(p.P1_05, Level::Low, OutputDrive::Standard);
459        spawner.spawn(buzzer_task(buzzer_pwm, buzzer_enable).unwrap());
460        // Boot chirp — independent of USB, so headless boots also signal life.
461        umsh_bsp_t1000e::BUZZER_SIGNAL.signal(&buzzer_melodies::POWER_ON);
462
463        // ── NVMC storage ─────────────────────────────────────────────────────
464        // 64 KB at 0xE4000..0xF4000 (top of app window, per memory.x).
465        let storage: &'static NvmcStorage = STORAGE.init(NvmcStorage::new(Nvmc::new(p.NVMC)));
466
467        // ── Local identity ────────────────────────────────────────────────────
468        // The hardware-TRNG RNG built here is the single RNG path for this
469        // firmware — used for first-boot identity generation AND passed
470        // ownership-by-value into `Mac::new` below as `Platform::Rng`.
471        //
472        // Load identity from flash on subsequent boots; TRNG-generate on
473        // first boot. We do NOT fall back to any PRNG on failure — a
474        // predictable long-term key is worse than panicking.
475        let mut rng = Nrf52840Rng::new(p.RNG);
476        let sk_bytes: [u8; 32] = match storage.load_sk().await {
477            Ok(Some(sk)) => sk,
478            Ok(None) => {
479                let mut sk = [0u8; 32];
480                rng.fill_bytes(&mut sk);
481                storage
482                    .store_sk(&sk)
483                    .await
484                    .unwrap_or_else(|_| panic!("identity persist"));
485                sk
486            }
487            Err(_) => panic!("storage init failed"),
488        };
489        let identity = SoftwareIdentity::from_secret_bytes(&sk_bytes);
490        let local_key = *identity.public_key();
491
492        // ── LR1110 LoRa radio ─────────────────────────────────────────────────
493        // Pin map (confirmed against MeshCore variants/t1000-e):
494        //   SPI bus: SCK=P0.11, MISO=P1.08, MOSI=P1.09 (TWISPI0)
495        //   CS=P0.12, RST=P1.10, IRQ/DIO1=P1.01, BUSY=P0.07
496        //   DIO3: 1.6 V TCXO control (handled by lora-phy SetDIO3AsTCXOCtrl)
497        //   DIO5-8: internal RF switch (handled by RfSwitchConfig via SetDioAsRfSwitch)
498        let t_frame_ms = umsh_radio_loraphy::airtime_ms(
499            SpreadingFactor::_7,
500            Bandwidth::_62KHz,
501            umsh_radio_loraphy::MAX_PAYLOAD,
502        );
503        {
504            let mut cfg = SpimConfig::default();
505            cfg.frequency = Frequency::M8;
506            let radio_bus = Spim::new(
507                p.TWISPI0, Irqs, p.P0_11, // SCK
508                p.P1_08, // MISO
509                p.P1_09, // MOSI
510                cfg,
511            );
512            let radio_cs = Output::new(p.P0_12, Level::High, OutputDrive::Standard);
513            let radio_spi = ExclusiveDevice::new(radio_bus, radio_cs, Delay).unwrap();
514
515            // Pull::Down on DIO1: the LR1110's IRQ output is push-pull active-high,
516            // so a pull-down prevents floating reads when IRQ is de-asserted.
517            let radio_interrupt = Input::new(p.P1_01, Pull::Down);
518            let radio_busy = Input::new(p.P0_07, Pull::None);
519
520            let iv = GenericLr1110InterfaceVariant::new(
521                radio_rst,
522                radio_interrupt,
523                radio_busy,
524                None, // rf_switch_rx: not external — DIO5-8 handle it internally
525                None, // rf_switch_tx: same
526            )
527            .unwrap_or_else(|_| panic!("lr1110 iv"));
528
529            let lora_config = LoraConfig {
530                // HP PA — SetTx will route through tx_hp (0x0A = DIO6+DIO8)
531                // on our RF-switch table. Combined with TX_POWER_DBM=22 this
532                // is the maximum output the chip + board can produce.
533                chip: Lr1110Chip::with_pa(PaSelection::Hp),
534                tcxo_ctrl: Some(TcxoCtrlVoltage::Ctrl1V6),
535                use_dcdc: false, // T1000-E module has no external inductor for BST
536                rx_boost: true,
537                rf_switch: Some(RF_SWITCH),
538            };
539
540            // enable_public_network=false → private sync word 0x1424,
541            // matching MeshCore's RADIOLIB_SX126X_SYNC_WORD_PRIVATE = 0x12.
542            let mut lora = LoRa::new(Lr1110::new(radio_spi, iv, lora_config), false, Delay)
543                .await
544                .unwrap_or_else(|_| panic!("radio init"));
545
546            // MeshCore-US on-air parameters tuned for LR1110.
547            //
548            // Phase 2.5 RX bringup proved that preamble_length=16 (matching
549            // MeshCore's RadioLib TX preamble) reliably triggers
550            // SyncWordHeaderValid → RxDone on the LR1110. The shared
551            // `umsh_radio_loraphy::meshcore_us_params` helper uses 8 for RX,
552            // which is fine for the SX1262 on T-Echo but loses packets on
553            // LR1110. We pin both rx and tx to 16 here to match what worked
554            // on real hardware.
555            let mdltn = lora
556                .create_modulation_params(
557                    SpreadingFactor::_7,
558                    Bandwidth::_62KHz,
559                    CodingRate::_4_5,
560                    910_525_000,
561                )
562                .unwrap_or_else(|_| panic!("modulation params"));
563            let rx_pkt = lora
564                .create_rx_packet_params(
565                    16,    // preamble length: LR1110 needs 16 for MeshCore-US
566                    false, // explicit header
567                    255,   // max payload
568                    true,  // CRC on
569                    false, // IQ normal
570                    &mdltn,
571                )
572                .unwrap_or_else(|_| panic!("rx packet params"));
573            let tx_pkt = lora
574                .create_tx_packet_params(
575                    16,    // preamble length: matches MeshCore RadioLib
576                    false, // explicit header
577                    true,  // CRC on
578                    false, // IQ normal
579                    &mdltn,
580                )
581                .unwrap_or_else(|_| panic!("tx packet params"));
582
583            spawner.spawn(radio_runner_task(lora, mdltn, rx_pkt, tx_pkt).unwrap());
584        }
585
586        // ── MAC coordinator ───────────────────────────────────────────────────
587        let radio_handle = umsh_radio_loraphy::LoraphyRadio::new(&RADIO_CH, t_frame_ms);
588        let crypto = CryptoEngine::new(SoftwareAes, SoftwareSha256);
589        let mut mac = T1000EMac::new(
590            radio_handle,
591            crypto,
592            EmbassyClock,
593            rng,
594            NvmcCounterStore::new(storage),
595            RepeaterConfig::default(),
596            OperatingPolicy::default(),
597        );
598        let identity_id = mac
599            .add_identity(identity)
600            .unwrap_or_else(|_| panic!("identity"));
601        // Restore TX frame-counter boundary so the counter never rewinds.
602        mac.load_persisted_counter(identity_id)
603            .await
604            .unwrap_or_else(|_| panic!("tx counter load"));
605        let mac_cell: &'static AsyncRefCell<T1000EMac> = MAC_CELL.init(AsyncRefCell::new(mac));
606
607        // ── USB stack ─────────────────────────────────────────────────────────
608        let driver = Driver::new(p.USBD, Irqs, HardwareVbusDetect::new(Irqs));
609
610        let mut config = Config::new(0x2886, 0x0057);
611        config.manufacturer = Some("Seeed");
612        config.product = Some("T1000-E UMSH CLI");
613        config.serial_number = Some("umsh-t1000e");
614        config.max_power = 100;
615        config.max_packet_size_0 = 64;
616
617        static CONFIG_DESC: StaticCell<[u8; 256]> = StaticCell::new();
618        static BOS_DESC: StaticCell<[u8; 256]> = StaticCell::new();
619        static MSOS_DESC: StaticCell<[u8; 0]> = StaticCell::new();
620        static CONTROL_BUF: StaticCell<[u8; 64]> = StaticCell::new();
621        static STATE: StaticCell<State> = StaticCell::new();
622
623        let mut builder = Builder::new(
624            driver,
625            config,
626            CONFIG_DESC.init([0; 256]),
627            BOS_DESC.init([0; 256]),
628            MSOS_DESC.init([0; 0]),
629            CONTROL_BUF.init([0; 64]),
630        );
631
632        let class = CdcAcmClass::new(&mut builder, STATE.init(State::new()), 64);
633        let mut usb = builder.build();
634
635        let (tx, raw_rx, ctrl) = class.split_with_control();
636        let rx = CdcAcmRescue::new(raw_rx, ctrl);
637
638        // ── Battery ADC ───────────────────────────────────────────────────────
639        // P0.02 = AIN0 via 2:1 divider; sensor rail P1.06 gates the path.
640        let sensor_rail = Output::new(p.P1_06, Level::Low, OutputDrive::Standard);
641        let saadc = Saadc::new(
642            p.SAADC,
643            Irqs,
644            SaadcConfig::default(), // 12-bit, no oversample
645            [ChannelConfig::single_ended(p.P0_02)],
646        );
647
648        // ── Host + LocalNode ──────────────────────────────────────────────────
649        // Build the Host and add the local identity's node here in main() so
650        // we can clone the node for the beacon task before moving Host into
651        // mac_task. The Host's internal node store and the cloned LocalNode
652        // share Rc state, so events route correctly regardless of which task
653        // holds which copy.
654        let handle = MacHandle::new(mac_cell);
655        let mut host: T1000EHost = Host::new(handle);
656        let node = host.add_node(identity_id);
657        let beacon_node = node.clone();
658
659        // Register persisted peers and channels into the MAC at boot, before
660        // spawning tasks. The CLI task must not be the first thing that
661        // registers these — it only runs after a host opens the CDC port, and
662        // the MAC needs the keys from the very first packet.
663        {
664            let mut peer_buf: heapless::Vec<([u8; 32], Option<heapless::String<16>>), 8> =
665                heapless::Vec::new();
666            let _ = storage.load_all_peers(&mut peer_buf).await;
667            let mut ch_buf: heapless::Vec<(heapless::String<16>, [u8; 32]), 2> =
668                heapless::Vec::new();
669            let _ = storage.load_all_channels(&mut ch_buf).await;
670            for (pk, _alias) in peer_buf.iter() {
671                let _ = node.peer(PublicKey(*pk)).await;
672            }
673            for (name, key_bytes) in ch_buf.iter() {
674                let channel =
675                    umsh_node::Channel::private(umsh_core::ChannelKey(*key_bytes), name.as_str());
676                let _ = node.join(&channel).await;
677            }
678        }
679
680        // Restore RX counter boundaries before the MAC starts processing
681        // packets so the replay window starts above the last accepted frame.
682        // Runs after the peer registration above so the persisted boundaries
683        // actually land on registered peers.
684        MacHandle::new(mac_cell)
685            .load_all_persisted_rx_counters()
686            .await
687            .ok();
688
689        spawner.spawn(output_task(tx).unwrap());
690        spawner.spawn(button_task(button).unwrap());
691        spawner.spawn(shutdown_task().unwrap());
692        spawner.spawn(power_task(saadc, sensor_rail).unwrap());
693        spawner.spawn(mac_task(host).unwrap());
694        spawner.spawn(beacon_task(beacon_node).unwrap());
695        spawner
696            .spawn(cli_task(node, local_key, storage, rx, prev_panic_buf, prev_panic_len).unwrap());
697
698        join(usb.run(), heartbeat(led, wdt_handle)).await;
699    }
700}
