18/// Signal-to-noise ratio represented in centibels (0.1 dB units).
19///
20/// This uses a slightly finer unit than whole decibels while still staying
21/// compact and integer-friendly. Some common LoRa radios report SNR in
22/// quarter-dB steps. Converting those readings into centibels requires
23/// rounding, introducing at most 0.5 cB (0.05 dB) of error.
24#[derive(Clone, Copy, Default, PartialEq, Eq, PartialOrd, Ord, Hash)]
25pub struct Snr(i16);
26
27impl Snr {
28    /// Construct an SNR value directly from centibels.
29    pub const fn from_centibels(centibels: i16) -> Self {
30        Self(centibels)
31    }
32
33    /// Construct an SNR value from whole decibels.
34    pub const fn from_decibels(db: i8) -> Self {
35        Self((db as i16) * 10)
36    }
37
38    /// Construct an SNR value from quarter-dB steps, rounding to the nearest
39    /// centibel.
40    pub const fn from_quarter_db_steps(steps: i16) -> Self {
41        let scaled = steps * 25;
42        let rounded = if scaled >= 0 {
43            (scaled + 5) / 10
44        } else {
45            (scaled - 5) / 10
46        };
47        Self(rounded)
48    }
49
50    /// Return the stored value in centibels.
51    pub const fn as_centibels(self) -> i16 {
52        self.0
53    }
54
55    /// Return the stored value in decibels.
56    pub const fn as_decibels(self) -> i16 {
57        self.0 / 10
58    }
59}
60
61impl core::fmt::Display for Snr {
62    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
63        let db = self.as_decibels();
64        let cdb = self.0 % 10;
65        write!(f, "{}.{:01}dB", db, cdb)
66    }
67}
68
69impl core::fmt::Debug for Snr {
70    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
71        write!(f, "Snr({self})")
72    }
73}

88/// Channel-activity-detection (CAD) policy applied before a transmit.
89#[derive(Clone, Copy, Debug, Default, Eq, PartialEq)]
90pub enum CadPolicy {
91    /// Skip CAD entirely and transmit immediately.
92    #[default]
93    Skip,
94    /// Perform CAD once and transmit only if the channel is currently clear.
95    /// Equivalent to a retry budget of zero.
96    Gate,
97    /// Retry CAD until the channel is clear or `timeout_ms` elapses.

99}

117/// Half-duplex radio abstraction used by the MAC coordinator.
118pub trait Radio {
119    type Error;
120
121    /// Transmit a complete raw UMSH frame.
122    async fn transmit(
123        &mut self,
124        data: &[u8],
125        options: TxOptions,
126    ) -> Result<(), TxError<Self::Error>>;
127
128    /// Poll reception of one frame into `buf`.
129    ///
130    /// `Poll::Pending` means no frame is currently available right now. The
131    /// call does not reserve any receive state; a later poll after transmit
132    /// completion can resume probing immediately.
133    fn poll_receive(
134        &mut self,
135        cx: &mut Context<'_>,
136        buf: &mut [u8],
137    ) -> Poll<Result<RxInfo, Self::Error>>;
138
139    /// Return the largest supported raw frame size.
140    fn max_frame_size(&self) -> usize;
141    /// Return the approximate airtime for a maximum-length frame.
142    fn t_frame_ms(&self) -> u32;
143}

145/// Monotonic millisecond clock.
146pub trait Clock {
147    /// Return milliseconds since an arbitrary monotonic epoch.
148    fn now_ms(&self) -> u64;
149
150    /// Poll a delay that completes when the monotonic clock reaches `deadline_ms`.
151    ///
152    /// Returns `Poll::Ready(())` if the deadline has already passed. Otherwise
153    /// the implementation MUST register `cx.waker()` with a platform timer and
154    /// return `Poll::Pending`, so the task is woken when the deadline elapses.
155    ///
156    /// The default implementation has no real timer: it schedules an immediate
157    /// re-poll (via `cx.waker().wake_by_ref()`) and returns `Poll::Pending`, causing
158    /// the caller to busy-poll until the monotonic clock reaches the deadline.
159    /// This is correct but wastes CPU; platform clocks backed by a real timer
160    /// (tokio, embassy, etc.) MUST override this to sleep efficiently. A default
161    /// that returned `Ready` unconditionally would spin just as hard; one that
162    /// returned `Pending` without waking would stall timer-driven work entirely.
163    fn poll_delay_until(&self, cx: &mut Context<'_>, deadline_ms: u64) -> Poll<()> {
164        let _ = deadline_ms;
165        cx.waker().wake_by_ref();
166        Poll::Pending
167    }
168}

170/// Persistent peer directory for the node layer.
171///
172/// Stores and retrieves peer records keyed by the raw 32-byte Ed25519 public
173/// key. `alias` is an optional UTF-8 label (the implementation may silently
174/// truncate values longer than 16 bytes). Implementors MUST treat `store_peer`
175/// as an upsert — calling it twice for the same key overwrites the record.
176pub trait PeerStore {
177    type Error;
178
179    /// Upsert the peer record for `key`. `alias`, if present, is a UTF-8
180    /// display label; passing `None` removes any existing alias.
181    async fn store_peer(&self, key: &[u8; 32], alias: Option<&[u8]>) -> Result<(), Self::Error>;
182
183    /// Remove the peer record for `key`. A no-op if the key is not present.
184    async fn delete_peer(&self, key: &[u8; 32]) -> Result<(), Self::Error>;
185
186    /// Invoke `f` for every persisted peer record.
187    ///
188    /// `alias` is `None` when no alias was stored for that key. Callers that
189    /// cannot handle an error mid-iteration should collect into a local buffer
190    /// first, then process asynchronously.
191    async fn for_each_peer(
192        &self,
193        f: &mut dyn FnMut(&[u8; 32], Option<&[u8]>),
194    ) -> Result<(), Self::Error>;
195}
196
197/// No-op peer store — use when peer persistence is not needed.
198pub struct NoPeerStore;
199
200impl PeerStore for NoPeerStore {
201    type Error = core::convert::Infallible;
202
203    async fn store_peer(&self, _: &[u8; 32], _: Option<&[u8]>) -> Result<(), Self::Error> {
204        Ok(())
205    }
206
207    async fn delete_peer(&self, _: &[u8; 32]) -> Result<(), Self::Error> {
208        Ok(())
209    }
210
211    async fn for_each_peer(
212        &self,
213        _: &mut dyn FnMut(&[u8; 32], Option<&[u8]>),
214    ) -> Result<(), Self::Error> {
215        Ok(())
216    }
217}

263/// Persistent frame-counter storage.
264pub trait CounterStore {
265    type Error;
266
267    /// Load the stored counter for `context`, or `0` if missing.
268    async fn load(&self, context: &[u8]) -> Result<u32, Self::Error>;
269    /// Persist a counter value for `context`.
270    async fn store(&self, context: &[u8], value: u32) -> Result<(), Self::Error>;
271    /// Flush any buffered state to durable storage.

299/// Optional power-control hook for higher layers (e.g. the CLI).
300///
301/// Implementations request a controlled shutdown — the actual sequencing
302/// (display, storage flush, GPIO sense, entering System OFF, etc.) lives
303/// in the firmware that owns those peripherals. This call MUST return
304/// promptly; it's typically a `Signal::signal(())` to a shutdown task.
305pub trait PowerControl {
306    fn request_power_off(&self);
307
308    /// Request a soft reboot. The default implementation is a no-op; targets
309    /// without a wired reboot path can leave it as such. Like
310    /// [`request_power_off`](Self::request_power_off), this MUST return
311    /// promptly — typically a `Signal::signal(())` to a reboot task that
312    /// performs any final flushes before triggering a system reset.
313    fn request_reboot(&self) {}
314}
315
316/// No-op power control — use when shutdown is not implemented for the target.
317pub struct NoPowerControl;
318
319impl PowerControl for NoPowerControl {
320    fn request_power_off(&self) {}
321}
