1// LilyGO T-Echo bringup firmware with an interactive UMSH CLI on USB-CDC.
2//
3// Boot sequence:
4//   1. Bring up the peripheral rail (P0.12 HIGH).
5//   2. Arm the watchdog (8 s timeout, petted by the heartbeat task).
6//   3. Spawn the display task — initial boot screen ("UMSH bringup" + git
7//      short SHA + "MAC: 0") plus subsequent count-update refreshes.
8//   4. Initialize the SX1262 LoRa radio (MeshCore US settings) and spawn
9//      the radio runner task.
10//   5. Build a `Mac<TechoPlatform>`, park it in a `'static AsyncRefCell`,
11//      and spawn `umsh_task` which drives `Host::run` + `CliSession::run`
12//      concurrently over the shared `MacHandle`.
13//   6. Spawn `output_task` to own the USB `Sender` and drain `OUTPUT_CH`.
14//   7. Join usb.run / heartbeat in main; the CLI runs in spawned tasks.
15//
16// Task layout (steady state):
17//   - main():               joins usb.run / heartbeat
18//   - display_task:         renders the e-paper on count changes
19//   - radio_runner_task:    owns lora_phy::LoRa, RX/TX state machine
20//   - umsh_task:            host.run() + cli.run(), shares the MAC via MacHandle
21//   - output_task:          owns the USB Sender, drains OUTPUT_CH
22//
23// USB CDC flow control is preserved by the output_task / OUTPUT_CH split:
24// nothing blocks CdcInput::read_packet on TX progress, so the host's bulk
25// OUT NAK / retry mechanism handles backpressure correctly during pastes.
26//
27// Safety primitives inherited from the BSP (see umsh-bsp-nrf52840):
28//   * Panic capture into reserved RAM, dumped over USB on the next boot.
29//   * 1200-baud touchless reset and Ctrl-C × 3 + "dfu" escape to bootloader
30//     (baked into CdcAcmRescue).
31//   * Watchdog.
32

253    /// Drains `IDENTITY_SIGNAL` and persists received `NodeIdentityPayload`
254    /// bytes for known peers. Runs independently of the MAC/CLI task so that
255    /// NVMC writes (which stall the CPU for ~85 ms) don't affect radio timing.
256    #[embassy_executor::task]
257    async fn identity_persist_task(storage: &'static NvmcStorage) {
258        loop {
259            let (pk, payload, len) = IDENTITY_SIGNAL.wait().await;
260            if storage.peer_exists(&pk).await.unwrap_or(false) {
261                let _ = storage.update_peer_identity(&pk, &payload[..len]).await;
262            }
263        }
264    }

266    /// Drives the MAC coordinator and owns the identity-relay subscription.
267    /// Independent of USB so radio RX/TX and the MAC pump (including ping
268    /// auto-replies) keep running whether or not a host terminal is attached.
269    #[embassy_executor::task]
270    async fn mac_task(mut host: TechoHost, identity_id: LocalIdentityId) {
271        // Subscribe to raw packets so NodeIdentity payloads from known peers
272        // can be relayed to identity_persist_task for durable storage.
273        // The guard must remain live for the duration of the task.
274        let sub_node = host.node(identity_id).expect("node just added");
275        let _identity_sub = sub_node.on_receive(|pkt| {
276            if pkt.payload_type() != PayloadType::NodeIdentity { return false; }
277            let Some(from) = pkt.from_key() else { return false; };
278            let raw = pkt.payload();
279            let len = raw.len().min(256);
280            let mut buf = [0u8; 256];
281            buf[..len].copy_from_slice(&raw[..len]);
282            IDENTITY_SIGNAL.signal((from.0, buf, len));
283            false // don't consume — let other handlers see it too
284        });
285
286        let _ = host.run().await;
287        panic!("host exited");
288    }

290    /// Runs the `CliSession` over USB-CDC. The only task that blocks on a host
291    /// terminal connection — the radio, MAC pump, and identity relay all run
292    /// without it.
293    #[embassy_executor::task]
294    async fn cli_task(
295        node: TechoNode,
296        local_key: PublicKey,
297        storage: &'static NvmcStorage,
298        rx: TechoRescue,
299        prev_panic_buf: &'static [u8; 256],
300        prev_panic_len: usize,
301    ) {
302        use umsh_cli::CliSession;
303        use umsh_cli::io::CliOutput;
304        use umsh_cli::logger::NullLogger;
305
306        let mut input = cli_io::CdcInput::new(rx);
307        let mut out = cli_io::CdcOutput::new();
308
309        // Wait for the host to open the CDC port before writing the banner.
310        input.wait_connection().await;
311
312        let _ = out.write_line("").await;
313        let _ = out.write_line("UMSH CLI (T-Echo)").await;
314        let _ = out.write_line("type /help for commands").await;
315        if prev_panic_len > 0 {
316            let _ = out.write_line("[PREV PANIC]:").await;
317            if let Ok(s) = core::str::from_utf8(&prev_panic_buf[..prev_panic_len]) {
318                let _ = out.write_line(s).await;
319            }
320        }
321
322        let peer_store    = NvmcPeerStore::new(storage);
323        let channel_store = NvmcChannelStore::new(storage);
324        let mut cli: CliSession<_, _, _, _, _, _, 4, 4, 2, 8, 128> = CliSession::new(
325            node,
326            local_key,
327            out,
328            NullLogger::new(),
329            peer_store,
330            channel_store,
331            PowerSignaler,
332        );
333
334        // `run` loads peers/channels from storage and registers them with the
335        // MAC (idempotent) and the CLI display tables before entering the loop.
336        let _ = cli.run(&mut input).await;
337        panic!("cli exited");
338    }

340    /// Owns the e-paper SPI bus and pins. Renders the boot screen on
341    /// startup, then waits for `DISPLAY_COUNT_SIGNAL` and re-renders with
342    /// the latest count.
343    ///
344    /// Full refresh (with flashing) per update; partial refresh on this
345    /// panel requires RED-RAM previous-frame tracking which is a separate
346    /// change. `DISPLAY_THROTTLE` caps the visible refresh rate.
347    #[embassy_executor::task]
348    async fn display_task(
349        mut spi:  Spim<'static>,
350        mut cs:   Output<'static>,
351        mut dc:   Output<'static>,
352        mut rst:  Output<'static>,
353        mut busy: Input<'static>,
354    ) {
355        use core::fmt::Write as _;
356        use embedded_graphics::geometry::Point;
357        use embedded_graphics::mono_font::ascii::FONT_10X20;
358        use embedded_graphics::mono_font::MonoTextStyle;
359        use embedded_graphics::pixelcolor::BinaryColor;
360        use embedded_graphics::text::{Baseline, Text};
361        use embedded_graphics::Drawable;
362        use heapless::String;
363
364        let sha   = env!("GIT_SHORT_SHA");
365        let style = MonoTextStyle::new(&FONT_10X20, BinaryColor::On);
366
367        // Fill `buf` with a frame containing the boot text and the supplied count.
368        let mut buf = [0xFFu8; display::BUF_SIZE];
369        let render = |buf: &mut [u8; display::BUF_SIZE], count: u32| {
370            buf.fill(0xFF);  // all-white background
371            let mut fb = display::EpdFb(buf);
372
373            // Center each line by its glyph count.
374            let center_x = |text: &str| (display::WIDTH as i32 - text.len() as i32 * FONT_W) / 2;
375
376            let title = "UMSH bringup";
377            let _ = Text::with_baseline(title, Point::new(center_x(title), TITLE_Y), style, Baseline::Top).draw(&mut fb);
378            let _ = Text::with_baseline(sha,   Point::new(center_x(sha),   SHA_Y),   style, Baseline::Top).draw(&mut fb);
379
380            let mut count_str: String<16> = String::new();
381            let _ = write!(count_str, "MAC: {}", count);
382            let _ = Text::with_baseline(&count_str, Point::new(center_x(&count_str), COUNT_Y), style, Baseline::Top).draw(&mut fb);
383        };
384
385        // Renders centred lines (one per slice element) onto an all-white frame.
386        let render_lines = |buf: &mut [u8; display::BUF_SIZE], lines: &[(&str, i32)]| {
387            buf.fill(0xFF);
388            let mut fb = display::EpdFb(buf);
389            for (text, y) in lines {
390                let cx = (display::WIDTH as i32 - text.len() as i32 * FONT_W) / 2;
391                let _ = Text::with_baseline(text, Point::new(cx, *y), style, Baseline::Top)
392                    .draw(&mut fb);
393            }
394        };
395
396        // Initial boot screen (count = 0).
397        render(&mut buf, 0);
398        display::init(&mut spi, &mut cs, &mut dc, &mut rst, &mut busy).await;
399        display::render(&mut spi, &mut cs, &mut dc, &mut busy, &buf).await;
400
401        // Update loop. Races count updates against the shutdown signal.
402        // We deliberately do NOT reset DISPLAY_COUNT_SIGNAL after the
403        // throttle: any packet that fired during render+throttle stays
404        // pending, so the next iteration starts immediately with the
405        // newest count. Throttle still caps the refresh rate.
406        loop {
407            match select(DISPLAY_COUNT_SIGNAL.wait(), DISPLAY_SHUTDOWN_SIGNAL.wait()).await {
408                Either::First(()) => {
409                    let count = PACKET_COUNT.load(Ordering::Relaxed);
410                    render(&mut buf, count);
411                    display::render(&mut spi, &mut cs, &mut dc, &mut busy, &buf).await;
412                    Timer::after(DISPLAY_THROTTLE).await;
413                }
414                Either::Second(()) => {
415                    // Final frame, then deep sleep (RAM-retaining; the panel
416                    // wakes via hardware reset on the next boot).
417                    render_lines(&mut buf, &[("Powered off", 100)]);
418                    display::render(&mut spi, &mut cs, &mut dc, &mut busy, &buf).await;
419                    display::sleep(&mut spi, &mut cs, &mut dc).await;
420                    DISPLAY_SHUTDOWN_DONE.signal(());
421                    // Park forever; the shutdown task will System OFF shortly.
422                    core::future::pending::<()>().await;
423                }
424            }
425        }
426    }

428    /// Long-press watcher for the user button on P1.10 (active-low, pull-up).
429    /// Two-second hold fires [`SHUTDOWN_SIGNAL`]. Releases before 2 s are
430    /// ignored — there's no short-press action defined yet.
431    #[embassy_executor::task]
432    async fn button_task(mut button: Input<'static>) {
433        const HOLD: Duration = Duration::from_secs(2);
434        loop {
435            button.wait_for_low().await;
436            match select(button.wait_for_high(), Timer::after(HOLD)).await {
437                Either::First(()) => {
438                    // Released before HOLD — no-op.
439                }
440                Either::Second(()) => {
441                    SHUTDOWN_SIGNAL.signal(());
442                    // Wait for release so we don't keep re-triggering.
443                    button.wait_for_high().await;
444                }
445            }
446        }
447    }

449    /// Orchestrates the controlled power-off:
450    ///   1. tell the display to render the final frame and sleep,
451    ///   2. flush any pending TX frame-counter reservations (RX counters are
452    ///      drained on every `next_event` in the parallel host task),
453    ///   3. wait for the display task to acknowledge (cap at 5 s),
454    ///   4. drop the peripheral power rail (P0.12) so LoRa / sensors / GNSS
455    ///      lose power before the chip parks,
456    ///   5. configure user-button DETECT-low and enter System OFF.
457    ///
458    /// Diverges via [`power_off`].
459    #[embassy_executor::task]
460    async fn shutdown_task(
461        mac_cell: &'static AsyncRefCell<TechoMac>,
462        peripheral_power: Output<'static>,
463    ) -> ! {
464        SHUTDOWN_SIGNAL.wait().await;
465
466        DISPLAY_SHUTDOWN_SIGNAL.signal(());
467
468        let handle = MacHandle::new(mac_cell);
469        let _ = handle.service_counter_persistence().await;
470
471        let _ = select(
472            DISPLAY_SHUTDOWN_DONE.wait(),
473            Timer::after(Duration::from_secs(5)),
474        )
475        .await;
476
477        // Tri-state all peripheral signal pins before cutting power.
478        //
479        // Two reasons:
480        //   1. Output pins driving into an unpowered peripheral leak current
481        //      through ESD diodes back onto its unpowered VCC rail.
482        //   2. Input pins with PIN_CNF SENSE configured by embassy's async
483        //      GPIO layer (e.g. radio DIO1 / BUSY mid-wait) will fire DETECT
484        //      and immediately wake the chip from System OFF.
485        //
486        // tristate_pin() writes PIN_CNF = 0x02 (DIR=input, INPUT=disconnect,
487        // PULL=none, DRIVE=0, SENSE=disabled) — clearing any SENSE bits.
488        //
489        // E-paper SPI bus (SPIM2): SCK=P0.31, MOSI=P1.07, MISO=P0.29
490        // E-paper control:         CS=P0.30, DC=P0.28, RST=P0.02, BUSY=P0.03
491        // Radio SPI bus (TWISPI1): SCK=P0.19, MOSI=P0.22, MISO=P0.23
492        // Radio control:           CS=P0.24, RST=P0.25, BUSY=P0.17, DIO1=P0.20
493        for (port, pin) in [
494            (Port::P0, 31u8), // e-paper SCK
495            (Port::P1,  7u8), // e-paper MOSI
496            (Port::P0, 29u8), // e-paper MISO
497            (Port::P0, 30u8), // e-paper CS
498            (Port::P0, 28u8), // e-paper DC
499            (Port::P0,  2u8), // e-paper RST
500            (Port::P0,  3u8), // e-paper BUSY
501            (Port::P0, 19u8), // radio SCK
502            (Port::P0, 22u8), // radio MOSI
503            (Port::P0, 23u8), // radio MISO
504            (Port::P0, 24u8), // radio CS
505            (Port::P0, 25u8), // radio RST
506            (Port::P0, 17u8), // radio BUSY
507            (Port::P0, 20u8), // radio DIO1  ← has SENSE set by async radio wait
508        ] {
509            tristate_pin(port, pin);
510        }
511
512        // Drop the peripheral rail so the LoRa module, GNSS, sensors, and
513        // e-paper bias generator all lose power before we enter System OFF.
514        drop(peripheral_power);
515
516        // P1.10 is the side user button. Active-low, pull-up → DETECT-low wakes.
517        power_off(&[WakePin { port: Port::P1, pin: 10, sense: WakeSense::Low }])
518    }

520    // ─── Main ────────────────────────────────────────────────────────────────
521
522    #[embassy_executor::main]
523    async fn main(spawner: Spawner) {
524        // Initialize the heap allocator before any alloc-using code runs.
525        // 4 KiB is negligible on nRF52840 (256 KiB RAM); actual runtime
526        // alloc usage is near-zero since we don't create a MacHandle.
527        {
528            use core::mem::MaybeUninit;
529            const HEAP_SIZE: usize = 8192;
530            static mut HEAP: [MaybeUninit<u8>; HEAP_SIZE] = [MaybeUninit::uninit(); HEAP_SIZE];
531            unsafe { crate::ALLOCATOR.init(core::ptr::addr_of!(HEAP) as usize, HEAP_SIZE) }
532        }
533
534        let p = embassy_nrf::init(umsh_bsp_nrf52840::clocks::default_config());
535
536        // Peripheral power enable (P0.12). Must be high before display, LoRa,
537        // or GNSS is addressed, including on battery power. Ownership is later
538        // transferred to `shutdown_task` so it can drop the rail before entering
539        // System OFF.
540        let peripheral_power = Output::new(p.P0_12, Level::High, OutputDrive::Standard);
541
542        // WDT: 8 s timeout, petted by the heartbeat task every ~2 s.
543        let mut wdt_config = WdtConfig::default();
544        wdt_config.timeout_ticks = 32768 * 8;
545        let (_wdt, [wdt_handle]) =
546            Watchdog::try_new::<_, 1>(p.WDT, wdt_config).unwrap_or_else(|_| panic!("wdt"));
547
548        // Pick up any panic message left by the previous boot.
549        static PREV_PANIC_BUF: StaticCell<[u8; 256]> = StaticCell::new();
550        let mut prev_panic_tmp = [0u8; 256];
551        let prev_panic_len = {
552            let mut slot = PanicSlot::new(super::panic::panic_region());
553            if let Some(msg) = slot.read() {
554                let n = msg.len().min(prev_panic_tmp.len());
555                prev_panic_tmp[..n].copy_from_slice(&msg[..n]);
556                slot.clear();
557                n
558            } else {
559                0
560            }
561        };
562        let prev_panic_buf: &'static [u8; 256] = PREV_PANIC_BUF.init(prev_panic_tmp);
563
564        // ── E-paper display task ──────────────────────────────────────────────
565        // P1.11 is the e-paper backlight on this module; drive it LOW
566        // explicitly so leakage / external pullups can't turn it on.
567        let _backlight = Output::new(p.P1_11, Level::Low, OutputDrive::Standard);
568        {
569            let mut cfg = SpimConfig::default();
570            cfg.frequency = Frequency::M4;
571            let disp_spi  = Spim::new(p.SPI2, Irqs, p.P0_31, p.P1_07, p.P0_29, cfg);
572            let disp_cs   = Output::new(p.P0_30, Level::High, OutputDrive::Standard);
573            let disp_dc   = Output::new(p.P0_28, Level::Low,  OutputDrive::Standard);
574            let disp_rst  = Output::new(p.P0_02, Level::High, OutputDrive::Standard);
575            let disp_busy = Input::new(p.P0_03, Pull::None);
576            spawner.spawn(display_task(disp_spi, disp_cs, disp_dc, disp_rst, disp_busy).unwrap());
577        }
578
579        // ── SX1262 LoRa radio ────────────────────────────────────────────────
580        // Pin assignment (T-Echo hardware, firmware-confirmed):
581        //   SPI bus: SCK=P0.19, MOSI=P0.22, MISO=P0.23 (TWISPI1)
582        //   CS=P0.24, RST=P0.25, BUSY=P0.17, DIO1=P0.20
583        //   DIO2: internal RF switch (lora-phy sends SetDIO2AsRfSwitchCtrl).
584        //   DIO3: 1.8 V TCXO (lora-phy sends SetDIO3AsTcxoCtrl).
585        let t_frame_ms = umsh_radio_loraphy::airtime_ms(
586            SpreadingFactor::_7,
587            Bandwidth::_62KHz,
588            umsh_radio_loraphy::MAX_PAYLOAD,
589        );
590        {
591            let mut cfg = SpimConfig::default();
592            // SX1262 datasheet §8.2: max SCK = 16 MHz, Mode 0 (CPOL=0, CPHA=0).
593            cfg.frequency = Frequency::M16;
594            let radio_bus = Spim::new(
595                p.TWISPI1, Irqs,
596                p.P0_19,  // SCK
597                p.P0_23,  // MISO
598                p.P0_22,  // MOSI
599                cfg,
600            );
601            let radio_cs  = Output::new(p.P0_24, Level::High, OutputDrive::Standard);
602            let radio_spi = ExclusiveDevice::new(radio_bus, radio_cs, Delay).unwrap();
603
604            let radio_rst  = Output::new(p.P0_25, Level::High, OutputDrive::Standard);
605            let radio_dio1 = Input::new(p.P0_20, Pull::None);
606            let radio_busy = Input::new(p.P0_17, Pull::None);
607
608            let iv = GenericSx126xInterfaceVariant::new(
609                radio_rst,
610                radio_dio1,
611                radio_busy,
612                None,   // rf_switch_rx: DIO2 wired internally on the T-Echo module
613                None,   // rf_switch_tx: same
614            ).unwrap();
615
616            let lora_config = LoraConfig {
617                chip: Sx1262,
618                tcxo_ctrl: Some(TcxoCtrlVoltage::Ctrl1V8),  // DIO3 → 1.8 V TCXO
619                use_dcdc: true,   // T-Echo SX1262 module has DC-DC converter
620                rx_boost: true,   // boosted LNA gain per MeshCore SX126X_RX_BOOSTED_GAIN=1
621            };
622
623            // enable_public_network=false → sync word 0x1424 (private),
624            // matching MeshCore's RADIOLIB_SX126X_SYNC_WORD_PRIVATE = 0x12.
625            let mut lora = LoRa::new(Sx126x::new(radio_spi, iv, lora_config), false, Delay)
626                .await
627                .unwrap_or_else(|_| panic!("radio init"));
628
629            let (mdltn, rx_pkt, tx_pkt) = umsh_radio_loraphy::meshcore_us_params(&mut lora)
630                .unwrap_or_else(|_| panic!("radio params"));
631
632            spawner.spawn(radio_runner_task(lora, mdltn, rx_pkt, tx_pkt).unwrap());
633        }
634
635        // ── NV storage ────────────────────────────────────────────────────────
636        let storage: &'static NvmcStorage =
637            STORAGE.init(NvmcStorage::new(Nvmc::new(p.NVMC)));
638
639        // ── MAC coordinator ───────────────────────────────────────────────────
640        // The hardware-TRNG RNG built here is the single RNG path for this
641        // firmware — used for first-boot identity generation AND passed
642        // ownership-by-value into `Mac::new` below as `Platform::Rng`.
643        //
644        // Load identity from flash on subsequent boots; TRNG-generate on
645        // first boot. We do NOT fall back to any PRNG on failure — a
646        // predictable long-term key is worse than refusing to start.
647        let mut rng = Nrf52840Rng::new(p.RNG);
648        let sk_bytes: [u8; 32] = match storage.load_sk().await {
649            Ok(Some(sk)) => sk,
650            Ok(None) => {
651                let mut sk = [0u8; 32];
652                rng.fill_bytes(&mut sk);
653                storage.store_sk(&sk).await.unwrap_or_else(|_| panic!("identity persist"));
654                sk
655            }
656            Err(_) => panic!("storage init failed"),
657        };
658        let identity  = SoftwareIdentity::from_secret_bytes(&sk_bytes);
659        let local_key = *identity.public_key();
660
661        let radio_handle = umsh_radio_loraphy::LoraphyRadio::new(&RADIO_CH, t_frame_ms);
662        let crypto       = CryptoEngine::new(SoftwareAes, SoftwareSha256);
663        let mut mac = TechoMac::new(
664            radio_handle,
665            crypto,
666            EmbassyClock,
667            rng,
668            NvmcCounterStore::new(storage),
669            RepeaterConfig::default(),
670            OperatingPolicy::default(),
671        );
672        let identity_id = mac.add_identity(identity).unwrap_or_else(|_| panic!("identity"));
673        // Restore the TX frame-counter boundary so the counter never rewinds.
674        mac.load_persisted_counter(identity_id)
675            .await
676            .unwrap_or_else(|_| panic!("tx counter load"));
677        let mac_cell: &'static AsyncRefCell<TechoMac> =
678            MAC_CELL.init(AsyncRefCell::new(mac));
679
680        // ── Host + node + boot-time peer/channel registration ─────────────────
681        // Build the Host/node here so the MAC pump (`mac_task`) is independent
682        // of USB, and register persisted peer/channel keys into the MAC now —
683        // not from the CLI task, which only runs after a host opens the CDC
684        // port. Without this the coordinator had no keys until a serial client
685        // attached, so it couldn't authenticate inbound secure frames and
686        // silently dropped every ping.
687        let handle = MacHandle::new(mac_cell);
688        let mut host: TechoHost = Host::new(handle);
689        let node = host.add_node(identity_id);
690
691        {
692            let mut peer_buf: heapless::Vec<([u8; 32], Option<heapless::String<16>>), 8> =
693                heapless::Vec::new();
694            let _ = storage.load_all_peers(&mut peer_buf).await;
695            let mut ch_buf: heapless::Vec<(heapless::String<16>, [u8; 32]), 2> =
696                heapless::Vec::new();
697            let _ = storage.load_all_channels(&mut ch_buf).await;
698            for (pk, _alias) in peer_buf.iter() {
699                let _ = node.peer(PublicKey(*pk)).await;
700            }
701            for (name, key_bytes) in ch_buf.iter() {
702                let channel = Channel::private(ChannelKey(*key_bytes), name.as_str());
703                let _ = node.join(&channel).await;
704            }
705        }
706        // Restore RX counter boundaries after peer registration so the persisted
707        // boundaries land on registered peers.
708        MacHandle::new(mac_cell)
709            .load_all_persisted_rx_counters()
710            .await
711            .ok();
712
713        // ── USB stack + steady-state services ────────────────────────────────
714        let led    = Output::new(p.P0_14, Level::High, OutputDrive::Standard);
715        let driver = Driver::new(p.USBD, Irqs, HardwareVbusDetect::new(Irqs));
716
717        let mut config = Config::new(0x16c0, 0x27dd);
718        config.manufacturer      = Some("UMSH");
719        config.product           = Some("T-Echo Bringup");
720        config.serial_number     = Some("companion-cli-techo");
721        config.max_power         = 100;
722        config.max_packet_size_0 = 64;
723
724        static CONFIG_DESC: StaticCell<[u8; 256]> = StaticCell::new();
725        static BOS_DESC:    StaticCell<[u8; 256]> = StaticCell::new();
726        static MSOS_DESC:   StaticCell<[u8; 0]>   = StaticCell::new();
727        static CONTROL_BUF: StaticCell<[u8; 64]>  = StaticCell::new();
728        static STATE:       StaticCell<State>     = StaticCell::new();
729
730        let mut builder = Builder::new(
731            driver,
732            config,
733            CONFIG_DESC.init([0; 256]),
734            BOS_DESC.init([0; 256]),
735            MSOS_DESC.init([0; 0]),
736            CONTROL_BUF.init([0; 64]),
737        );
738
739        let class = CdcAcmClass::new(&mut builder, STATE.init(State::new()), 64);
740        let mut usb = builder.build();
741
742        let (tx, raw_rx, ctrl) = class.split_with_control();
743        let rx = CdcAcmRescue::new(raw_rx, ctrl);
744
745        spawner.spawn(output_task(tx).unwrap());
746        spawner.spawn(identity_persist_task(storage).unwrap());
747        spawner.spawn(mac_task(host, identity_id).unwrap());
748        spawner.spawn(
749            cli_task(node, local_key, storage, rx, prev_panic_buf, prev_panic_len).unwrap(),
750        );
751
752        // User button (P1.10, active-low). Pull-up so DETECT can wake from
753        // System OFF on the falling edge.
754        let button = Input::new(p.P1_10, Pull::Up);
755        spawner.spawn(button_task(button).unwrap());
756        spawner.spawn(shutdown_task(mac_cell, peripheral_power).unwrap());
757
758        join(
759            usb.run(),
760            heartbeat(led, wdt_handle),
761        ).await;
762    }

764    // ─── Heartbeat + WDT pet ─────────────────────────────────────────────────
765
766    async fn heartbeat(mut led: Output<'static>, mut wdt: WatchdogHandle) -> ! {
767        let mut engine = LedEngine::new(LedTimings::default(), Instant::now().as_millis());
768        loop {
769            wdt.pet();
770            let decision = engine.tick(Instant::now().as_millis());
771            // P0.14 is active-low: set_low() = LED on.
772            if decision.on { led.set_low() } else { led.set_high() }
773            Timer::at(Instant::from_millis(decision.next_deadline_ms)).await;
774        }
775    }
